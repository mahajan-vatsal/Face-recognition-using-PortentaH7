Machine Vision using Portenta H7 2
This project aims to develop a face recognition-based access control system using the Arduino Portenta H7 and Vision Shield, leveraging Edge Impulse for machine learning. The system captures facial images, processes them locally using an AI model deployed on the Portenta H7 and determines access based on authorised personnel.

28template <typename IndicesT, typename UpdatesT>
29inline TfLiteStatus ScatterNd(const RuntimeShape& indices_shape,
30 const IndicesT* indices_data,
31 const RuntimeShape& updates_shape,
32 const UpdatesT* updates_data,
33 const RuntimeShape& output_shape,
34 UpdatesT* output_data) {
35 int n_slices = 1;
36 int slice_size = 1;
37 const int outer_dims = indices_shape.DimensionsCount() - 1;
38 const int indices_nd = indices_shape.Dims(outer_dims);
39 const int updates_dims = updates_shape.DimensionsCount();
40 for (int i = 0; i < outer_dims; ++i) {
41 n_slices *= indices_shape.Dims(i);
42 }
43 for (int i = outer_dims; i < updates_dims; ++i) {
44 slice_size *= updates_shape.Dims(i);
45 }
46
47 int output_flat_size = output_shape.FlatSize();
48 int remain_flat_size = output_flat_size;
49 std::vector<int> dims_to_count(indices_nd, 0);
50 for (int i = 0; i < indices_nd; ++i) {
51 dims_to_count[i] = remain_flat_size / output_shape.Dims(i);
52 remain_flat_size = dims_to_count[i];
53 }
54
55 if (n_slices * slice_size > updates_shape.FlatSize()) {
56 return kTfLiteError;
57 }
58 memset(output_data, 0, sizeof(UpdatesT) * output_flat_size);
59 for (int i = 0; i < n_slices; ++i) {
60 int to_pos = 0;
61 for (int j = 0; j < indices_nd; ++j) {
62 IndicesT idx = indices_data[i * indices_nd + j];
63 to_pos += idx * dims_to_count[j];
64 }
65 if (to_pos < 0 || to_pos + slice_size > output_flat_size) {
66 return kTfLiteError;
67 }
68 for (int j = 0; j < slice_size; j++) {
69 output_data[to_pos + j] += updates_data[i * slice_size + j];
70 }
71 }
72 return kTfLiteOk;
73}
