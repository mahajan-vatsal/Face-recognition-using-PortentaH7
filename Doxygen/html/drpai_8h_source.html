122void get_udmabuf_memory_start_addr()
123{ /* Obtain udmabuf memory area starting address */
124
125 int8_t fd = 0;
126 char addr[1024];
127 int32_t read_ret = 0;
128 errno = 0;
129
130 fd = open("/sys/class/u-dma-buf/udmabuf0/phys_addr", O_RDONLY);
131 if (0 > fd)
132 {
133 fprintf(stderr, "[ERROR] Failed to open udmabuf0/phys_addr : errno=%d\n", errno);
134 }
135
136 read_ret = read(fd, addr, 1024);
137 if (0 > read_ret)
138 {
139 fprintf(stderr, "[ERROR] Failed to read udmabuf0/phys_addr : errno=%d\n", errno);
140 close(fd);
141 }
142
143 sscanf(addr, "%lx", &udmabuf_address);
144 close(fd);
145
146 /* Filter the bit higher than 32 bit */
147 udmabuf_address &=0xFFFFFFFF;
148}

150uint8_t drpai_init_mem(uint32_t input_frame_size) {
151 int32_t i = 0;
152
153 int udmabuf_fd0 = open("/dev/udmabuf0", O_RDWR);
154 if (udmabuf_fd0 < 0) {
155 return -1;
156 }
157
158 // input_frame_size === data_in_size
159 uint8_t *addr =
160 (uint8_t *)mmap(NULL, input_frame_size,
161 PROT_READ | PROT_WRITE, MAP_SHARED, udmabuf_fd0, 0);
162
163 drpai_input_buf = addr;
164
165 /* Write once to allocate physical memory to u-dma-buf virtual space.
166 * Note: Do not use memset() for this.
167 * Because it does not work as expected. */
168 for (i = 0; i < input_frame_size; i++) {
169 drpai_input_buf[i] = 0;
170 }
171
172
173 get_udmabuf_memory_start_addr();
174 if (0 == udmabuf_address) {
175 return EI_IMPULSE_DRPAI_INIT_FAILED;
176 }
177
178 return 0;
179}

181/*****************************************
182 * Function Name : read_addrmap_txt
183 * Description : Loads address and size of DRP-AI Object files into struct
184 *addr. Arguments : addr_file = filename of addressmap file (from
185 *DRP-AI Object files) Return value : 0 if succeeded not 0 otherwise
186 ******************************************/
187static int8_t read_addrmap_txt() {
188 // create a stream from the DRP-AI model data without copying
189 std::istringstream ifs;
190 ifs.rdbuf()->pubsetbuf((char *)ei_ei_addrmap_intm_txt, ei_ei_addrmap_intm_txt_len);
191
192 std::string str;
193 unsigned long l_addr;
194 unsigned long l_size;
195 std::string element, a, s;
196
197 if (ifs.fail()) {
198 return -1;
199 }
200
201 while (getline(ifs, str)) {
202 std::istringstream iss(str);
203 iss >> element >> a >> s;
204 l_addr = strtol(a.c_str(), NULL, 16);
205 l_size = strtol(s.c_str(), NULL, 16);
206
207 if (element == "drp_config") {
208 drpai_address.drp_config_addr = l_addr;
209 drpai_address.drp_config_size = l_size;
210 } else if (element == "desc_aimac") {
211 drpai_address.desc_aimac_addr = l_addr;
212 drpai_address.desc_aimac_size = l_size;
213 } else if (element == "desc_drp") {
214 drpai_address.desc_drp_addr = l_addr;
215 drpai_address.desc_drp_size = l_size;
216 } else if (element == "drp_param") {
217 drpai_address.drp_param_addr = l_addr;
218 drpai_address.drp_param_size = l_size;
219 } else if (element == "weight") {
220 drpai_address.weight_addr = l_addr;
221 drpai_address.weight_size = l_size;
222 } else if (element == "data_in") {
223 drpai_address.data_in_addr = l_addr;
224 drpai_address.data_in_size = l_size;
225 } else if (element == "data") {
226 drpai_address.data_addr = l_addr;
227 drpai_address.data_size = l_size;
228 } else if (element == "data_out") {
229 drpai_address.data_out_addr = l_addr;
230 drpai_address.data_out_size = l_size;
231 } else if (element == "work") {
232 drpai_address.work_addr = l_addr;
233 drpai_address.work_size = l_size;
234 }
235 }
236
237 return 0;
238}

240/*****************************************
241 * Function Name : load_data_to_mem
242 * Description : Loads a binary blob DRP-AI Driver Memory
243 * Arguments : data_ptr = pointer to the bytes to write
244 * drpai_fd = file descriptor of DRP-AI Driver
245 * from = memory start address where the data is
246 *written size = data size to be written Return value : 0 if succeeded not 0
247 *otherwise
248 ******************************************/
249static int8_t load_data_to_mem(unsigned char *data_ptr, int drpai_fd,
250 unsigned long from, unsigned long size) {
251 drpai_data_t drpai_data;
252
253 drpai_data.address = from;
254 drpai_data.size = size;
255
256 errno = 0;
257 if (-1 == ioctl(drpai_fd, DRPAI_ASSIGN, &drpai_data)) {
258 return -1;
259 }
260
261 errno = 0;
262 if (-1 == write(drpai_fd, data_ptr, size)) {
263 return -1;
264 }
265
266 return 0;
267}

269/*****************************************
270 * Function Name : load_drpai_data
271 * Description : Loads DRP-AI Object files to memory via DRP-AI Driver.
272 * Arguments : drpai_fd = file descriptor of DRP-AI Driver
273 * Return value : 0 if succeeded
274 * : not 0 otherwise
275 ******************************************/
276static int load_drpai_data(int drpai_fd) {
277 unsigned long addr, size;
278 unsigned char *data_ptr;
279 for (int i = 0; i < 5; i++) {
280 switch (i) {
281 case (INDEX_W):
282 addr = drpai_address.weight_addr;
283 size = drpai_address.weight_size;
284 data_ptr = ei_ei_weight_dat;
285 break;
286 case (INDEX_C):
287 addr = drpai_address.drp_config_addr;
288 size = drpai_address.drp_config_size;
289 data_ptr = ei_ei_drpcfg_mem;
290 break;
291 case (INDEX_P):
292 addr = drpai_address.drp_param_addr;
293 size = drpai_address.drp_param_size;
294 data_ptr = ei_drp_param_bin;
295 break;
296 case (INDEX_A):
297 addr = drpai_address.desc_aimac_addr;
298 size = drpai_address.desc_aimac_size;
299 data_ptr = ei_aimac_desc_bin;
300 break;
301 case (INDEX_D):
302 addr = drpai_address.desc_drp_addr;
303 size = drpai_address.desc_drp_size;
304 data_ptr = ei_drp_desc_bin;
305 break;
306 default:
307 return -1;
308 break;
309 }
310 if (0 != load_data_to_mem(data_ptr, drpai_fd, addr, size)) {
311 return -1;
312 }
313 }
314 return 0;
315}

317EI_IMPULSE_ERROR drpai_init_classifier() {
318 // retval for drpai status
319 int ret_drpai;
320
321 // Read DRP-AI Object files address and size
322 if (0 != read_addrmap_txt()) {
323 ei_printf("ERR: read_addrmap_txt failed : %d\n", errno);
324 return EI_IMPULSE_DRPAI_INIT_FAILED;
325 }
326
327 // DRP-AI Driver Open
328 drpai_fd = open("/dev/drpai0", O_RDWR);
329 if (drpai_fd < 0) {
330 ei_printf("ERR: Failed to Open DRP-AI Driver: errno=%d\n", errno);
331 return EI_IMPULSE_DRPAI_INIT_FAILED;
332 }
333
334 // Load DRP-AI Data from Filesystem to Memory via DRP-AI Driver
335 ret_drpai = load_drpai_data(drpai_fd);
336 if (ret_drpai != 0) {
337 ei_printf("ERR: Failed to load DRPAI Data\n");
338 if (0 != close(drpai_fd)) {
339 ei_printf("ERR: Failed to Close DRPAI Driver: errno=%d\n", errno);
340 }
341 return EI_IMPULSE_DRPAI_INIT_FAILED;
342 }
343
344 // statically store DRP object file addresses and sizes
345 proc[DRPAI_INDEX_INPUT].address = (uint32_t)udmabuf_address;
346 proc[DRPAI_INDEX_INPUT].size = drpai_address.data_in_size;
347 proc[DRPAI_INDEX_DRP_CFG].address = drpai_address.drp_config_addr;
348 proc[DRPAI_INDEX_DRP_CFG].size = drpai_address.drp_config_size;
349 proc[DRPAI_INDEX_DRP_PARAM].address = drpai_address.drp_param_addr;
350 proc[DRPAI_INDEX_DRP_PARAM].size = drpai_address.drp_param_size;
351 proc[DRPAI_INDEX_AIMAC_DESC].address = drpai_address.desc_aimac_addr;
352 proc[DRPAI_INDEX_AIMAC_DESC].size = drpai_address.desc_aimac_size;
353 proc[DRPAI_INDEX_DRP_DESC].address = drpai_address.desc_drp_addr;
354 proc[DRPAI_INDEX_DRP_DESC].size = drpai_address.desc_drp_size;
355 proc[DRPAI_INDEX_WEIGHT].address = drpai_address.weight_addr;
356 proc[DRPAI_INDEX_WEIGHT].size = drpai_address.weight_size;
357 proc[DRPAI_INDEX_OUTPUT].address = drpai_address.data_out_addr;
358 proc[DRPAI_INDEX_OUTPUT].size = drpai_address.data_out_size;
359
360 EI_LOGD("proc[DRPAI_INDEX_INPUT] addr: %p, size: %p\r\n", proc[DRPAI_INDEX_INPUT].address, proc[DRPAI_INDEX_INPUT].size);
361 EI_LOGD("proc[DRPAI_INDEX_DRP_CFG] addr: %p, size: %p\r\n", proc[DRPAI_INDEX_DRP_CFG].address, proc[DRPAI_INDEX_DRP_CFG].size);
362 EI_LOGD("proc[DRPAI_INDEX_DRP_PARAM] addr: %p, size: %p\r\n", proc[DRPAI_INDEX_DRP_PARAM].address, proc[DRPAI_INDEX_DRP_PARAM].size);
363 EI_LOGD("proc[DRPAI_INDEX_AIMAC_DESC] addr: %p, size: %p\r\n", proc[DRPAI_INDEX_AIMAC_DESC].address, proc[DRPAI_INDEX_AIMAC_DESC].size);
364 EI_LOGD("proc[DRPAI_INDEX_DRP_DESC] addr: %p, size: %p\r\n", proc[DRPAI_INDEX_DRP_DESC].address, proc[DRPAI_INDEX_DRP_DESC].size);
365 EI_LOGD("proc[DRPAI_INDEX_WEIGHT] addr: %p, size: %p\r\n", proc[DRPAI_INDEX_WEIGHT].address, proc[DRPAI_INDEX_WEIGHT].size);
366 EI_LOGD("proc[DRPAI_INDEX_OUTPUT] addr: %p, size: %p\r\n", proc[DRPAI_INDEX_OUTPUT].address, proc[DRPAI_INDEX_OUTPUT].size);
367
368 drpai_output_buf = (float *)ei_malloc(drpai_address.data_out_size);
369
370 return EI_IMPULSE_OK;
371}

373EI_IMPULSE_ERROR drpai_run_classifier_image_quantized() {
374#if EI_CLASSIFIER_COMPILED == 1
375#error "DRP-AI is not compatible with EON Compiler"
376#endif
377 // output data from DRPAI model
378 drpai_data_t drpai_data;
379 // status used to query if any internal errors occured during inferencing
380 drpai_status_t drpai_status;
381 // descriptor used for checking if DRPAI is done inferencing
382 fd_set rfds;
383 // struct used to define DRPAI timeout
384 struct timespec tv;
385 // retval for drpai status
386 int ret_drpai;
387 // retval when querying drpai status
388 int inf_status = 0;
389
390 // DRP-AI Output Memory Preparation
391 drpai_data.address = drpai_address.data_out_addr;
392 drpai_data.size = drpai_address.data_out_size;
393
394 // Start DRP-AI driver
395 EI_LOGD("Start DRPAI inference\r\n");
396 int ioret = ioctl(drpai_fd, DRPAI_START, &proc[0]);
397 if (0 != ioret) {
398 EI_LOGE("Failed to Start DRPAI Inference: %d\n", errno);
399 return EI_IMPULSE_DRPAI_RUNTIME_FAILED;
400 }
401
402 // Settings For pselect - this is how DRPAI signals inferencing complete
403 FD_ZERO(&rfds);
404 FD_SET(drpai_fd, &rfds);
405 // Define a timeout for DRP-AI to complete
406 tv.tv_sec = DRPAI_TIMEOUT;
407 tv.tv_nsec = 0;
408
409 // Wait until DRP-AI ends
410 EI_LOGD("Waiting on DRPAI inference results\r\n");
411 ret_drpai = pselect(drpai_fd + 1, &rfds, NULL, NULL, &tv, NULL);
412 if (ret_drpai == 0) {
413 EI_LOGE("DRPAI Inference pselect() Timeout: %d\n", errno);
414 return EI_IMPULSE_DRPAI_RUNTIME_FAILED;
415 } else if (ret_drpai < 0) {
416 EI_LOGE("DRPAI Inference pselect() Error: %d\n", errno);
417 return EI_IMPULSE_DRPAI_RUNTIME_FAILED;
418 }
419
420 // Checks for DRPAI inference status errors
421 EI_LOGD("Getting DRPAI Status\r\n");
422 inf_status = ioctl(drpai_fd, DRPAI_GET_STATUS, &drpai_status);
423 if (inf_status != 0) {
424 EI_LOGE("DRPAI Internal Error: %d\n", errno);
425 return EI_IMPULSE_DRPAI_RUNTIME_FAILED;
426 }
427
428 EI_LOGD("Getting inference results\r\n");
429 if (ioctl(drpai_fd, DRPAI_ASSIGN, &drpai_data) != 0) {
430 EI_LOGE("Failed to Assign DRPAI data: %d\n", errno);
431 return EI_IMPULSE_DRPAI_RUNTIME_FAILED;
432 }
433
434 if (read(drpai_fd, drpai_output_buf, drpai_data.size) < 0) {
435 EI_LOGE("Failed to read DRPAI output data: %d\n", errno);
436 return EI_IMPULSE_DRPAI_RUNTIME_FAILED;
437 }
438 return EI_IMPULSE_OK;
439}

442EI_IMPULSE_ERROR drpai_close(uint32_t input_frame_size) {
443 munmap(drpai_input_buf, input_frame_size);
444 free(drpai_output_buf);

446 if (0 != close(drpai_fd)) {

451 }
452 return EI_IMPULSE_OK;
453}

455#if ((EI_CLASSIFIER_OBJECT_DETECTION == 1) && (EI_CLASSIFIER_OBJECT_DETECTION_LAST_LAYER == EI_CLASSIFIER_LAST_LAYER_YOLOV5_V5_DRPAI))
456EI_IMPULSE_ERROR drpai_run_yolov5_postprocessing(
457 const ei_impulse_t *impulse,
458 ei_learning_block_config_tflite_graph_t *block_config,
459 signal_t *signal,
460 ei_impulse_result_t *result,
461 bool debug = false)
462{
463
464 static std::unique_ptr<tflite::FlatBufferModel> model = nullptr;
465 static std::unique_ptr<tflite::Interpreter> interpreter = nullptr;
466
467 if (!model) {
468 model = tflite::FlatBufferModel::BuildFromBuffer((const char*)yolov5_part2, yolov5_part2_len);
469 if (!model) {
470 ei_printf("Failed to build TFLite model from buffer\n");
471 return EI_IMPULSE_TFLITE_ERROR;
472 }
473
474 tflite::ops::builtin::BuiltinOpResolver resolver;
475 tflite::InterpreterBuilder builder(*model, resolver);
476 builder(&interpreter);
477
478 if (!interpreter) {
479 ei_printf("Failed to construct interpreter\n");
480 return EI_IMPULSE_TFLITE_ERROR;
481 }
482
483 if (interpreter->AllocateTensors() != kTfLiteOk) {
484 ei_printf("AllocateTensors failed\n");
485 return EI_IMPULSE_TFLITE_ERROR;
486 }
487
488 int hw_thread_count = (int)std::thread::hardware_concurrency();
489 hw_thread_count -= 1; // leave one thread free for the other application
490 if (hw_thread_count < 1) {
491 hw_thread_count = 1;
492 }
493
494 if (interpreter->SetNumThreads(hw_thread_count) != kTfLiteOk) {
495 ei_printf("SetNumThreads failed\n");
496 return EI_IMPULSE_TFLITE_ERROR;
497 }
498 }
499
500 const size_t drpai_buff_size = drpai_address.data_out_size / sizeof(float);
501 const size_t drpai_features = drpai_buff_size;
502
503 const size_t els_per_grid = drpai_features / ((NUM_GRID_1 * NUM_GRID_1) + (NUM_GRID_2 * NUM_GRID_2) + (NUM_GRID_3 * NUM_GRID_3));
504
505 const size_t grid_1_offset = 0;
506 const size_t grid_1_size = (NUM_GRID_1 * NUM_GRID_1) * els_per_grid;
507
508 const size_t grid_2_offset = grid_1_offset + grid_1_size;
509 const size_t grid_2_size = (NUM_GRID_2 * NUM_GRID_2) * els_per_grid;
510
511 const size_t grid_3_offset = grid_2_offset + grid_2_size;
512 const size_t grid_3_size = (NUM_GRID_3 * NUM_GRID_3) * els_per_grid;
513
514 // Now we don't know the exact tensor order for some reason
515 // so let's do that dynamically
516 for (size_t ix = 0; ix < 3; ix++) {
517 TfLiteTensor * tensor = interpreter->input_tensor(ix);
518 size_t tensor_size = 1;
519 for (size_t ix = 0; ix < tensor->dims->size; ix++) {
520 tensor_size *= tensor->dims->data[ix];
521 }
522
523 EI_LOGD("input tensor %d, tensor_size=%d\n", (int)ix, (int)tensor_size);
524
525 float *input = interpreter->typed_input_tensor<float>(ix);
526
527 if (tensor_size == grid_1_size) {
528 memcpy(input, drpai_output_buf + grid_1_offset, grid_1_size * sizeof(float));
529 }
530 else if (tensor_size == grid_2_size) {
531 memcpy(input, drpai_output_buf + grid_2_offset, grid_2_size * sizeof(float));
532 }
533 else if (tensor_size == grid_3_size) {
534 memcpy(input, drpai_output_buf + grid_3_offset, grid_3_size * sizeof(float));
535 }
536 else {
537 ei_printf("ERR: Cannot determine which grid to use for input tensor %d with %d tensor size\n",
538 (int)ix, (int)tensor_size);
539 return EI_IMPULSE_TFLITE_ERROR;
540 }
541 }
542
543 uint64_t ctx_start_us = ei_read_timer_us();
544
545 interpreter->Invoke();
546
547 uint64_t ctx_end_us = ei_read_timer_us();
548
549 EI_LOGD("Invoke took %d ms.\n", (int)((ctx_end_us - ctx_start_us) / 1000));
550
551 float* out_data = interpreter->typed_output_tensor<float>(0);
552
553 const size_t out_size = impulse->tflite_output_features_count;
554
555 if (debug) {
556 printf("First 20 bytes: ");
557 for (size_t ix = 0; ix < 20; ix++) {
558 ei_printf("%f ", out_data[ix]);
559 }
560 ei_printf("\n");
561 }
562
563 // printf("Last 5 bytes: ");
564 // for (size_t ix = out_size - 5; ix < out_size; ix++) {
565 // printf("%f ", out_data[ix]);
566 // }
567 // printf("\n");
568
569 return fill_result_struct_f32_yolov5(impulse, block_config, result, 5, out_data, out_size);
570}

600EI_IMPULSE_ERROR run_nn_inference_image_quantized(
601 const ei_impulse_t *impulse,
602 signal_t *signal,
603 ei_impulse_result_t *result,
604 void *config_ptr,
605 bool debug = false)
606{
607 ei_learning_block_config_tflite_graph_t *block_config = (ei_learning_block_config_tflite_graph_t*)config_ptr;
608
609 // this needs to be changed for multi-model, multi-impulse
610 static bool first_run = true;
611 uint64_t ctx_start_us;
612 uint64_t dsp_start_us = ei_read_timer_us();
613
614 if (first_run) {
615 // map memory regions to the DRP-AI UDMA. This is required for passing data
616 // to and from DRP-AI
617 int t = drpai_init_mem(impulse->nn_input_frame_size);
618 if (t != 0) {
619 return EI_IMPULSE_DRPAI_INIT_FAILED;
620 }
621
622 EI_IMPULSE_ERROR ret = drpai_init_classifier();
623 if (ret != EI_IMPULSE_OK) {
624 drpai_close(impulse->nn_input_frame_size);
625 return EI_IMPULSE_DRPAI_INIT_FAILED;
626 }
627
628 EI_LOGI("Initialized input and output buffers:\r\n");
629 EI_LOGI("input buf (addr: %p, size: 0x%x)\r\n", drpai_input_buf, drpai_address.data_in_size);
630 EI_LOGI("output buf (addr: %p, size: 0x%x)\r\n", drpai_output_buf, drpai_address.data_out_size);
631 EI_LOGI("udmabuf_addr: %p\n", udmabuf_address);
632 }
633
634 EI_LOGD("Starting DSP...\n");
635 int ret;
636
637 EI_LOGD("fmatrix size == Bpp * signal.total_length ( %p == %p * %p = %p )\r\n", proc[DRPAI_INDEX_INPUT].size, 3, signal->total_length, 3 * signal->total_length);
638 // Creates a features matrix mapped to the DRP-AI UDMA input region
639 ei::matrix_u8_t features_matrix(1, proc[DRPAI_INDEX_INPUT].size, drpai_input_buf);
640
641 // Grabs the raw image buffer from the signal, DRP-AI will automatically
642 // extract features
643 ret = extract_drpai_features_quantized(
644 signal,
645 &features_matrix,
646 impulse->dsp_blocks[0].config,
647 impulse->frequency);
648 if (ret != EIDSP_OK) {
649 ei_printf("ERR: Failed to run DSP process (%d)\n", ret);
650 return EI_IMPULSE_DSP_ERROR;
651 }
652
653 if (ei_run_impulse_check_canceled() == EI_IMPULSE_CANCELED) {
654 return EI_IMPULSE_CANCELED;
655 }
656
657 result->timing.dsp_us = ei_read_timer_us() - dsp_start_us;
658 result->timing.dsp = (int)(result->timing.dsp_us / 1000);
659 if (debug) {
660 ei_printf("Features (%d ms.): ", result->timing.dsp);
661 for (size_t ix = 0; ix < EI_CLASSIFIER_NN_INPUT_FRAME_SIZE; ix++) {
662 ei_printf("0x%hhx, ", drpai_input_buf[ix]);
663 }
664 ei_printf("\n");
665 }
666
667 ctx_start_us = ei_read_timer_us();
668
669 // Run DRP-AI inference, a static buffer is used to store the raw output
670 // results
671 ret = drpai_run_classifier_image_quantized();
672
673 // close driver to reset memory, file pointer
674 if (ret != EI_IMPULSE_OK) {
675 drpai_close(impulse->nn_input_frame_size);
676 first_run = true;
677 }
678 else {
679 // drpai_reset();
680 first_run = false;
681 }
682
683 EI_IMPULSE_ERROR fill_res = EI_IMPULSE_OK;
684
685 if (block_config->object_detection) {
686 switch (block_config->object_detection_last_layer) {
687 case EI_CLASSIFIER_LAST_LAYER_FOMO: {
688 if (debug) {
689 ei_printf("DEBUG: raw drpai output");
690 ei_printf("\n[");
691 for (uint32_t i = 0; i < impulse->tflite_output_features_count; i++) {
692 ei_printf_float(drpai_output_buf[i]);
693 ei_printf(" ");
694 }
695 ei_printf("]\n");
696 }
697
698 fill_res = fill_result_struct_f32_fomo(
699 impulse,
700 block_config,
701 result,
702 drpai_output_buf,
703 impulse->fomo_output_size,
704 impulse->fomo_output_size);
705 break;
706 }
707 case EI_CLASSIFIER_LAST_LAYER_SSD: {
708 ei_printf("ERR: MobileNet SSD models are not implemented for DRP-AI (%d)\n",
709 block_config->object_detection_last_layer);
710 return EI_IMPULSE_UNSUPPORTED_INFERENCING_ENGINE;
711 }
712 case EI_CLASSIFIER_LAST_LAYER_YOLOV5_V5_DRPAI: {
713 if (debug) {
714 ei_printf("DEBUG: raw drpai output");
715 ei_printf("\n[");
716 // impulse->tflite_output_features_count can't be used here as this is not the final output
717 // so print only the first 10 values.
718 for (uint32_t i = 0; i < 10; i++) {
719 ei_printf_float(drpai_output_buf[i]);
720 ei_printf(" ");
721 }
722 ei_printf("]\n");
723 }
724
725#if ((EI_CLASSIFIER_OBJECT_DETECTION == 1) && (EI_CLASSIFIER_OBJECT_DETECTION_LAST_LAYER == EI_CLASSIFIER_LAST_LAYER_YOLOV5_V5_DRPAI))
726 // do post processing
727 fill_res = drpai_run_yolov5_postprocessing(impulse, block_config, signal, result, debug);
728#endif
729 break;
730 }
731 default: {
732 ei_printf("ERR: Unsupported object detection last layer (%d)\n",
733 block_config->object_detection_last_layer);
734 return EI_IMPULSE_UNSUPPORTED_INFERENCING_ENGINE;
735 }
736 }
737 }
738 else {
739 fill_res = fill_result_struct_f32(impulse, result, drpai_output_buf, debug);
740 }
741
742 if (fill_res != EI_IMPULSE_OK) {
743 return fill_res;
744 }
745
746 result->timing.classification_us = ei_read_timer_us() - ctx_start_us;
747 result->timing.classification = (int)(result->timing.classification_us / 1000);
748 return EI_IMPULSE_OK;
749}
