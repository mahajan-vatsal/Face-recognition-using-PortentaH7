44template <unsigned int tOpCount>
45class MicroMutableOpResolver : public MicroOpResolver {
46 public:
47 TF_LITE_REMOVE_VIRTUAL_DELETE
48
49 explicit MicroMutableOpResolver() {}
50
51 const TFLMRegistration* FindOp(tflite::BuiltinOperator op) const override {
52 if (op == BuiltinOperator_CUSTOM) return nullptr;
53
54 for (unsigned int i = 0; i < registrations_len_; ++i) {
55 const TFLMRegistration& registration = registrations_[i];
56 if (registration.builtin_code == op) {
57 return &registration;
58 }
59 }
60 return nullptr;
61 }
62
63 const TFLMRegistration* FindOp(const char* op) const override {
64 for (unsigned int i = 0; i < registrations_len_; ++i) {
65 const TFLMRegistration& registration = registrations_[i];
66 if ((registration.builtin_code == BuiltinOperator_CUSTOM) &&
67 (strcmp(registration.custom_name, op) == 0)) {
68 return &registration;
69 }
70 }
71 return nullptr;
72 }
73
74 TfLiteBridgeBuiltinParseFunction GetOpDataParser(
75 BuiltinOperator op) const override {
76 TFLITE_DCHECK(num_buitin_ops_ <= tOpCount);
77 for (unsigned int i = 0; i < num_buitin_ops_; ++i) {
78 if (builtin_codes_[i] == op) return builtin_parsers_[i];
79 }
80 return nullptr;
81 }

83 // Registers a Custom Operator with the MicroOpResolver.
84 //
85 // Only the first call for a given name will be successful. i.e. if this
86 // function is called again for a previously added Custom Operator, the
87 // MicroOpResolver will be unchanged and this function will return
88 // kTfLiteError.
89 TfLiteStatus AddCustom(const char* name,
90 const TFLMRegistration* registration) {
91 if (registrations_len_ >= tOpCount) {
92 MicroPrintf(
93 "Couldn't register custom op '%s', resolver size is too"
94 "small (%d)",
95 name, tOpCount);
96 return kTfLiteError;
97 }
98
99 if (FindOp(name) != nullptr) {
100 MicroPrintf("Calling AddCustom for the same op more than once ");
101 MicroPrintf("is not supported (Op: %s).", name);
102 return kTfLiteError;
103 }
104
105 TFLMRegistration* new_registration = &registrations_[registrations_len_];
106 registrations_len_ += 1;
107
108 *new_registration = *registration;
109 new_registration->builtin_code = BuiltinOperator_CUSTOM;
110 new_registration->custom_name = name;
111 return kTfLiteOk;
112 }

665 TfLiteStatus AddBuiltin(tflite::BuiltinOperator op,
666 const TFLMRegistration& registration,
667 TfLiteBridgeBuiltinParseFunction parser) {
668 if (op == BuiltinOperator_CUSTOM) {
669 MicroPrintf("Invalid parameter BuiltinOperator_CUSTOM to the ");
670 MicroPrintf("AddBuiltin function.");
671 return kTfLiteError;
672 }
673
674 if (FindOp(op) != nullptr) {
675 MicroPrintf("Calling AddBuiltin with the same op more than ");
676 MicroPrintf("once is not supported (Op: #%d).", op);
