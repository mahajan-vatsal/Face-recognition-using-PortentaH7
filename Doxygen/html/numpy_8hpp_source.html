364 static void transpose_in_place(matrix_t *matrix) {
365 // Don't bother if either dim is one, just need to swap the dimension sizes
366 if( matrix->rows != 1 && matrix->cols != 1) {
367 size_t size = matrix->cols * matrix->rows - 1;
368 float temp; // temp for swap
369 size_t next; // next item to swap
370 size_t cycleBegin; // index of start of cycle
371 size_t i; // location in matrix
372 size_t all_done_mark = 1;
373 ei_vector<bool> done(size+1,false);
374
375 i = 1; // Note that matrix[0] and last element of matrix won't move
376 while (1)
377 {
378 cycleBegin = i;
379 temp = matrix->buffer[i];
380 do
381 {
382 size_t col = i % matrix->cols;
383 size_t row = i / matrix->cols;
384 // swap row and col to make new idx, b/c we want to know where in the transposed matrix
385 next = col*matrix->rows + row;
386 float temp2 = matrix->buffer[next];
387 matrix->buffer[next] = temp;
388 temp = temp2;
389 done[next] = true;
390 i = next;
391 }
392 while (i != cycleBegin);
393
394 // start next cycle by find next not done
395 for (i = all_done_mark; done[i]; i++) {
396 all_done_mark++; // move the high water mark so we don't look again
397 if(i>=size) { goto LOOP_END; }
398 }
399 }
400 }
401 LOOP_END:
402 // finally, swap the row and column dimensions
403 std::swap(matrix->rows, matrix->cols);
404 }

532 if (!fft_data_out) {
533 return ei::EIDSP_OUT_OF_MEM;
534 }
535
536 float *fft_data_in = (float*)ei_dsp_calloc(fft_data_in_size, 1);
537 if (!fft_data_in) {
538 ei_dsp_free(fft_data_out, fft_data_out_size);
539 return ei::EIDSP_OUT_OF_MEM;
540 }
541
542 // Preprocess the input buffer with the data from the vector
543 size_t halfLen = len / 2;
544 for (size_t i = 0; i < halfLen; i++) {
545 fft_data_in[i] = vector[i * 2];
546 fft_data_in[len - 1 - i] = vector[i * 2 + 1];
547 }
548 if (len % 2 == 1) {
549 fft_data_in[halfLen] = vector[len - 1];
550 }
551
552 int r = ei::numpy::rfft(fft_data_in, len, fft_data_out, (len / 2 + 1), len);
553 if (r != 0) {
554 ei_dsp_free(fft_data_in, fft_data_in_size);
555 ei_dsp_free(fft_data_out, fft_data_out_size);
556 return r;
557 }
558
559 size_t i = 0;
560 for (; i < len / 2 + 1; i++) {
561 float temp = i * M_PI / (len * 2);
562 vector[i] = fft_data_out[i].r * cos(temp) + fft_data_out[i].i * sin(temp);
563 }
564 //take advantage of hermetian symmetry to calculate remainder of signal
565 for (; i < len; i++) {
566 float temp = i * M_PI / (len * 2);
567 int conj_idx = len-i;
568 // second half bins not calculated would have just been the conjugate of the first half (note minus of imag)
569 vector[i] = fft_data_out[conj_idx].r * cos(temp) - fft_data_out[conj_idx].i * sin(temp);
570 }
571 ei_dsp_free(fft_data_in, fft_data_in_size);
572 ei_dsp_free(fft_data_out, fft_data_out_size);
573
574 return 0;
575 }

628 static uint8_t quantize_zero_one(float value) {
629 const size_t length = sizeof(quantized_values_one_zero) / sizeof(float);
630
631 // look in the table
632 for (size_t ix = 0; ix < length; ix++) {
633 if (quantized_values_one_zero[ix] == value) return ix;
634 }
635
636 // no match?
637
638 if (value < quantized_values_one_zero[0]) {
639 return quantized_values_one_zero[0];
640 }
641 if (value > quantized_values_one_zero[length - 1]) {
642 return quantized_values_one_zero[length - 1];
643 }
644
645 int lo = 0;
646 int hi = length - 1;
647
648 while (lo <= hi) {
649 int mid = (hi + lo) / 2;
650
651 if (value < quantized_values_one_zero[mid]) {
652 hi = mid - 1;
653 } else if (value > quantized_values_one_zero[mid]) {
654 lo = mid + 1;
655 } else {
656 return quantized_values_one_zero[mid];
657 }
658 }
659
660 // lo == hi + 1
661 return (quantized_values_one_zero[lo] - value) < (value - quantized_values_one_zero[hi]) ?
662 lo :
663 hi;
664 }

1142 static int skew(matrix_t *input_matrix, matrix_t *output_matrix) {
1143 if (input_matrix->rows != output_matrix->rows) {
1144 EIDSP_ERR(EIDSP_MATRIX_SIZE_MISMATCH);
1145 }
1146 if (output_matrix->cols != 1) {
1147 EIDSP_ERR(EIDSP_MATRIX_SIZE_MISMATCH);
1148 }
1149
1150 for (size_t row = 0; row < input_matrix->rows; row++) {
1151#if EIDSP_USE_CMSIS_DSP
1152 float mean;
1153 float var;
1154
1155 // Calculate the mean & variance
1156 arm_mean_f32(input_matrix->buffer + (row * input_matrix->cols), input_matrix->cols, &mean);
1157 cmsis_arm_variance(&input_matrix->buffer[(row * input_matrix->cols)], input_matrix->cols, &var);

1160 float m_3;
1161 cmsis_arm_third_moment(&input_matrix->buffer[(row * input_matrix->cols)], input_matrix->cols, mean, &m_3);
1162
1163 // Calculate (variance)^(3/2)
1164 arm_sqrt_f32(var * var * var, &var);
1165
1166 // Calculate skew = (m_3) / (variance)^(3/2)
1167 if (var == 0.0f) {
1168 output_matrix->buffer[row] = 0.0f;

1178 sum += input_matrix->buffer[( row * input_matrix->cols ) + col];
1179 }

1183 float m_3 = 0.0f;

1187 float diff;
1188 diff = input_matrix->buffer[( row * input_matrix->cols ) + col] - mean;
1189 m_3 += diff * diff * diff;
1190 m_2 += diff * diff;
1191 }
1192 m_3 = m_3 / input_matrix->cols;
1193 m_2 = m_2 / input_matrix->cols;
1194
1195 // Calculate (m_2)^(3/2)
1196 m_2 = sqrt(m_2 * m_2 * m_2);
1197
1198 // Calculate skew = (m_3) / (m_2)^(3/2)
1199 if (m_2 == 0.0f) {
1200 output_matrix->buffer[row] = 0.0f;
1201 } else {
1202 output_matrix->buffer[row] = m_3 / m_2;

1204#endif

1215 static int kurtosis(matrix_t *input_matrix, matrix_t *output_matrix) {
1216 if (input_matrix->rows != output_matrix->rows) {
1217 EIDSP_ERR(EIDSP_MATRIX_SIZE_MISMATCH);
1218 }
1219 if (output_matrix->cols != 1) {
1220 EIDSP_ERR(EIDSP_MATRIX_SIZE_MISMATCH);
1221 }
1222
1223 for (size_t row = 0; row < input_matrix->rows; row++) {
1224#if EIDSP_USE_CMSIS_DSP
1225 float mean;
1226 float var;
1227
1228 // Calculate mean & variance
1229 arm_mean_f32(input_matrix->buffer + (row * input_matrix->cols), input_matrix->cols, &mean);
1230 cmsis_arm_variance(&input_matrix->buffer[(row * input_matrix->cols)], input_matrix->cols, &var);
1231
1232 // Calculate m_4
1233 float m_4;
1234 cmsis_arm_fourth_moment(&input_matrix->buffer[(row * input_matrix->cols)], input_matrix->cols, mean, &m_4);
1235
1236 // Calculate Fisher kurtosis = (m_4 / variance^2) - 3
1237 var = var * var;
1238 if (var == 0.0f) {
1239 output_matrix->buffer[row] = -3.0f;
1240 } else {
1241 output_matrix->buffer[row] = (m_4 / var) - 3.0f;
1242 }

1246 float sum = 0.0f;
1247
1248 for (size_t col = 0; col < input_matrix->cols; col++) {
1249 sum += input_matrix->buffer[( row * input_matrix->cols ) + col];
1250 }
1251 mean = sum / input_matrix->cols;
1252

1254 float m_4 = 0.0f;

1256
1257 for (size_t col = 0; col < input_matrix->cols; col++) {
1258 float diff;
1259 diff = input_matrix->buffer[(row * input_matrix->cols) + col] - mean;
1260 float square_diff = diff * diff;
1261 variance += square_diff;
1262 m_4 += square_diff * square_diff;
1263 }
1264 m_4 = m_4 / input_matrix->cols;
1265 variance = variance / input_matrix->cols;
1266
1267 // Square the variance
1268 variance = variance * variance;
1269 // Calculate Fisher kurtosis = (m_4 / variance^2) - 3
1270 if (variance == 0.0f) {
1271 output_matrix->buffer[row] = -3.0f;
1272 } else {
1273 output_matrix->buffer[row] = (m_4 / variance) - 3.0f;
1274 }
1275#endif
1276 }
1277
1278 return EIDSP_OK;
1279 }

2406 for (size_t i = 0; i < n; i++) {
2407 mean += data[i];
2408 }

2413 float y_mean = mean;
2414 float numerator = 0.0;
2415 float denominator = 0.0;

2417 numerator += (i + 1 - x_mean) * (data[i] - y_mean);
2418 denominator += (i + 1 - x_mean) * (i + 1 - x_mean);
