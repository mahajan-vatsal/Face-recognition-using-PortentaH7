276// Calculates the useful range of an activation layer given its activation
277// tensor.a
278template <typename T>
279void CalculateActivationRange(TfLiteFusedActivation activation,
280 T* activation_min, T* activation_max) {
281 if (activation == kTfLiteActRelu) {
282 *activation_min = 0;
283 *activation_max = std::numeric_limits<T>::max();
284 } else if (activation == kTfLiteActRelu6) {
285 *activation_min = 0;
286 *activation_max = 6;
287 } else if (activation == kTfLiteActReluN1To1) {
288 *activation_min = -1;
289 *activation_max = 1;
290 } else {
291 *activation_min = std::numeric_limits<T>::lowest();
292 *activation_max = std::numeric_limits<T>::max();
293 }
294}
