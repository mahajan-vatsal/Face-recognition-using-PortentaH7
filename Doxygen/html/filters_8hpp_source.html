53 static void butterworth_lowpass(
54 int filter_order,
55 float sampling_freq,
56 float cutoff_freq,
57 const float *src,
58 float *dest,
59 size_t size)
60 {
61 int n_steps = filter_order / 2;
62 float a = tan(M_PI * cutoff_freq / sampling_freq);
63 float a2 = pow(a, 2);
64 float *A = (float*)ei_dsp_calloc(n_steps, sizeof(float));
65 float *d1 = (float*)ei_dsp_calloc(n_steps, sizeof(float));
66 float *d2 = (float*)ei_dsp_calloc(n_steps, sizeof(float));
67 float *w0 = (float*)ei_dsp_calloc(n_steps, sizeof(float));
68 float *w1 = (float*)ei_dsp_calloc(n_steps, sizeof(float));
69 float *w2 = (float*)ei_dsp_calloc(n_steps, sizeof(float));
70
71 // Calculate the filter parameters
72 for(int ix = 0; ix < n_steps; ix++) {
73 float r = sin(M_PI * ((2.0 * ix) + 1.0) / (2.0 * filter_order));
74 sampling_freq = a2 + (2.0 * a * r) + 1.0;
75 A[ix] = a2 / sampling_freq;
76 d1[ix] = 2.0 * (1 - a2) / sampling_freq;
77 d2[ix] = -(a2 - (2.0 * a * r) + 1.0) / sampling_freq;
78 }
79
80 // Apply the filter
81 for (size_t sx = 0; sx < size; sx++) {
82 dest[sx] = src[sx];
83
84 for (int i = 0; i < n_steps; i++) {
85 w0[i] = d1[i] * w1[i] + d2[i] * w2[i] + dest[sx];
86 dest[sx] = A[i] * (w0[i] + (2.0 * w1[i]) + w2[i]);
87 w2[i] = w1[i];
88 w1[i] = w0[i];
89 }
90 }
91
92 ei_dsp_free(A, n_steps*sizeof(float));
93 ei_dsp_free(d1, n_steps*sizeof(float));
94 ei_dsp_free(d2, n_steps*sizeof(float));
95 ei_dsp_free(w0, n_steps*sizeof(float));
96 ei_dsp_free(w1, n_steps*sizeof(float));
97 ei_dsp_free(w2, n_steps*sizeof(float));
98 }
