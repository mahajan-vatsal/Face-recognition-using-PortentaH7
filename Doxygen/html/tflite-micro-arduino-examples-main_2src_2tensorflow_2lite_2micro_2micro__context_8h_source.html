23namespace tflite {
24// MicroContext is eventually going to become the API between TFLM and the
25// kernels, replacing all the functions in TfLiteContext. The end state is code
26// kernels to have code like:
27//
28// MicroContext* micro_context = GetMicroContext(context);
29// micro_context-><TFLM kernel API>
30class MicroContext {
31 public:
32 // Does not take any ownership, and all pointers must refer to valid objects
33 // that outlive the one constructed.
34 explicit MicroContext(MicroAllocator* allocator, const Model* model,
35 MicroGraph* graph);
36 virtual ~MicroContext();
37
38 // Allocate persistent buffer which has the same life time as the interpreter.
39 // Returns nullptr on failure.
40 // The memory is allocated from the tail.
41 // This method is only available in Init or Prepare stage.
42 // Virtual so that it can be faked for kernel tests.
43 virtual void* AllocatePersistentBuffer(size_t bytes);
44
45 // Request a scratch buffer in the arena through static memory planning.
46 // This method is only available in Prepare stage and the buffer is allocated
47 // by the interpreter between Prepare and Eval stage. In Eval stage,
48 // GetScratchBuffer API can be used to fetch the address.
49 // Virtual so that it can be faked for kernel tests.
50 virtual TfLiteStatus RequestScratchBufferInArena(size_t bytes,
51 int* buffer_idx);
52
53 // Get the scratch buffer pointer.
54 // This method is only available in Eval stage.
55 // Virtual so that it can be faked for kernel tests.
56 virtual void* GetScratchBuffer(int buffer_idx);
57
58 // Returns a temporary TfLiteTensor struct for a given index.
59 // Virtual so that it can be faked for kernel tests.
60 virtual TfLiteTensor* AllocateTempTfLiteTensor(int tensor_idx);
61
62 // Returns a temporary TfLiteTensor struct for the specified input tensor of a
63 // given mode. This is the recommended API over the deprecated
64 // GetInput/GetInputSafe to get a temp input tensor. The returned tensor shall
65 // be freed via calling DeallocateTempTfLiteTensor.
66 virtual TfLiteTensor* AllocateTempInputTensor(const TfLiteNode* node,
67 int index);
68
69 // Returns a temporary TfLiteTensor struct for the specified output tensor of
70 // a given mode. This is the recommended API over the deprecated
71 // GetOutput/GetOutputSafe to get a temp output tensor. The returned tensor
72 // shall be freed via calling DeallocateTempTfLiteTensor.
73 virtual TfLiteTensor* AllocateTempOutputTensor(const TfLiteNode* node,
74 int index);
75
76 // Returns a temporary TfLiteTensor struct for the specified intermediate
77 // tensor of a given mode. This is the recommended API over the deprecated
78 // GetIntermediates/GetIntermediatesSafe to get a temp intermediate tensor.
79 // The returned tensor shall be freed via calling DeallocateTempTfLiteTensor.
80 virtual TfLiteTensor* AllocateTempIntermediateTensor(const TfLiteNode* node,
81 int index);
82
83 // Deallocates a temp TfLiteTensor.
84 // Virtual so that it can be faked for kernel tests.
85 virtual void DeallocateTempTfLiteTensor(TfLiteTensor* tensor);
86
87 // Returns a TfLiteEvalTensor struct for a given index.
88 // Virtual so that it can be faked for kernel tests.
89 virtual TfLiteEvalTensor* GetEvalTensor(int tensor_idx);
90
91 // Does not take ownership of the pointer and the pointer must refer to valid
92 // an object that outlive this class instance.
93 // This can only be called once to set one external context.
94 TfLiteStatus set_external_context(void* external_context_payload);
95
96 void* external_context() { return external_context_payload_; }
97
98 MicroGraph& graph() { return graph_; }
99
100 // Sets the pointer to a list of ScratchBufferHandle instances.
101 // Not API between TFLM and kernels. Primarily used by the framework for
102 // housekeeping in MicroContext.
103 void SetScratchBufferHandles(ScratchBufferHandle* scratch_buffer_handles);
104
105 private:
106 // Return the tensor index as tensor_indices[index]. tensor_indices is of
107 // max_size. Return -1 if index is not in the valid range of tensor_indices.
108 int GetTensorIndex(int index, int max_size, const int* tensor_indices);
109
110 MicroAllocator& allocator_;
111 MicroGraph& graph_;
112 const Model* model_;
113
114 ScratchBufferHandle* scratch_buffer_handles_ = nullptr;
115 void* external_context_payload_ = nullptr;
116
117 TF_LITE_REMOVE_VIRTUAL_DELETE
118};

120inline MicroContext* GetMicroContext(const struct TfLiteContext* context) {
121 return reinterpret_cast<MicroContext*>(context->impl_);
122}

124// Deprecated API. Prefer to using the MicroContext API directly from the
125// kernels.
126// TODO(b/213010668): migrate all existing kernels to use MicroContext, delete
127// these functions, and remove corresponding members from the TfLiteContext
128// struct for TFLM.
129inline void* MicroContextAllocatePersistentBuffer(TfLiteContext* ctx,
130 size_t bytes) {

132}

134 size_t bytes,

137}
138inline void* MicroContextGetScratchBuffer(TfLiteContext* ctx, int buffer_idx) {
139 return GetMicroContext(ctx)->GetScratchBuffer(buffer_idx);
140}
141inline TfLiteTensor* MicroContextGetTensor(const struct TfLiteContext* context,
142 int tensor_idx) {
143 return GetMicroContext(context)->AllocateTempTfLiteTensor(tensor_idx);
144}
145inline TfLiteEvalTensor* MicroContextGetEvalTensor(
146 const struct TfLiteContext* context, int tensor_idx) {
147 return GetMicroContext(context)->GetEvalTensor(tensor_idx);
148}
