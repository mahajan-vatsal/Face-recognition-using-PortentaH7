The system captures facial images, processes them locally using an AI model deployed on the Portenta H7 and determines access based on authorised personnel.

43class PerfCal {
44public:
45 PerfCal(
46 const ei_performance_calibration_config_t *config,
47 uint32_t n_labels,
48 uint32_t sample_length,
49 float sample_interval_ms)
50 {
51 this->_score_array = nullptr;
52 this->_running_sum = nullptr;
53 this->_detection_threshold = config->detection_threshold;
54 this->_suppression_flags = config->suppression_flags;
55 this->_should_boost = config->is_configured;
56 this->_n_labels = n_labels;
57
58 /* Determine sample length in ms */
59 float sample_length_ms = (static_cast<float>(sample_length) * sample_interval_ms);
60
61 /* Calculate number of inference runs needed for the duration window */
62 this->_average_window_duration_samples =
63 (config->average_window_duration_ms < static_cast<uint32_t>(sample_length_ms))
64 ? 1
65 : static_cast<uint32_t>(static_cast<float>(config->average_window_duration_ms) / sample_length_ms);
66
67 /* Calculate number of inference runs for suppression */
68 this->_suppression_samples = (config->suppression_ms < static_cast<uint32_t>(sample_length_ms))
69 ? 0
70 : static_cast<uint32_t>(static_cast<float>(config->suppression_ms) / sample_length_ms);
71
72 /* Detection threshold should be high enough to only classify 1 possible output */
73 if (this->_detection_threshold <= (1.f / this->_n_labels)) {
74 EI_LOGE("Classifier detection threshold too low\r\n");
75 return;
76 }
77
78 /* Array to store scores for all labels */
79 this->_score_array = (float *)ei_malloc(
80 this->_average_window_duration_samples * this->_n_labels * sizeof(float));
81
82 if (this->_score_array == NULL) {
83 ei_printf(MEM_ERROR);
84 return;
85 }
86
87 for (uint32_t i = 0; i < this->_average_window_duration_samples * this->_n_labels; i++) {
88 this->_score_array[i] = 0.f;
89 }
90 this->_score_idx = 0;
91
92 /* Running sum for all labels */
93 this->_running_sum = (float *)ei_malloc(this->_n_labels * sizeof(float));
94
95 if (this->_running_sum != NULL) {
96 for (uint32_t i = 0; i < this->_n_labels; i++) {
97 this->_running_sum[i] = 0.f;
98 }
99 }
100 else {
101 ei_printf(MEM_ERROR);
102 return;
103 }
104
105 this->_suppression_count = this->_suppression_samples;
106 this->_n_scores_in_array = 0;
107 }
108

112 ei_free((void *)this->_score_array);
113 }
114 if (this->_running_sum) {
115 ei_free((void *)this->_running_sum);
116 }
117 }

134 int32_t trigger(ei_impulse_result_classification_t *scores)
135 {
136 int32_t recognized_event = EI_PC_RET_NO_EVENT_DETECTED;
137 float current_top_score = 0.f;
138 uint32_t current_top_index = 0;
139
140 /* Check pointers */
141 if (this->_score_array == NULL || this->_running_sum == NULL) {
142 return EI_PC_RET_MEMORY_ERROR;
143 }
144
145 /* Update the score array and running sum */
146 for (uint32_t i = 0; i < this->_n_labels; i++) {
147 this->_running_sum[i] -= this->_score_array[(this->_score_idx * this->_n_labels) + i];
148 this->_running_sum[i] += scores[i].value;
149 this->_score_array[(this->_score_idx * this->_n_labels) + i] = scores[i].value;
150 }
151
152 if (++this->_score_idx >= this->_average_window_duration_samples) {
153 this->_score_idx = 0;
154 }
155
156 /* Number of samples to average, increases until the buffer is full */
157 if (this->_n_scores_in_array < this->_average_window_duration_samples) {
158 this->_n_scores_in_array++;
159 }
160
161 /* Average data and place in scores & determine top score */
162 for (uint32_t i = 0; i < this->_n_labels; i++) {
163 scores[i].value = this->_running_sum[i] / this->_n_scores_in_array;
164
165 if (scores[i].value > current_top_score) {
166 if(this->_suppression_flags == 0) {
167 current_top_score = scores[i].value;
168 current_top_index = i;
169 }
170 else if(this->_suppression_flags & (1 << i)) {
171 current_top_score = scores[i].value;
172 current_top_index = i;
173 }
174 }
175 }
176
177 /* Check threshold, suppression */
178 if (this->_suppression_samples && this->_suppression_count < this->_suppression_samples) {
179 this->_suppression_count++;
180 }
181 else {
182 if (current_top_score >= this->_detection_threshold) {
183 recognized_event = current_top_index;
184
185 if (this->_suppression_flags & (1 << current_top_index)) {
186 this->_suppression_count = 0;
187 }
188 }
189 }
190
191 return recognized_event;
192 };

245EI_IMPULSE_ERROR process_perfcal(ei_impulse_handle_t *handle,
246 ei_impulse_result_t *result,
247 void *config,
248 void *state)
249{
250 const ei_impulse_t *impulse = handle->impulse;
251 PerfCal *perf_cal = (PerfCal*)state;
252
253 if (impulse->sensor == EI_CLASSIFIER_SENSOR_MICROPHONE) {
254 if((void *)perf_cal != NULL) {
255
256 // perfcal is configured
257 static bool has_printed_msg = false;
258 result->postprocessed_output.perf_cal_output = *std::unique_ptr<ei_perf_cal_output_t>(new ei_perf_cal_output_t).get();
259
260 if (!has_printed_msg) {
261 ei_printf("\nPerformance calibration is configured for your project. If no event is detected, all values are 0.\r\n\n");
262 has_printed_msg = true;
263 }
264
265 int label_detected = perf_cal->trigger(result->classification);
266
267 if (perf_cal->should_boost()) {
268 for (int i = 0; i < impulse->label_count; i++) {
269 if (i == label_detected) {
270 result->classification[i].value = 1.0f;
271 result->postprocessed_output.perf_cal_output.detected_label = (char*)result->classification[i].label;
272 }
273 else {
274 result->classification[i].value = 0.0f;
275 }
276 }
277 }
278 }
279 }
280 return EI_IMPULSE_OK;
281}
