50 const TfLiteRegistration* FindOp(tflite::BuiltinOperator op) const override {
51 if (op == BuiltinOperator_CUSTOM) return nullptr;
52
53 for (unsigned int i = 0; i < registrations_len_; ++i) {
54 const TfLiteRegistration& registration = registrations_[i];
55 if (registration.builtin_code == op) {
56 return &registration;
57 }
58 }
59 return nullptr;
60 }
61
62 const TfLiteRegistration* FindOp(const char* op) const override {
63 for (unsigned int i = 0; i < registrations_len_; ++i) {
64 const TfLiteRegistration& registration = registrations_[i];
65 if ((registration.builtin_code == BuiltinOperator_CUSTOM) &&
66 (strcmp(registration.custom_name, op) == 0)) {
67 return &registration;
68 }
69 }
70 return nullptr;
71 }
72
73 TfLiteBridgeBuiltinParseFunction GetOpDataParser(
74 BuiltinOperator op) const override {
75 TFLITE_DCHECK(num_buitin_ops_ <= tOpCount);
76 for (unsigned int i = 0; i < num_buitin_ops_; ++i) {
77 if (builtin_codes_[i] == op) return builtin_parsers_[i];
78 }
79 return nullptr;
80 }

82 // Registers a Custom Operator with the MicroOpResolver.
83 //
84 // Only the first call for a given name will be successful. i.e. if this
85 // function is called again for a previously added Custom Operator, the
86 // MicroOpResolver will be unchanged and this function will return
87 // kTfLiteError.
88 TfLiteStatus AddCustom(const char* name, TfLiteRegistration* registration) {
89 if (registrations_len_ >= tOpCount) {
90 MicroPrintf(
91 "Couldn't register custom op '%s', resolver size is too"
92 "small (%d)",
93 name, tOpCount);
94 return kTfLiteError;
95 }
96
97 if (FindOp(name) != nullptr) {
98 MicroPrintf("Calling AddCustom for the same op more than once ");
99 MicroPrintf("is not supported (Op: %s).", name);
100 return kTfLiteError;
101 }
102
103 TfLiteRegistration* new_registration = &registrations_[registrations_len_];
104 registrations_len_ += 1;
105
106 *new_registration = *registration;
107 new_registration->builtin_code = BuiltinOperator_CUSTOM;
108 new_registration->custom_name = name;
109 return kTfLiteOk;
110 }

654 private:
655 TfLiteStatus AddBuiltin(tflite::BuiltinOperator op,
656 const TfLiteRegistration& registration,
657 TfLiteBridgeBuiltinParseFunction parser) {
658 if (op == BuiltinOperator_CUSTOM) {
659 MicroPrintf("Invalid parameter BuiltinOperator_CUSTOM to the ");
660 MicroPrintf("AddBuiltin function.");
661 return kTfLiteError;
662 }
663
664 if (FindOp(op) != nullptr) {
665 MicroPrintf("Calling AddBuiltin with the same op more than ");
666 MicroPrintf("once is not supported (Op: #%d).", op);
667 return kTfLiteError;
668 }
669
670 if (registrations_len_ >= tOpCount) {
671 MicroPrintf("Couldn't register builtin op #%d, resolver size ", op);
672 MicroPrintf("is too small (%d).", tOpCount);
673 return kTfLiteError;
674 }
675
676 registrations_[registrations_len_] = registration;
677 // Strictly speaking, the builtin_code is not necessary for TFLM but filling
678 // it in regardless.
