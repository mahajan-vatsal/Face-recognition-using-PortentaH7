51template <typename T> std::vector<intptr_t> argsort_iter(const std::vector<T> &v)
52{
53 std::vector<intptr_t> index(v.size());
54 std::iota(index.begin(), index.end(), 0);
55 std::sort(index.begin(), index.end(), [&v](intptr_t i, intptr_t j)
56 {return v[i] < v[j];});
57 return index;
58}

60static intptr_t
61augmenting_path(intptr_t nc, double *cost, std::vector<double>& u,
62 std::vector<double>& v, std::vector<intptr_t>& path,
63 std::vector<intptr_t>& row4col,
64 std::vector<double>& shortestPathCosts, intptr_t i,
65 std::vector<bool>& SR, std::vector<bool>& SC,
66 std::vector<intptr_t>& remaining, double* p_minVal)
67{
68 double minVal = 0;
69
70 // Crouse's pseudocode uses set complements to keep track of remaining
71 // nodes. Here we use a vector, as it is more efficient in C++.
72 intptr_t num_remaining = nc;
73 for (intptr_t it = 0; it < nc; it++) {
74 // Filling this up in reverse order ensures that the solution of a
75 // constant cost matrix is the identity matrix (c.f. #11602).
76 remaining[it] = nc - it - 1;
77 }
78
79 std::fill(SR.begin(), SR.end(), false);
80 std::fill(SC.begin(), SC.end(), false);
81 std::fill(shortestPathCosts.begin(), shortestPathCosts.end(), INFINITY);
82
83 // find shortest augmenting path
84 intptr_t sink = -1;
85 while (sink == -1) {
86
87 intptr_t index = -1;
88 double lowest = INFINITY;
89 SR[i] = true;
90
91 for (intptr_t it = 0; it < num_remaining; it++) {
92 intptr_t j = remaining[it];
93
94 double r = minVal + cost[i * nc + j] - u[i] - v[j];
95 if (r < shortestPathCosts[j]) {
96 path[j] = i;
97 shortestPathCosts[j] = r;
98 }
99
100 // When multiple nodes have the minimum cost, we select one which
101 // gives us a new sink node. This is particularly important for
102 // integer cost matrices with small co-efficients.
103 if (shortestPathCosts[j] < lowest ||
104 (shortestPathCosts[j] == lowest && row4col[j] == -1)) {
105 lowest = shortestPathCosts[j];
106 index = it;
107 }
108 }
109
110 minVal = lowest;
111 if (minVal == INFINITY) { // infeasible cost matrix
112 return -1;
113 }
114
115 intptr_t j = remaining[index];
116 if (row4col[j] == -1) {
117 sink = j;
118 } else {
119 i = row4col[j];
120 }
121
122 SC[j] = true;
123 remaining[index] = remaining[--num_remaining];
124 }
125
126 *p_minVal = minVal;
127 return sink;
128}

130static int solve(intptr_t nr, intptr_t nc, double* cost, bool maximize,
131 int64_t* a, int64_t* b) {
132 // handle trivial inputs
133 if (nr == 0 || nc == 0) {
134 return 0;
135 }
136
137 // tall rectangular cost matrix must be transposed
138 bool transpose = nc < nr;
139
140 // make a copy of the cost matrix if we need to modify it
141 std::vector<double> temp;
142 if (transpose || maximize) {
143 temp.resize(nr * nc);
144
145 if (transpose) {
146 for (intptr_t i = 0; i < nr; i++) {
147 for (intptr_t j = 0; j < nc; j++) {
148 temp[j * nr + i] = cost[i * nc + j];
149 }
150 }
151
152 std::swap(nr, nc);
153 }
154 else {
155 std::copy(cost, cost + nr * nc, temp.begin());
156 }
157
158 // negate cost matrix for maximization
159 if (maximize) {
160 for (intptr_t i = 0; i < nr * nc; i++) {
161 temp[i] = -temp[i];
162 }
163 }
164
165 cost = temp.data();
166 }
167
168 // test for NaN and -inf entries
169 for (intptr_t i = 0; i < nr * nc; i++) {
170 if (cost[i] != cost[i] || cost[i] == -INFINITY) {
171 return RECTANGULAR_LSAP_INVALID;
172 }
173 }
174
175 // initialize variables
176 std::vector<double> u(nr, 0);
177 std::vector<double> v(nc, 0);
178 std::vector<double> shortestPathCosts(nc);
179 std::vector<intptr_t> path(nc, -1);
180 std::vector<intptr_t> col4row(nr, -1);
181 std::vector<intptr_t> row4col(nc, -1);
182 std::vector<bool> SR(nr);
183 std::vector<bool> SC(nc);
184 std::vector<intptr_t> remaining(nc);
185
186 // iteratively build the solution
187 for (intptr_t curRow = 0; curRow < nr; curRow++) {
188
189 double minVal;
190 intptr_t sink = augmenting_path(nc, cost, u, v, path, row4col,
191 shortestPathCosts, curRow, SR, SC,
192 remaining, &minVal);
193 if (sink < 0) {
194 return RECTANGULAR_LSAP_INFEASIBLE;
195 }
196
197 // update dual variables
198 u[curRow] += minVal;
199 for (intptr_t i = 0; i < nr; i++) {
200 if (SR[i] && i != curRow) {
201 u[i] += minVal - shortestPathCosts[col4row[i]];
202 }
203 }
204
205 for (intptr_t j = 0; j < nc; j++) {
206 if (SC[j]) {
207 v[j] -= minVal - shortestPathCosts[j];
208 }
209 }
210
211 // augment previous solution
212 intptr_t j = sink;
213 while (1) {
214 intptr_t i = path[j];
215 row4col[j] = i;
216 std::swap(col4row[i], j);
217 if (i == curRow) {
218 break;
219 }
220 }
221 }
222
223 if (transpose) {
224 intptr_t i = 0;
225 for (auto v: argsort_iter(col4row)) {
226 a[i] = col4row[v];
227 b[i] = v;
228 i++;
229 }
230 }
231 else {
232 for (intptr_t i = 0; i < nr; i++) {
233 a[i] = i;
234 b[i] = col4row[i];
235 }
236 }
237
238 return 0;
239}
