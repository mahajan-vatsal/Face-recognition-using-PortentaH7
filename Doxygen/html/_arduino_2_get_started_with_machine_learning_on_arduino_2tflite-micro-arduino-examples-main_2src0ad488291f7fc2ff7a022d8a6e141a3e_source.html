189// Check whether value is true, and if not return kTfLiteError from
190// the current function (and report the error string msg).
191#define TF_LITE_ENSURE_MSG(context, value, msg) \
192 do { \
193 if (!(value)) { \
194 TF_LITE_KERNEL_LOG((context), __FILE__ " " msg); \
195 return kTfLiteError; \
196 } \
197 } while (0)

199// Check whether the value `a` is true, and if not return kTfLiteError from
200// the current function, while also reporting the location of the error.
201#define TF_LITE_ENSURE(context, a) \
202 do { \
203 if (!(a)) { \
204 TF_LITE_KERNEL_LOG((context), "%s:%d %s was not true.", __FILE__, \
205 __LINE__, #a); \
206 return kTfLiteError; \
207 } \
208 } while (0)
209
210#define TF_LITE_ENSURE_STATUS(a) \
211 do { \
212 const TfLiteStatus s = (a); \
213 if (s != kTfLiteOk) { \
214 return s; \
215 } \
216 } while (0)

218// Check whether the value `a == b` is true, and if not return kTfLiteError from
219// the current function, while also reporting the location of the error.
220// `a` and `b` may be evaluated more than once, so no side effects or
221// extremely expensive computations should be done.
222// NOTE: Use TF_LITE_ENSURE_TYPES_EQ if comparing TfLiteTypes.
223#define TF_LITE_ENSURE_EQ(context, a, b) \
224 do { \
225 if ((a) != (b)) { \
226 TF_LITE_KERNEL_LOG((context), "%s:%d %s != %s (%d != %d)", __FILE__, \
227 __LINE__, #a, #b, (a), (b)); \
228 return kTfLiteError; \
229 } \
230 } while (0)
231
232#define TF_LITE_ENSURE_TYPES_EQ(context, a, b) \
233 do { \
234 if ((a) != (b)) { \
235 TF_LITE_KERNEL_LOG((context), "%s:%d %s != %s (%s != %s)", __FILE__, \
236 __LINE__, #a, #b, TfLiteTypeGetName(a), \
237 TfLiteTypeGetName(b)); \
238 return kTfLiteError; \
239 } \
240 } while (0)
241
242#define TF_LITE_ENSURE_NEAR(context, a, b, epsilon) \
243 do { \
244 auto delta = ((a) > (b)) ? ((a) - (b)) : ((b) - (a)); \
245 if (delta > epsilon) { \
246 TF_LITE_KERNEL_LOG((context), "%s:%d %s not near %s (%f != %f)", \
247 __FILE__, __LINE__, #a, #b, static_cast<double>(a), \
248 static_cast<double>(b)); \
249 return kTfLiteError; \
250 } \
251 } while (0)

253#define TF_LITE_ENSURE_OK(context, status) \
254 do { \
255 const TfLiteStatus s = (status); \
256 if ((s) != kTfLiteOk) { \
257 return s; \
258 } \
259 } while (0)
