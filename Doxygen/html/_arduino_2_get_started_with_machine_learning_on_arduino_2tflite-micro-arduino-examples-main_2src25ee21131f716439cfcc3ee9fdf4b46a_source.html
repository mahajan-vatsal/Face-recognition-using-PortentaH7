90// Manages the slice position (offset), slice length (sliced tensor shape),
91// and update rules for input/output/hidden state/cell state tensors at each
92// time step.

101 void UpdateTime();
102 void UpdateBatch();
103
104 void ResetTime() { current_time_ = 0; }

106 RuntimeShape StateShape() const;

125// Calculates a single LSTM gate.
126// Implements the following formula:
127// gate = activate(FC(input) + FC(recurrent))
128// Activation is sigmoid except for the "cell" gate (configurable, usually tanh)
129template <typename ActivationType, typename WeightType, typename CellType,
130 typename BiasType>
131void CalculateLstmGate(
132 const LstmStepManager& step_info, const GateParameters& gate_params,
133 // Input FC
134 const TfLiteEvalTensor* input, const TfLiteEvalTensor* input_weight,
135 const TfLiteEvalTensor* input_bias,
136 // Recurrent FC
137 const TfLiteEvalTensor* recurrent, const TfLiteEvalTensor* recurrent_weight,
138 const TfLiteEvalTensor* recurrent_bias,
139 // Output
140 CellType* gate_output,
141 // Scratch arrays
142 CellType* fc_output_buffer, const TfLiteFusedActivation activation) {
143 const auto gate_output_shape = step_info.StateShape();
144 // Check offset validity to avoid memory overflow
145 TFLITE_DCHECK_LE(step_info.InputOffset() + step_info.InputShape().FlatSize(),
146 tflite::micro::GetTensorShape(input).FlatSize());
147 TFLITE_DCHECK_LE(
148 step_info.HiddenStateOffset() + step_info.StateShape().FlatSize(),
149 tflite::micro::GetTensorShape(recurrent).FlatSize());
150
151 // Input FC
152 FullyConnected(gate_params.input_fc_params, step_info.InputShape(),
153 tflite::micro::GetTensorData<ActivationType>(input) +
154 step_info.InputOffset(),
155 micro::GetTensorShape(input_weight),
156 tflite::micro::GetTensorData<WeightType>(input_weight),
157 tflite::micro::GetTensorShape(input_bias),
158 tflite::micro::GetOptionalTensorData<BiasType>(input_bias),
159 gate_output_shape, gate_output);
160
161 // Recurrent FC
162 FullyConnected(gate_params.recurrent_fc_params, step_info.StateShape(),
163 tflite::micro::GetTensorData<ActivationType>(recurrent) +
164 step_info.HiddenStateOffset(),
165 tflite::micro::GetTensorShape(recurrent_weight),
166 tflite::micro::GetTensorData<WeightType>(recurrent_weight),
167 tflite::micro::GetTensorShape(recurrent_bias),
168 tflite::micro::GetOptionalTensorData<BiasType>(recurrent_bias),
169 gate_output_shape, fc_output_buffer);
170
171 AddElementWise(gate_output, fc_output_buffer,
172 /*n_batch=*/gate_output_shape.DimsData()[0],
173 /*n_state=*/gate_output_shape.DimsData()[1], gate_output);
174 // Apply activation
175 switch (activation) {
176 case kTfLiteActSigmoid:
177 Sigmoid(gate_output_shape, gate_output);
178 break;
179 case kTfLiteActTanh: {
180 // Set the scale power to -12 to avoid shift
181 Tanh(/*cell_state_scale_power=*/-12, gate_output_shape, gate_output,
182 gate_output_shape, gate_output);
183 } break;
184 default:
185 // Only Sigmoid or Tanh is used.
186 TFLITE_ASSERT_FALSE;
187 }
188}

190// Update the cell state using the output from the forget gate, input gate, and
191// cell gate Formula: updated_cell_state = forget_gate_output*cell_state +
192// input_gate_output * cell_gate_output, where * denotes element wise
193// multiplication
194template <typename CellType>
195void UpdateLstmCell(const LstmStepManager& step_info,
196 TfLiteEvalTensor* cell_state,
197 // Gate outputs
198 CellType* forget_gate_output,
199 const CellType* input_gate_output,
200 const CellType* cell_gate_output,
201 // Mul parameters
202 const ArithmeticParams& forget_cell_mul_params,
203 const ArithmeticParams& input_mul_params,
204 const CellStateInfo& cell_state_info, CellType* buffer) {
205 // Check offset validity to avoid memory overflow
206 TFLITE_DCHECK_LE(
207 step_info.CellStateOffset() + step_info.StateShape().FlatSize(),
208 tflite::micro::GetTensorShape(cell_state).FlatSize());
209
210 auto cell_state_shape = step_info.StateShape();
211 // Forget Gate x Cell State
212 Mul(cell_state_shape, forget_cell_mul_params, forget_gate_output,
213 tflite::micro::GetTensorData<CellType>(cell_state) +
214 step_info.CellStateOffset(),
215 tflite::micro::GetTensorData<CellType>(cell_state) +
216 step_info.CellStateOffset());
217 // Input Gate x Cell Gate
218 Mul(cell_state_shape, input_mul_params, input_gate_output, cell_gate_output,
219 buffer);
220
221 // Update the cell state
222 AddElementWise(tflite::micro::GetTensorData<CellType>(cell_state) +
223 step_info.CellStateOffset(),
224 buffer,
225 /*n_batch=*/cell_state_shape.DimsData()[0],
226 /*n_state=*/cell_state_shape.DimsData()[1],
227 tflite::micro::GetTensorData<CellType>(cell_state) +
228 step_info.CellStateOffset());
229
230 if (cell_state_info.cell_clip > 0) {
231 Clipping(cell_state_shape.FlatSize(), cell_state_info,
232 tflite::micro::GetTensorData<CellType>(cell_state) +
233 step_info.CellStateOffset());
234 }
235}

237// Update the hidden state of the LSTM kernel using the following formula:
238// updated_hidden_state = Tanh(updated_cell_state) * output_gate_output, * means
239// element wise multiplication
240template <typename CellType, typename ActivationType>
241void UpdateLstmHidden(const LstmStepManager& step_info,
242 TfLiteEvalTensor* cell_state,
243 TfLiteEvalTensor* hidden_state,
244 const CellType* output_gate_output,
245 const ArithmeticParams& mul_params,
246 int32_t cell_state_scale_power, CellType* buffer) {
247 // Check offset validity to avoid memory overflow
248 TFLITE_DCHECK_LE(
249 step_info.CellStateOffset() + step_info.StateShape().FlatSize(),
250 tflite::micro::GetTensorShape(cell_state).FlatSize());
251 TFLITE_DCHECK_LE(
252 step_info.HiddenStateOffset() + step_info.StateShape().FlatSize(),
253 tflite::micro::GetTensorShape(hidden_state).FlatSize());
254
255 auto cell_state_shape = step_info.StateShape();
256 CellType* cell_state_data =
257 tflite::micro::GetTensorData<CellType>(cell_state) +
258 step_info.CellStateOffset();
259 // Tanh(cell_state)
260 Tanh(cell_state_scale_power, cell_state_shape, cell_state_data,
261 cell_state_shape, buffer);
262 // Update the hidden state
263 Mul(cell_state_shape, mul_params, buffer, output_gate_output,
264 tflite::micro::GetTensorData<ActivationType>(hidden_state) +
265 step_info.HiddenStateOffset());
266}

268template <typename ActivationType, typename WeightType, typename CellType,
269 typename BiasType>
270void LstmStep(const LstmStepManager& step_info, const OpDataLSTM& op_data,
271 LSTMKernelContents& kernel_content,
272 LSTMBuffers<CellType>& buffers) {
273 /*Step1: Calculate gate outputs to prepare cell state update*/
274 CellType* gate_internal_buffer = buffers.buffer3;
275 CellType* forget_gate_output = buffers.buffer0;
276 CalculateLstmGate<ActivationType, WeightType, CellType, BiasType>(
277 step_info, op_data.forget_gate_parameters,
278 // Input FC
279 kernel_content.GetInternalTensor(tflite::kLstmInputTensor),
280 kernel_content.GetInternalTensor(tflite::kLstmInputToForgetWeightsTensor),
281 kernel_content.GetInternalTensor(tflite::kLstmForgetGateBiasTensor),
282 // Recurrent FC
283 kernel_content.HiddenStateTensor(),
284 kernel_content.GetInternalTensor(
285 tflite::kLstmRecurrentToForgetWeightsTensor),
286 /*recurrent_bias*/ nullptr,
287 // Output
288 forget_gate_output,
289 // Scratch arrays
290 gate_internal_buffer, kTfLiteActSigmoid);
291
292 // Input Gate calculation;
293 CellType* input_gate_output = buffers.buffer1;
294 CalculateLstmGate<ActivationType, WeightType, CellType, BiasType>(
295 step_info, op_data.input_gate_parameters,
296 // Input FC
297 kernel_content.GetInternalTensor(tflite::kLstmInputTensor),
298 kernel_content.GetInternalTensor(tflite::kLstmInputToInputWeightsTensor),
299 kernel_content.GetInternalTensor(tflite::kLstmInputGateBiasTensor),
300 // Recurrent FC
301 kernel_content.HiddenStateTensor(),
302 kernel_content.GetInternalTensor(
303 tflite::kLstmRecurrentToInputWeightsTensor),
304 /*recurrent_bias*/ nullptr,
305 // Output
306 input_gate_output,
307 // Scratch arrays
308 gate_internal_buffer, kTfLiteActSigmoid);
309
310 // Cell Gate calculation
311 CellType* cell_gate_output = buffers.buffer2;
312 CalculateLstmGate<ActivationType, WeightType, CellType, BiasType>(
313 step_info, op_data.cell_gate_parameters,
314 // Input FC
315 kernel_content.GetInternalTensor(tflite::kLstmInputTensor),
316 kernel_content.GetInternalTensor(tflite::kLstmInputToCellWeightsTensor),
317 kernel_content.GetInternalTensor(tflite::kLstmCellGateBiasTensor),
318 // Recurrent FC
319 kernel_content.HiddenStateTensor(),
320 kernel_content.GetInternalTensor(
321 tflite::kLstmRecurrentToCellWeightsTensor),
322 /*recurrent_bias*/ nullptr,
323 // Output
324 cell_gate_output,
325 // Scratch arrays
326 gate_internal_buffer, op_data.cell_gate_nonlinear_type);
327
328 /*Step2: update the cell state */
329 const InterGateParameters& inter_gate_params = op_data.inter_gate_parameters;
330 CellType* updated_input_buffer = buffers.buffer1; // reuse buffer
331
332 UpdateLstmCell<CellType>(step_info, kernel_content.CellStateTensor(),
333 forget_gate_output, input_gate_output,
334 cell_gate_output,
335 inter_gate_params.forget_cell_mul_params,
336 inter_gate_params.input_mul_params,
337 op_data.cell_state_info, updated_input_buffer);
338
339 /*Step3: update the hidden state */
340 CellType* output_gate_output = buffers.buffer1; // reuse buffer
341 CalculateLstmGate<ActivationType, WeightType, CellType, BiasType>(
342 step_info, op_data.output_gate_parameters,
343 // Input FC
344 kernel_content.GetInternalTensor(tflite::kLstmInputTensor),
345 kernel_content.GetInternalTensor(tflite::kLstmInputToOutputWeightsTensor),
346 kernel_content.GetInternalTensor(tflite::kLstmOutputGateBiasTensor),
347 // Recurrent FC
348 kernel_content.HiddenStateTensor(),
349 kernel_content.GetInternalTensor(
350 tflite::kLstmRecurrentToOutputWeightsTensor),
351 /*recurrent_bias*/ nullptr,
352 // Output
353 output_gate_output,
354 // Scratch arrays
355 gate_internal_buffer, kTfLiteActSigmoid);
356
357 CellType* tanh_activated_cell_buffer = buffers.buffer0; // reuse buffer
358 tflite::lstm_internal::UpdateLstmHidden<CellType, ActivationType>(
359 step_info, kernel_content.CellStateTensor(),
360 kernel_content.HiddenStateTensor(), output_gate_output,
361 inter_gate_params.output_mul_params,
362 op_data.cell_state_info.cell_state_scale_power,
363 tanh_activated_cell_buffer);
364
365 /*Step4: copy the update the hidden state to output*/
366 // Check offset validity to avoid memory overflow
367 TFLITE_DCHECK_LE(
368 step_info.OutputOffset() + step_info.StateShape().FlatSize(),
369 tflite::micro::GetTensorShape(kernel_content.output_tensor).FlatSize());
370 // record the output (from the updated hidden state)
371 ActivationType* output_ptr = tflite::micro::GetTensorData<ActivationType>(
372 kernel_content.output_tensor);
373 const auto* hidden_state = kernel_content.HiddenStateTensor();
374 std::memcpy(output_ptr + step_info.OutputOffset(),
375 tflite::micro::GetTensorData<ActivationType>(hidden_state) +
376 step_info.HiddenStateOffset(),
377 step_info.StateShape().FlatSize() * sizeof(ActivationType));
378}

382// Evaulate the LSTM kernel with (potential) multi-steps and multi-batch input
383// Since
384template <typename ActivationType, typename WeightType, typename CellType,
385 typename BiasType>
386TfLiteStatus EvalLstm(const OpDataLSTM& op_data,
387 LSTMKernelContents& kernel_content,
388 LSTMBuffers<CellType>& buffers) {
389 lstm_internal::LstmStepManager step_info(&op_data.size_info);
390 const auto& size_info = op_data.size_info;
391 // time is the first dimention, enable batch computation
392 if (size_info.time_major) {
393 for (int t = 0; t < size_info.time_steps; t++) {
394 lstm_internal::LstmStep<ActivationType, WeightType, CellType, BiasType>(
395 step_info, op_data, kernel_content, buffers);
396 // prepare for the next time step
397 step_info.UpdateTime();
398 }
399 } else {
400 // batch first, unable to size the input data. single batch inference
401 for (int b = 0; b < size_info.batch_size; b++) {
402 for (int t = 0; t < size_info.time_steps; t++) {
403 lstm_internal::LstmStep<ActivationType, WeightType, CellType, BiasType>(
404 step_info, op_data, kernel_content, buffers);
405 // prepare for the next time step
406 step_info.UpdateTime();
407 }
408 // prepare for the next batch
409 step_info.UpdateBatch();
410 step_info.ResetTime();
411 }
412 }
413 return kTfLiteOk;
414}
