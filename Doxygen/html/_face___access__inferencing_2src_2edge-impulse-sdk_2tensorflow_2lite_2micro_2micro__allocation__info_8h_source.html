27// Used to hold information used during allocation calculations.
28struct AllocationInfo {
29 size_t bytes;
30 void** output_ptr;
31 int first_created;
32 int last_used;
33 int32_t offline_offset;
34 bool needs_allocating;
35};

37// Used to hold the allocation info list and related metadata for the entire
38// graph (including subgraphs). Since all subgraphs are planned together, the
39// allocation info list contains allocations for all subgraphs. Track the offset
40// into this list for each subgraph then reserve space to track all allocations.
41//
42// The AllocationInfo list is a contiguous list of allocations across all
43// subgraphs and scratch buffers. Each element here is marked as
44// s<subgraph index>t<tensor index>. The following is a possible
45// AllocationInfo list:
46// [s0t0, s0t1, s1t0, s2t1, s1t2, s3t0, s3t1, scratch0, scratch1, scratch2]
47//
48// For this example, the subgraph offsets would be [0, 2, 5] and the scratch
49// offset would be 7.
50struct GraphAllocationInfo {
51 AllocationInfo* allocation_info;
52 size_t allocation_info_count;
53 size_t* subgraph_offsets;
54 size_t scratch_offset;
55 size_t tensor_count;
56 size_t scratch_buffer_count;
57};

59// A helper class to construct AllocationInfo array. This array contains the
60// lifetime of tensors / scratch_buffer and will be used to calculate the memory
61// plan. Methods need to be called in order from `Create`, Init`, `Add*`, to
62// `Finish`.
63class AllocationInfoBuilder {
64 public:
65 AllocationInfoBuilder(const Model* model,
66 INonPersistentBufferAllocator* non_persistent_allocator)
67 : model_(model), non_persistent_allocator_(non_persistent_allocator) {}
68
69 // Check if model contains offline planned buffer offsets.
70 // - If there's no metadata available, offline_planner_offsets is not set
71 // - If there's metadata available, offline_planner_offsets will point to the
72 // first offset in the metadata buffer list.
73 TfLiteStatus GetOfflinePlannedOffsets(
74 const int32_t** offline_planner_offsets);
75
76 // Allocate memory for the allocation info array as well as offsets into that
77 // array for each subgraph.
78 TfLiteStatus CreateAllocationInfo(int scratch_buffer_request_count);
79
80 // Release memory used for the allocation info array.
81 TfLiteStatus FreeAllocationInfo();
82
83 // Initialize AllocationInfo for all tensors and scratch buffers in the graph.
84 TfLiteStatus InitializeAllocationInfo(const int32_t* offline_offsets,
85 SubgraphAllocations* allocations);
86
87 // Mark the scope of each tensor and scratch buffer across the graph. Enter
88 // all possible subgraphs invoked by each control flow operator. This method
89 // marks the maximum lifetime of each buffer so that tensors are correctly
90 // planned for all valid invocation flows.
91 TfLiteStatus MarkAllocationLifetimes(
92 int subgraph_idx, internal::ScratchBufferRequest* scratch_buffer_request,
93 ScratchBufferHandle* scratch_buffer_handles,
94 SubgraphAllocations* allocations);
95
96 // Identify control flow operators and recursively mark all subgraphs which
97 // that operator can invoke. The lifetime of all tensors within a subgraph
98 // can only be extended. The order of subgraph invocation does not matter
99 // since subgraphs within the same control flow operator are executed
100 // within their own allocation scope (planned buffers in a subgraph cannot
101 // persist beyond the end of that subgraph's invocation).
102 TfLiteStatus MarkSubgraphLifetimesIfNecessary(
103 const Operator* op,
104 internal::ScratchBufferRequest* scratch_buffer_requests,
105 ScratchBufferHandle* scratch_buffer_handles,
106 SubgraphAllocations* allocations);
107
108 // Returns the number of allocations.
109 int AllocationCount() const { return info_.allocation_info_count; }
110
111 // Returns a pointer to the built AllocationInfo array.
112 AllocationInfo* Finish() const { return info_.allocation_info; }
113
114 private:
115 // Mark the given Allocation info as first created at the specified allocation
116 // scope count. Only the first creation must be recorded since the allocation
117 // scope count monotonically increases throughout the lifetime marking
118 // process.
119 void UpdateFirstCreated(AllocationInfo* current, int allocation_scope_count);
120
121 // Mark the given AllocationInfo as last used at the specified allocation
122 // scope
123 // count. Update the last used marker every time, since the allocation scope
124 // count monotonically increases through the lifetime marking process.
125 void UpdateLastUsed(AllocationInfo* current, int allocation_scope_count);
126
127 // Validate if a subgraph satisfies assumptions.
128 TfLiteStatus ValidateSubgraph(const SubGraph* subgraph,
129 TfLiteEvalTensor* eval_tensors);
130
131 const tflite::Model* model_ = nullptr;
132 INonPersistentBufferAllocator* non_persistent_allocator_ = nullptr;
133 GraphAllocationInfo info_;
134 int allocation_scope_count_ = 0;
135};
