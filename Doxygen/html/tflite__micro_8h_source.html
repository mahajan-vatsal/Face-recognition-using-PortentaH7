The system captures facial images, processes them locally using an AI model deployed on the Portenta H7 and determines access based on authorised personnel.

60static EI_IMPULSE_ERROR inference_tflite_setup(
61 ei_learning_block_config_tflite_graph_t *block_config,
62 uint64_t *ctx_start_us,
63 TfLiteTensor** input,
64 TfLiteTensor** output,
65 TfLiteTensor** output_labels,
66 TfLiteTensor** output_scores,
67 tflite::MicroInterpreter** micro_interpreter,
68 ei_unique_ptr_t& p_tensor_arena) {
69
70 *ctx_start_us = ei_read_timer_us();
71
72 ei_config_tflite_graph_t *graph_config = (ei_config_tflite_graph_t*)block_config->graph_config;
73
74#ifdef EI_CLASSIFIER_ALLOCATION_STATIC
75 // Assign a no-op lambda to the "free" function in case of static arena
76 static uint8_t tensor_arena[EI_CLASSIFIER_TFLITE_LARGEST_ARENA_SIZE] ALIGN(16);
77 p_tensor_arena = ei_unique_ptr_t(tensor_arena, [](void*){});
78#else
79 // Create an area of memory to use for input, output, and intermediate arrays.
80 uint8_t *tensor_arena = (uint8_t*)ei_aligned_calloc(16, graph_config->arena_size);
81 if (tensor_arena == NULL) {
82 ei_printf("Failed to allocate TFLite arena (%zu bytes)\n", graph_config->arena_size);
83 return EI_IMPULSE_TFLITE_ARENA_ALLOC_FAILED;
84 }
85 p_tensor_arena = ei_unique_ptr_t(tensor_arena, ei_aligned_free);
86#endif
87
88 static bool tflite_first_run = true;
89 static uint8_t *model_arr = NULL;
90
91 if (model_arr != graph_config->model) {
92 tflite_first_run = true;
93 model_arr = (uint8_t*)graph_config->model;
94 }
95
96 static const tflite::Model* model = nullptr;
97
98 // ======
99 // Initialization code start
100 // This part can be run once, but that would require the TFLite arena
101 // to be allocated at all times, which is not ideal (e.g. when doing MFCC)
102 // ======
103 if (tflite_first_run) {
104 // Map the model into a usable data structure. This doesn't involve any
105 // copying or parsing, it's a very lightweight operation.
106 model = tflite::GetModel(graph_config->model);
107 if (model->version() != TFLITE_SCHEMA_VERSION) {
108 ei_printf(
109 "Model provided is schema version %d not equal "
110 "to supported version %d.",
111 model->version(), TFLITE_SCHEMA_VERSION);
112 return EI_IMPULSE_TFLITE_ERROR;
113 }
114 tflite_first_run = false;
115 }
116
117#ifdef EI_TFLITE_RESOLVER
118 EI_TFLITE_RESOLVER
119#else
120 static tflite::AllOpsResolver resolver; // needs static to match the life of the interpreter
121#endif
122
123 // Build an interpreter to run the model with.
124 tflite::MicroInterpreter *interpreter = new tflite::MicroInterpreter(
125 model, resolver, tensor_arena, graph_config->arena_size);
126
127 *micro_interpreter = interpreter;
128
129 // Allocate memory from the tensor_arena for the model's tensors.
130 TfLiteStatus allocate_status = interpreter->AllocateTensors(true);
131 if (allocate_status != kTfLiteOk) {
132 ei_printf("AllocateTensors() failed");
133 return EI_IMPULSE_TFLITE_ERROR;
134 }
135
136 // Obtain pointers to the model's input and output tensors.
137 *input = interpreter->input(0);
138 *output = interpreter->output(block_config->output_data_tensor);
139
140 if (block_config->object_detection_last_layer == EI_CLASSIFIER_LAST_LAYER_SSD) {
141 *output_scores = interpreter->output(block_config->output_score_tensor);
142 *output_labels = interpreter->output(block_config->output_labels_tensor);
143 }
144
145 if (tflite_first_run) {
146 tflite_first_run = false;
147 }
148
149 return EI_IMPULSE_OK;
150}

164static EI_IMPULSE_ERROR inference_tflite_run(
165 const ei_impulse_t *impulse,
166 ei_learning_block_config_tflite_graph_t *block_config,
167 uint64_t ctx_start_us,
168 TfLiteTensor* output,
169 TfLiteTensor* labels_tensor,
170 TfLiteTensor* scores_tensor,
171 tflite::MicroInterpreter* interpreter,
172 uint8_t* tensor_arena,
173 ei_impulse_result_t *result,
174 bool debug) {
175
176
177 // Run inference, and report any error
178 TfLiteStatus invoke_status = interpreter->Invoke();
179 if (invoke_status != kTfLiteOk) {
180 delete interpreter;
181 ei_printf("Invoke failed (%d)\n", invoke_status);
182 return EI_IMPULSE_TFLITE_ERROR;
183 }
184
185 uint64_t ctx_end_us = ei_read_timer_us();
186
187 result->timing.classification_us = ctx_end_us - ctx_start_us;
188 result->timing.classification = (int)(result->timing.classification_us / 1000);
189
190 // Read the predicted y value from the model's output tensor
191 if (debug) {
192 ei_printf("Predictions (time: %d ms.):\n", result->timing.classification);
193 }
194
195 EI_IMPULSE_ERROR fill_res = fill_result_struct_from_output_tensor_tflite(
196 impulse, block_config, output, labels_tensor, scores_tensor, result, debug);
197
198 delete interpreter;
199
200 if (fill_res != EI_IMPULSE_OK) {
201 return fill_res;
202 }
203
204 if (ei_run_impulse_check_canceled() == EI_IMPULSE_CANCELED) {
205 return EI_IMPULSE_CANCELED;
206 }
207
208 return EI_IMPULSE_OK;
209}

221EI_IMPULSE_ERROR run_nn_inference_from_dsp(
222 ei_learning_block_config_tflite_graph_t *config,
223 signal_t *signal,
224 matrix_t *output_matrix)
225{
226 TfLiteTensor* input;
227 TfLiteTensor* output;
228 TfLiteTensor* output_scores;
229 TfLiteTensor* output_labels;
230 uint64_t ctx_start_us = ei_read_timer_us();
231 ei_unique_ptr_t p_tensor_arena(nullptr, ei_aligned_free);
232
233 tflite::MicroInterpreter* interpreter;
234 EI_IMPULSE_ERROR init_res = inference_tflite_setup(
235 config,
236 &ctx_start_us,
237 &input, &output,
238 &output_labels,
239 &output_scores,
240 &interpreter, p_tensor_arena);
241
242 if (init_res != EI_IMPULSE_OK) {
243 return init_res;
244 }
245
246 auto input_res = fill_input_tensor_from_signal(signal, input);
247 if (input_res != EI_IMPULSE_OK) {
248 return input_res;
249 }
250
251 // Run inference, and report any error
252 TfLiteStatus invoke_status = interpreter->Invoke();
253 if (invoke_status != kTfLiteOk) {
254 ei_printf("Invoke failed (%d)\n", invoke_status);
255 return EI_IMPULSE_TFLITE_ERROR;
256 }
257
258 auto output_res = fill_output_matrix_from_tensor(output, output_matrix);
259 if (output_res != EI_IMPULSE_OK) {
260 return output_res;
261 }
262
263 delete interpreter;
264
265 return EI_IMPULSE_OK;
266}

277EI_IMPULSE_ERROR run_nn_inference(
278 const ei_impulse_t *impulse,
279 ei_feature_t *fmatrix,
280 uint32_t learn_block_index,
281 uint32_t* input_block_ids,
282 uint32_t input_block_ids_size,
283 ei_impulse_result_t *result,
284 void *config_ptr,
285 bool debug = false)
286{
287 ei_learning_block_config_tflite_graph_t *block_config = (ei_learning_block_config_tflite_graph_t*)config_ptr;
288
289 TfLiteTensor* input;
290 TfLiteTensor* output;
291 TfLiteTensor* output_scores;
292 TfLiteTensor* output_labels;
293 uint64_t ctx_start_us = ei_read_timer_us();
294 ei_unique_ptr_t p_tensor_arena(nullptr, ei_aligned_free);
295
296 tflite::MicroInterpreter* interpreter;
297 EI_IMPULSE_ERROR init_res = inference_tflite_setup(
298 block_config,
299 &ctx_start_us,
300 &input, &output,
301 &output_labels,
302 &output_scores,
303 &interpreter,
304 p_tensor_arena);
305
306 if (init_res != EI_IMPULSE_OK) {
307 return init_res;
308 }
309
310 uint8_t* tensor_arena = static_cast<uint8_t*>(p_tensor_arena.get());
311
312 size_t mtx_size = impulse->dsp_blocks_size + impulse->learning_blocks_size;
313 auto input_res = fill_input_tensor_from_matrix(fmatrix, input, input_block_ids, input_block_ids_size, mtx_size);
314 if (input_res != EI_IMPULSE_OK) {
315 return input_res;
316 }
317
318 EI_IMPULSE_ERROR run_res = inference_tflite_run(
319 impulse,
320 block_config,
321 ctx_start_us,
322 output,
323 output_labels,
324 output_scores,
325 interpreter, tensor_arena, result, debug);
326
327 if (result->copy_output) {
328 auto output_res = fill_output_matrix_from_tensor(output, fmatrix[impulse->dsp_blocks_size + learn_block_index].matrix);
329 if (output_res != EI_IMPULSE_OK) {
330 return output_res;
331 }
332 }
333
334 if (run_res != EI_IMPULSE_OK) {
335 return run_res;
336 }
337
338 return EI_IMPULSE_OK;
339}

355
356 memset(result, 0, sizeof(ei_impulse_result_t));
357
358 uint64_t ctx_start_us;
359 TfLiteTensor* input;

371 &output_scores,
372 &interpreter,
373 p_tensor_arena);
374
375 if (init_res != EI_IMPULSE_OK) {
376 return init_res;
377 }
378
379 if (input->type != TfLiteType::kTfLiteInt8 && input->type != TfLiteType::kTfLiteUInt8) {
380 return EI_IMPULSE_ONLY_SUPPORTED_FOR_IMAGES;
381 }
382
383 uint64_t dsp_start_us = ei_read_timer_us();
384
385 // features matrix maps around the input tensor to not allocate any memory
386 ei::matrix_i8_t features_matrix(1, impulse->nn_input_frame_size, input->data.int8);
387
388 // run DSP process and quantize automatically
389 int ret = extract_image_features_quantized(signal, &features_matrix, impulse->dsp_blocks[0].config, input->params.scale, input->params.zero_point,
390 impulse->frequency, impulse->learning_blocks[0].image_scaling);
391 if (ret != EIDSP_OK) {
392 ei_printf("ERR: Failed to run DSP process (%d)\n", ret);
393 return EI_IMPULSE_DSP_ERROR;
394 }
395
396 if (ei_run_impulse_check_canceled() == EI_IMPULSE_CANCELED) {
397 return EI_IMPULSE_CANCELED;
398 }
399
400 result->timing.dsp_us = ei_read_timer_us() - dsp_start_us;
401 result->timing.dsp = (int)(result->timing.dsp_us / 1000);
402
403 if (debug) {
404 ei_printf("Features (%d ms.): ", result->timing.dsp);
405 for (size_t ix = 0; ix < features_matrix.cols; ix++) {
406 ei_printf_float((features_matrix.buffer[ix] - input->params.zero_point) * input->params.scale);
407 ei_printf(" ");
408 }
409 ei_printf("\n");
410 }
411
412 ctx_start_us = ei_read_timer_us();
413
414 EI_IMPULSE_ERROR run_res = inference_tflite_run(impulse,
415 block_config,
416 ctx_start_us,
417 output,
418 output_labels,
419 output_scores,
420 interpreter,
421 static_cast<uint8_t*>(p_tensor_arena.get()),
422 result, debug);
423
424 if (run_res != EI_IMPULSE_OK) {
425 return run_res;
426 }
427
428 return EI_IMPULSE_OK;
429}
