64 static int filterbanks(
65#if EIDSP_QUANTIZE_FILTERBANK
66 quantized_matrix_t *filterbanks,
67#else
68 matrix_t *filterbanks,
69#endif
70 uint16_t num_filter, int coefficients, uint32_t sampling_freq,
71 uint32_t low_freq, uint32_t high_freq,
72 bool output_transposed = false
73 )
74 {
75 const size_t mels_mem_size = (num_filter + 2) * sizeof(float);
76 const size_t hertz_mem_size = (num_filter + 2) * sizeof(float);
77 const size_t freq_index_mem_size = (num_filter + 2) * sizeof(int);
78
79 float *mels = (float*)ei_dsp_malloc(mels_mem_size);
80 if (!mels) {
81 EIDSP_ERR(EIDSP_OUT_OF_MEM);
82 }
83
84 if (filterbanks->rows != num_filter || filterbanks->cols != static_cast<uint32_t>(coefficients)) {
85 EIDSP_ERR(EIDSP_MATRIX_SIZE_MISMATCH);
86 }
87
88#if EIDSP_QUANTIZE_FILTERBANK
89 memset(filterbanks->buffer, 0, filterbanks->rows * filterbanks->cols * sizeof(uint8_t));
90#else
91 memset(filterbanks->buffer, 0, filterbanks->rows * filterbanks->cols * sizeof(float));
92#endif
93
94 // Computing the Mel filterbank
95 // converting the upper and lower frequencies to Mels.
96 // num_filter + 2 is because for num_filter filterbanks we need
97 // num_filter+2 point.
98 numpy::linspace(
99 functions::frequency_to_mel(static_cast<float>(low_freq)),
100 functions::frequency_to_mel(static_cast<float>(high_freq)),
101 num_filter + 2,
102 mels);
103
104 // we should convert Mels back to Hertz because the start and end-points
105 // should be at the desired frequencies.
106 float *hertz = (float*)ei_dsp_malloc(hertz_mem_size);
107 if (!hertz) {
108 ei_dsp_free(mels, mels_mem_size);
109 EIDSP_ERR(EIDSP_OUT_OF_MEM);
110 }
111 for (uint16_t ix = 0; ix < num_filter + 2; ix++) {
112 hertz[ix] = functions::mel_to_frequency(mels[ix]);
113 if (hertz[ix] < low_freq) {
114 hertz[ix] = low_freq;
115 }
116 if (hertz[ix] > high_freq) {
117 hertz[ix] = high_freq;
118 }
119
120 // here is a really annoying bug in Speechpy which calculates the frequency index wrong for the last bucket
121 // the last 'hertz' value is not 8,000 (with sampling rate 16,000) but 7,999.999999
122 // thus calculating the bucket to 64, not 65.
123 // we're adjusting this here a tiny bit to ensure we have the same result
124 if (ix == num_filter + 2 - 1) {
125 hertz[ix] -= 0.001;
126 }
127 }
128 ei_dsp_free(mels, mels_mem_size);
129
130 // The frequency resolution required to put filters at the
131 // exact points calculated above should be extracted.
132 // So we should round those frequencies to the closest FFT bin.
133 int *freq_index = (int*)ei_dsp_malloc(freq_index_mem_size);
134 if (!freq_index) {
135 ei_dsp_free(hertz, hertz_mem_size);
136 EIDSP_ERR(EIDSP_OUT_OF_MEM);
137 }
138 for (uint16_t ix = 0; ix < num_filter + 2; ix++) {
139 freq_index[ix] = static_cast<int>(floor((coefficients + 1) * hertz[ix] / sampling_freq));
140 }
141 ei_dsp_free(hertz, hertz_mem_size);
142
143 for (size_t i = 0; i < num_filter; i++) {
144 int left = freq_index[i];
145 int middle = freq_index[i + 1];
146 int right = freq_index[i + 2];
147
148 EI_DSP_MATRIX(z, 1, (right - left + 1));
149 if (!z.buffer) {
150 ei_dsp_free(freq_index, freq_index_mem_size);
151 EIDSP_ERR(EIDSP_OUT_OF_MEM);
152 }
153 numpy::linspace(left, right, (right - left + 1), z.buffer);
154 functions::triangle(z.buffer, (right - left + 1), left, middle, right);
155
156 // so... z now contains some values that we need to overwrite in the filterbank
157 for (int zx = 0; zx < (right - left + 1); zx++) {
158 size_t index = (i * filterbanks->cols) + (left + zx);
159
160 if (output_transposed) {
161 index = ((left + zx) * filterbanks->rows) + i;
162 }
163
164#if EIDSP_QUANTIZE_FILTERBANK
165 filterbanks->buffer[index] = numpy::quantize_zero_one(z.buffer[zx]);
166#else
167 filterbanks->buffer[index] = z.buffer[zx];
168#endif
169 }
170 }
171
172 if (output_transposed) {
173 uint16_t r = filterbanks->rows;
174 filterbanks->rows = filterbanks->cols;
175 filterbanks->cols = r;
176 }
177
178 ei_dsp_free(freq_index, freq_index_mem_size);
179
180 return EIDSP_OK;
181 }

191 static int get_fft_bin_from_hertz(uint16_t fft_size, float hertz, uint32_t sampling_freq)
192 {
193 return static_cast<int>(floor((fft_size + 1) * hertz / sampling_freq));
194 }

216 static int mfe(matrix_t *out_features, matrix_t *out_energies,
217 signal_t *signal,
218 uint32_t sampling_frequency,
219 float frame_length, float frame_stride, uint16_t num_filters,
220 uint16_t fft_length, uint32_t low_frequency, uint32_t high_frequency,
221 uint16_t version
222 )
223 {
224 int ret = 0;
225
226 if (high_frequency == 0) {
227 high_frequency = sampling_frequency / 2;
228 }
229
230 if (version<4) {
231 if (low_frequency == 0) {
232 low_frequency = 300;
233 }
234 }
235
236 stack_frames_info_t stack_frame_info = { 0 };
237 stack_frame_info.signal = signal;
238
239 ret = processing::stack_frames(
240 &stack_frame_info,
241 sampling_frequency,
242 frame_length,
243 frame_stride,
244 false,
245 version
246 );
247 if (ret != 0) {
248 EIDSP_ERR(ret);
249 }
250
251 if (stack_frame_info.frame_ixs.size() != out_features->rows) {
252 EIDSP_ERR(EIDSP_MATRIX_SIZE_MISMATCH);
253 }
254
255 if (num_filters != out_features->cols) {
256 EIDSP_ERR(EIDSP_MATRIX_SIZE_MISMATCH);
257 }
258
259 if (out_energies) {
260 if (stack_frame_info.frame_ixs.size() != out_energies->rows || out_energies->cols != 1) {
261 EIDSP_ERR(EIDSP_MATRIX_SIZE_MISMATCH);
262 }
263 }
264
265 for (uint32_t i = 0; i < out_features->rows * out_features->cols; i++) {
266 *(out_features->buffer + i) = 0;
267 }
268
269 const size_t power_spectrum_frame_size = (fft_length / 2 + 1);
270 // Computing the Mel filterbank
271 // converting the upper and lower frequencies to Mels.
272 // num_filter + 2 is because for num_filter filterbanks we need
273 // num_filter+2 point.
274 float *mels;
275 const int MELS_SIZE = num_filters + 2;
276 const size_t mem_size = MELS_SIZE * sizeof(float);
277 mels = (float*)ei_dsp_calloc(MELS_SIZE, sizeof(float));
278 EI_ERR_AND_RETURN_ON_NULL(mels, EIDSP_OUT_OF_MEM);
279 ei_unique_ptr_t __ptr__(mels,[mem_size](void* ptr){ei::ei_dsp_free_func(ptr, mem_size);});
280 uint16_t* bins = reinterpret_cast<uint16_t*>(mels); // alias the mels array so we can reuse the space
281
282 numpy::linspace(
283 functions::frequency_to_mel(static_cast<float>(low_frequency)),
284 functions::frequency_to_mel(static_cast<float>(high_frequency)),
285 num_filters + 2,
286 mels);
287
288 uint16_t max_bin = version >= 4 ? fft_length : power_spectrum_frame_size; // preserve a bug in v<4
289 // go to -1 size b/c special handling, see after
290 for (uint16_t ix = 0; ix < MELS_SIZE-1; ix++) {
291 mels[ix] = functions::mel_to_frequency(mels[ix]);
292 if (mels[ix] < low_frequency) {
293 mels[ix] = low_frequency;
294 }
295 if (mels[ix] > high_frequency) {
296 mels[ix] = high_frequency;
297 }
298 bins[ix] = get_fft_bin_from_hertz(max_bin, mels[ix], sampling_frequency);
299 }
300
301 // here is a really annoying bug in Speechpy which calculates the frequency index wrong for the last bucket
302 // the last 'hertz' value is not 8,000 (with sampling rate 16,000) but 7,999.999999
303 // thus calculating the bucket to 64, not 65.
304 // we're adjusting this here a tiny bit to ensure we have the same result
305 mels[MELS_SIZE-1] = functions::mel_to_frequency(mels[MELS_SIZE-1]);
306 if (mels[MELS_SIZE-1] > high_frequency) {
307 mels[MELS_SIZE-1] = high_frequency;
308 }
309 mels[MELS_SIZE-1] -= 0.001;
310 bins[MELS_SIZE-1] = get_fft_bin_from_hertz(max_bin, mels[MELS_SIZE-1], sampling_frequency);
311
312 EI_DSP_MATRIX(power_spectrum_frame, 1, power_spectrum_frame_size);
313 if (!power_spectrum_frame.buffer) {
314 EIDSP_ERR(EIDSP_OUT_OF_MEM);
315 }
316
317 // get signal data from the audio file
318 EI_DSP_MATRIX(signal_frame, 1, stack_frame_info.frame_length);
319
320 for (size_t ix = 0; ix < stack_frame_info.frame_ixs.size(); ix++) {
321 // don't read outside of the audio buffer... we'll automatically zero pad then
322 size_t signal_offset = stack_frame_info.frame_ixs.at(ix);
323 size_t signal_length = stack_frame_info.frame_length;
324 if (signal_offset + signal_length > stack_frame_info.signal->total_length) {
325 signal_length = signal_length -
326 (stack_frame_info.signal->total_length - (signal_offset + signal_length));
327 }
328
329 ret = stack_frame_info.signal->get_data(
330 signal_offset,
331 signal_length,
332 signal_frame.buffer
333 );
334 if (ret != 0) {
335 EIDSP_ERR(ret);
336 }
337
338 ret = numpy::power_spectrum(
339 signal_frame.buffer,
340 stack_frame_info.frame_length,
341 power_spectrum_frame.buffer,
342 power_spectrum_frame_size,
343 fft_length
344 );
345
346 if (ret != 0) {
347 EIDSP_ERR(ret);
348 }
349
350 float energy = numpy::sum(power_spectrum_frame.buffer, power_spectrum_frame_size);
351 if (energy == 0) {
352 energy = 1e-10;
353 }
354
355 if (out_energies) {
356 out_energies->buffer[ix] = energy;
357 }
358
359 auto row_ptr = out_features->get_row_ptr(ix);
360 for (size_t i = 0; i < num_filters; i++) {
361 size_t left = bins[i];
362 size_t middle = bins[i+1];
363 size_t right = bins[i+2];
364
365 assert(right < power_spectrum_frame_size);
366 // now we have weights and locations to move from fft to mel sgram
367 // both left and right become zero weights, so skip them
368
369 // middle always has weight of 1.0
370 // since we skip left and right, if left = middle we need to handle that
371 row_ptr[i] = power_spectrum_frame.buffer[middle];
372
373 for (size_t bin = left+1; bin < right; bin++) {
374 if (bin < middle) {
375 row_ptr[i] +=
376 ((static_cast<float>(bin) - left) / (middle - left)) * // weight *
377 power_spectrum_frame.buffer[bin];
378 }
379 // intentionally skip middle, handled above
380 if (bin > middle) {
381 row_ptr[i] +=
382 ((right - static_cast<float>(bin)) / (right - middle)) * // weight *
383 power_spectrum_frame.buffer[bin];
384 }
385 }
386 }
387
388 if (ret != 0) {
389 EIDSP_ERR(ret);
390 }
391 }
392
393 numpy::zero_handling(out_features);
394
395 return EIDSP_OK;
396 }

419 signal_t *signal,

421 float frame_length, float frame_stride, uint16_t num_filters,

423 uint16_t version
424 )
425 {
426 int ret = 0;
427

429 high_frequency = sampling_frequency / 2;
430 }
431

433 low_frequency = 300;
434 }
435

456 EIDSP_ERR(EIDSP_MATRIX_SIZE_MISMATCH);
457 }
458
459 if (out_energies) {
460 if (stack_frame_info.frame_ixs.size() != out_energies->rows || out_energies->cols != 1) {
461 EIDSP_ERR(EIDSP_MATRIX_SIZE_MISMATCH);
462 }
463 }
464
465 for (uint32_t i = 0; i < out_features->rows * out_features->cols; i++) {

467 }
468

472 // whenever they happen, but OK...
473#if EIDSP_QUANTIZE_FILTERBANK
474 EI_DSP_QUANTIZED_MATRIX(filterbanks, num_filters, coefficients, &numpy::dequantize_zero_one);

484 if (ret != 0) {
485 EIDSP_ERR(ret);
486 }
487 for (size_t ix = 0; ix < stack_frame_info.frame_ixs.size(); ix++) {
488 size_t power_spectrum_frame_size = (fft_length / 2 + 1);
489
490 EI_DSP_MATRIX(power_spectrum_frame, 1, power_spectrum_frame_size);
491 if (!power_spectrum_frame.buffer) {
492 EIDSP_ERR(EIDSP_OUT_OF_MEM);
493 }
494
495 // get signal data from the audio file
496 EI_DSP_MATRIX(signal_frame, 1, stack_frame_info.frame_length);
497
498 // don't read outside of the audio buffer... we'll automatically zero pad then
499 size_t signal_offset = stack_frame_info.frame_ixs.at(ix);
500 size_t signal_length = stack_frame_info.frame_length;
501 if (signal_offset + signal_length > stack_frame_info.signal->total_length) {
502 signal_length = signal_length -
503 (stack_frame_info.signal->total_length - (signal_offset + signal_length));
504 }
505
506 ret = stack_frame_info.signal->get_data(
507 signal_offset,
508 signal_length,
509 signal_frame.buffer
510 );
511 if (ret != 0) {
512 EIDSP_ERR(ret);
513 }
514
515 ret = numpy::power_spectrum(
516 signal_frame.buffer,
517 stack_frame_info.frame_length,
518 power_spectrum_frame.buffer,
519 power_spectrum_frame_size,
520 fft_length
521 );
522
523 if (ret != 0) {
524 EIDSP_ERR(ret);
525 }
526
527 float energy = numpy::sum(power_spectrum_frame.buffer, power_spectrum_frame_size);
528 if (energy == 0) {
529 energy = 1e-10;
530 }
531
532 if (out_energies) {
533 out_energies->buffer[ix] = energy;
534 }
535
536 // calculate the out_features directly here
537 ret = numpy::dot_by_row(
538 ix,
539 power_spectrum_frame.buffer,
540 power_spectrum_frame_size,
541 &filterbanks,
542 out_features
543 );
544
545 if (ret != 0) {
546 EIDSP_ERR(ret);
547 }
548 }
549
550 numpy::zero_handling(out_features);
551
552 return EIDSP_OK;
553 }

568 static int spectrogram(matrix_t *out_features,
569 signal_t *signal, float sampling_frequency,
570 float frame_length, float frame_stride, uint16_t fft_length,
571 uint16_t version
572 )
573 {
574 int ret = 0;
575
576 stack_frames_info_t stack_frame_info = { 0 };
577 stack_frame_info.signal = signal;
578
579 ret = processing::stack_frames(
580 &stack_frame_info,
581 sampling_frequency,
582 frame_length,
583 frame_stride,
584 false,
585 version
586 );
587 if (ret != 0) {
588 EIDSP_ERR(ret);
589 }
590
591 if (stack_frame_info.frame_ixs.size() != out_features->rows) {
592 EIDSP_ERR(EIDSP_MATRIX_SIZE_MISMATCH);
593 }
594
595 uint16_t coefficients = fft_length / 2 + 1;
596
597 if (coefficients != out_features->cols) {
598 EIDSP_ERR(EIDSP_MATRIX_SIZE_MISMATCH);
599 }
600
601 for (uint32_t i = 0; i < out_features->rows * out_features->cols; i++) {
602 *(out_features->buffer + i) = 0;
603 }
604
605 for (size_t ix = 0; ix < stack_frame_info.frame_ixs.size(); ix++) {
606 // get signal data from the audio file
607 EI_DSP_MATRIX(signal_frame, 1, stack_frame_info.frame_length);
608
609 // don't read outside of the audio buffer... we'll automatically zero pad then
610 size_t signal_offset = stack_frame_info.frame_ixs.at(ix);
611 size_t signal_length = stack_frame_info.frame_length;
612 if (signal_offset + signal_length > stack_frame_info.signal->total_length) {
613 signal_length = signal_length -
614 (stack_frame_info.signal->total_length - (signal_offset + signal_length));
615 }
616
617 ret = stack_frame_info.signal->get_data(
618 signal_offset,
619 signal_length,
620 signal_frame.buffer
621 );
622 if (ret != 0) {
623 EIDSP_ERR(ret);
624 }
625
626 // normalize data (only when version is 3)
627 if (version == 3) {
628 // it might be that everything is already normalized here...
629 bool all_between_min_1_and_1 = true;
630 for (size_t ix = 0; ix < signal_frame.rows * signal_frame.cols; ix++) {
631 if (signal_frame.buffer[ix] < -1.0f || signal_frame.buffer[ix] > 1.0f) {
632 all_between_min_1_and_1 = false;
633 break;
634 }
635 }
636
637 if (!all_between_min_1_and_1) {
638 ret = numpy::scale(&signal_frame, 1.0f / 32768.0f);
639 if (ret != 0) {
640 EIDSP_ERR(ret);
641 }
642 }
643 }
644
645 ret = numpy::power_spectrum(
646 signal_frame.buffer,
647 stack_frame_info.frame_length,
648 out_features->buffer + (ix * coefficients),
649 coefficients,
650 fft_length
651 );
652
653 if (ret != 0) {
654 EIDSP_ERR(ret);
655 }
656 }
657
658 numpy::zero_handling(out_features);
659
660 return EIDSP_OK;
661 }

715 static int mfcc(matrix_t *out_features, signal_t *signal,
716 uint32_t sampling_frequency, float frame_length, float frame_stride,
717 uint8_t num_cepstral, uint16_t num_filters, uint16_t fft_length,
718 uint32_t low_frequency, uint32_t high_frequency, bool dc_elimination,
719 uint16_t version)
720 {
721 if (out_features->cols != num_cepstral) {
722 EIDSP_ERR(EIDSP_MATRIX_SIZE_MISMATCH);
723 }
724
725 matrix_size_t mfe_matrix_size =
726 calculate_mfe_buffer_size(
727 signal->total_length,
728 sampling_frequency,
729 frame_length,
730 frame_stride,
731 num_filters,
732 version);
733
734 if (out_features->rows != mfe_matrix_size.rows) {
735 EIDSP_ERR(EIDSP_MATRIX_SIZE_MISMATCH);
736 }
737
738 int ret = EIDSP_OK;
739
740 // allocate some memory for the MFE result
741 EI_DSP_MATRIX(features_matrix, mfe_matrix_size.rows, mfe_matrix_size.cols);
742 if (!features_matrix.buffer) {
743 EIDSP_ERR(EIDSP_OUT_OF_MEM);
744 }
745
746 EI_DSP_MATRIX(energy_matrix, mfe_matrix_size.rows, 1);
747 if (!energy_matrix.buffer) {
748 EIDSP_ERR(EIDSP_OUT_OF_MEM);
749 }
750
751 ret = mfe(&features_matrix, &energy_matrix, signal,
752 sampling_frequency, frame_length, frame_stride, num_filters, fft_length,
753 low_frequency, high_frequency, version);
754 if (ret != EIDSP_OK) {
755 EIDSP_ERR(ret);
756 }
757
758 // ok... now we need to calculate the MFCC from this...
759 // first do log() over all features...
760 ret = numpy::log(&features_matrix);
761 if (ret != EIDSP_OK) {
762 EIDSP_ERR(ret);
763 }
764
765 // now do DST type 2
766 ret = numpy::dct2(&features_matrix, DCT_NORMALIZATION_ORTHO);
767 if (ret != EIDSP_OK) {
768 EIDSP_ERR(ret);
769 }
770
771 // replace first cepstral coefficient with log of frame energy for DC elimination
772 if (dc_elimination) {
773 for (size_t row = 0; row < features_matrix.rows; row++) {
