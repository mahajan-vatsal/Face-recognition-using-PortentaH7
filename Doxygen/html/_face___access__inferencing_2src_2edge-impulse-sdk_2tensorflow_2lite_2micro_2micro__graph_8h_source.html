Machine Vision using Portenta H7 2
This project aims to develop a face recognition-based access control system using the Arduino Portenta H7 and Vision Shield, leveraging Edge Impulse for machine learning. The system captures facial images, processes them locally using an AI model deployed on the Portenta H7 and determines access based on authorised personnel.

26// Abstracts the details of interacting with the tflite::Model.
27//
28// Provides methods to access, initialize, prepare, invoke and free any
29// subgraph in the tflite::Graph.
30class MicroGraph {
31 public:
32 // The lifetime of the context, model, allocator and resource_variables must
33 // be at least as long as that of the graph object, since the this class may
34 // need to access them at any time. If resource_variables is a nullptr,
35 // GetResourceVariables will return a nullptr.
36 MicroGraph(TfLiteContext* context, const Model* model,
37 MicroAllocator* allocator,
38 MicroResourceVariables* resource_variables);
39 virtual ~MicroGraph();
40
41 // Sets up builtin data and calls TfLiteRegistration->Init for every operator
42 // in every subgraph in the model.
43 virtual TfLiteStatus InitSubgraphs();
44
45 // Calls TfLiteRegistration->Prepare for every operator in every subgraph in
46 // the model.
47 virtual TfLiteStatus PrepareSubgraphs(bool run_all_prep_ops);
48
49 // Calls TfLiteRegistration->Free for every operator in every subgraph in the
50 // model.
51 virtual TfLiteStatus FreeSubgraphs();
52
53 // Calls TfLiteRegistration->Invoke for every operator in a single subgraph in
54 // the model.
55 virtual TfLiteStatus InvokeSubgraph(int subgraph_idx);
56
57 // Zeros out all variable tensors in all subgraphs in the model.
58 virtual TfLiteStatus ResetVariableTensors();
59
60 // Number of tensor inputs to a specified subgraph in the model.
61 virtual size_t NumSubgraphInputs(int subgraph_idx);
62
63 // Get the specified input tensor of a specified subgraph in the model.
64 virtual TfLiteEvalTensor* GetSubgraphInput(int subgraph_idx, int input_idx);
65
66 // Number of tensor outputs from a specified subgraph in the model.
67 virtual size_t NumSubgraphOutputs(int subgraph_idx);
68
69 // Get the specified output tensor of a specified subgraph in the model.
70 virtual TfLiteEvalTensor* GetSubgraphOutput(int subgraph_idx, int output_idx);
71
72 // Number of subgraphs in the model.
73 virtual int NumSubgraphs();
74
75 // Hook to pass in subgraph allocations tracked within the interpreter,
76 // allowing MicroGraph to init / prepare / invoke subgraphs in the model.
77 void SetSubgraphAllocations(SubgraphAllocations* subgraph_allocations);
78
79 // Get the current subgraph index. Within an on operator, this is guaranteed
80 // to be the subgraph of that operator.
81 int GetCurrentSubgraphIndex() { return current_subgraph_index_; }
82
83 // Set the current subgraph index.
84 void SetCurrentSubgraphIndex(int subgraph_idx)
85 {
86 current_subgraph_index_ = subgraph_idx;
87 }
88
89 // Gets the list of alloctions for each subgraph. This is the source of truth
90 // for all per-subgraph allocation data.
91 SubgraphAllocations* GetAllocations() { return subgraph_allocations_; }
92
93 // Get the resource variables for this TFLM graph.
94 MicroResourceVariables* GetResourceVariables() { return resource_variables_; }
95
96 private:
97 TfLiteContext* context_;
98 const Model* model_;
99 MicroAllocator* allocator_;
100 SubgraphAllocations* subgraph_allocations_ = nullptr;
101 int current_subgraph_index_;
102 MicroResourceVariables* resource_variables_;
103 const flatbuffers::Vector<flatbuffers::Offset<SubGraph>>* subgraphs_;
104
105 TF_LITE_REMOVE_VIRTUAL_DELETE
106};
