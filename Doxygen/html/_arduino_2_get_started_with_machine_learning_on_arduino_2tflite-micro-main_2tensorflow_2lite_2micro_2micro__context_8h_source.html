Machine Vision using Portenta H7 2
This project aims to develop a face recognition-based access control system using the Arduino Portenta H7 and Vision Shield, leveraging Edge Impulse for machine learning. The system captures facial images, processes them locally using an AI model deployed on the Portenta H7 and determines access based on authorised personnel.

26// MicroContext is eventually going to become the API between TFLM and the
27// kernels, replacing all the functions in TfLiteContext. The end state is code
28// kernels to have code like:
29//
30// MicroContext* micro_context = GetMicroContext(context);
31// micro_context-><TFLM kernel API>
32class MicroContext {
33 public:
34 virtual ~MicroContext() = default;
35
36 // Allocate persistent buffer which has the same life time as the interpreter.
37 // Returns nullptr on failure.
38 // The memory is allocated from the tail.
39 // This method is only available in Init or Prepare stage.
40 virtual void* AllocatePersistentBuffer(size_t bytes) = 0;
41
42 // Request a scratch buffer in the arena through static memory planning.
43 // This method is only available in Prepare stage and the buffer is allocated
44 // by the interpreter between Prepare and Eval stage. In Eval stage,
45 // GetScratchBuffer API can be used to fetch the address.
46 virtual TfLiteStatus RequestScratchBufferInArena(size_t bytes,
47 int* buffer_idx) = 0;
48
49 // Get the scratch buffer pointer.
50 // This method is only available in Eval stage.
51 virtual void* GetScratchBuffer(int buffer_idx) = 0;
52
53 // Returns a temporary TfLiteTensor struct for a given index.
54 virtual TfLiteTensor* AllocateTempTfLiteTensor(int tensor_idx) = 0;
55
56 // Returns a temporary TfLiteTensor struct for the specified input tensor of a
57 // given mode. This is the recommended API over the deprecated
58 // GetInput/GetInputSafe to get a temp input tensor. The returned tensor shall
59 // be freed via calling DeallocateTempTfLiteTensor.
60 TfLiteTensor* AllocateTempInputTensor(const TfLiteNode* node, int index);
61
62 // Returns a temporary TfLiteTensor struct for the specified output tensor of
63 // a given mode. This is the recommended API over the deprecated
64 // GetOutput/GetOutputSafe to get a temp output tensor. The returned tensor
65 // shall be freed via calling DeallocateTempTfLiteTensor.
66 TfLiteTensor* AllocateTempOutputTensor(const TfLiteNode* node, int index);
67
68 // Returns a temporary TfLiteTensor struct for the specified intermediate
69 // tensor of a given mode. This is the recommended API over the deprecated
70 // GetIntermediates/GetIntermediatesSafe to get a temp intermediate tensor.
71 // The returned tensor shall be freed via calling DeallocateTempTfLiteTensor.
72 TfLiteTensor* AllocateTempIntermediateTensor(const TfLiteNode* node,
73 int index);
74
75 // Deallocates a temp TfLiteTensor.
76 virtual void DeallocateTempTfLiteTensor(TfLiteTensor* tensor) = 0;
77
78 // Returns a pointer to a temporary buffer (from the arena).
79 // This API is only valid from the kernel's Prepare function and
80 // the buffer's lifetime is also that of the Prepare function.
81 virtual uint8_t* AllocateTempBuffer(size_t size, size_t alignment) = 0;
82
83 // Signals that the temporary buffer is no longer needed.
84 virtual void DeallocateTempBuffer(uint8_t* buffer) = 0;
85
86 // Returns a TfLiteEvalTensor struct for a given index.
87 virtual TfLiteEvalTensor* GetEvalTensor(int tensor_idx) = 0;
88
89 // Does not take ownership of the pointer and the pointer must refer to valid
90 // an object that outlive this class instance.
91 // This can only be called once to set one external context.
92 virtual TfLiteStatus set_external_context(void* external_context_payload) = 0;
93
94 virtual void* external_context() = 0;
95
96 virtual MicroGraph& graph() = 0;
97
98 private:
99 TF_LITE_REMOVE_VIRTUAL_DELETE
100};

102inline MicroContext* GetMicroContext(const struct TfLiteContext* context) {
103 return reinterpret_cast<MicroContext*>(context->impl_);
104}
