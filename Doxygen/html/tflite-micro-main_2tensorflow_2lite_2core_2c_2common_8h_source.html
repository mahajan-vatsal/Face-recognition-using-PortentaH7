186// Since we must not depend on any libraries, define a minimal subset of
187// error macros while avoiding names that have pre-conceived meanings like
188// assert and check.

190// Try to make all reporting calls through TF_LITE_KERNEL_LOG rather than
191// calling the context->ReportError function directly, so that message strings
192// can be stripped out if the binary size needs to be severely optimized.
193#ifndef TF_LITE_STRIP_ERROR_STRINGS
194#define TF_LITE_KERNEL_LOG(context, ...) \
195 do { \
196 (context)->ReportError((context), __VA_ARGS__); \
197 } while (false)
198
199#define TF_LITE_MAYBE_KERNEL_LOG(context, ...) \
200 do { \
201 if ((context) != nullptr) { \
202 (context)->ReportError((context), __VA_ARGS__); \
203 } \
204 } while (false)
205#else // TF_LITE_STRIP_ERROR_STRINGS
206#define ARGS_UNUSED(...) (void)sizeof(#__VA_ARGS__)
207#define TF_LITE_KERNEL_LOG(context, ...) ARGS_UNUSED(__VA_ARGS__)
208#define TF_LITE_MAYBE_KERNEL_LOG(context, ...) ARGS_UNUSED(__VA_ARGS__)
209#endif // TF_LITE_STRIP_ERROR_STRINGS

213#define TF_LITE_ENSURE_MSG(context, value, ...) \
214 do { \
215 if (!(value)) { \
216 TF_LITE_KERNEL_LOG((context), __FILE__ " " __VA_ARGS__); \
217 return kTfLiteError; \
218 } \
219 } while (0)
220
223#define TF_LITE_ENSURE(context, a) \
224 do { \
225 if (!(a)) { \
226 TF_LITE_KERNEL_LOG((context), "%s:%d %s was not true.", __FILE__, \
227 __LINE__, #a); \
228 return kTfLiteError; \
229 } \
230 } while (0)
231
232#define TF_LITE_ENSURE_STATUS(a) \
233 do { \
234 const TfLiteStatus s = (a); \
235 if (s != kTfLiteOk) { \
236 return s; \
237 } \
238 } while (0)
239
246#define TF_LITE_ENSURE_EQ(context, a, b) \
247 do { \
248 if ((a) != (b)) { \
249 TF_LITE_KERNEL_LOG((context), "%s:%d %s != %s (%d != %d)", __FILE__, \
250 __LINE__, #a, #b, (a), (b)); \
251 return kTfLiteError; \
252 } \
253 } while (0)
254
255#define TF_LITE_ENSURE_TYPES_EQ(context, a, b) \
256 do { \
257 if ((a) != (b)) { \
258 TF_LITE_KERNEL_LOG((context), "%s:%d %s != %s (%s != %s)", __FILE__, \
259 __LINE__, #a, #b, TfLiteTypeGetName(a), \
260 TfLiteTypeGetName(b)); \
261 return kTfLiteError; \
262 } \
263 } while (0)
264
265#define TF_LITE_ENSURE_NEAR(context, a, b, epsilon) \
266 do { \
267 auto delta = ((a) > (b)) ? ((a) - (b)) : ((b) - (a)); \
268 if (delta > epsilon) { \
269 TF_LITE_KERNEL_LOG((context), "%s:%d %s not near %s (%f != %f)", \
270 __FILE__, __LINE__, #a, #b, static_cast<double>(a), \
271 static_cast<double>(b)); \
272 return kTfLiteError; \
273 } \
274 } while (0)

277 do { \

279 if ((s) != kTfLiteOk) { \
280 return s; \
281 } \
282 } while (0)

586// Specific reduced TfLiteTensor struct for TF Micro runtime. This struct

600 // TODO(b/155784997): Consider consolidating these quantization fields:
601 // Quantization information. Replaces params field above.
602 TfLiteQuantization quantization;

604 // Quantization information.
605 TfLiteQuantizationParams params;
606
607 // A union of data pointers. The appropriate type should be used for a typed
608 // tensor based on `type`.
609 TfLitePtrUnion data;

612 // that the buffer should have. NOTE: the product of elements of `dims`
613 // and the element datatype size should be equal to `bytes` below.

615
616 // The number of bytes required to store the data of this Tensor. I.e.
617 // (bytes of each element) * dims[0] * ... * dims[n-1]. For example, if
618 // type is kTfLiteFloat32 and dims = {3, 2} then
619 // bytes = sizeof(float) * 3 * 2 = 4 * 3 * 2 = 24.

621
622 // The data type specification for data stored in `data`. This affects
623 // what member of `data` union should be used.
624 TfLiteType type;
625
626 // How memory is mapped
627 // kTfLiteMmapRo: Memory mapped read only.
628 // i.e. weights
629 // kTfLiteArenaRw: Arena allocated read write memory
630 // (i.e. temporaries, outputs).
631 TfLiteAllocationType allocation_type;

1462// --- TFLITE VARIANT TENSORS ----
1463// Programming languges usually define "variant" as a type that can hold an
1464// unbounded set of types. See std::any
1465// (https://en.cppreference.com/w/cpp/utility/any) for a related standard
1466// library construct. In tensorflow, variant tensors have a data member which is
1467// an Object that is destructible and copy constructible.
1468// Variant tensors are commonly used to represent non trivial data
1469// semantics that don't fit into simple primitives, such as lists of tensors and
1470// datasets. Additionally, they can facilitate containers for optimizing
1471// memory movement of tensor data.
1472//
1473// The following set of classes define the variant tensor member for tflite.
1474// They implement a type-erased container intended to be used behind the
1475// `data.data : void*` member of `TfLiteTensor`s. Runtime functions interact
1476// the variant member at the level of a `VariantData`, whereas kernels
1477// operate with the full knowledge of the un-erased type. The `VariantData`
1478// class provides abstract methods for destroying and copying `VariantData`.
1479// Invoking these methods will dispatch to the erased type opaquely.
1480// The contents of any object of type derived from `AbstractVariant` can be
1481// written to `TfLiteTensor::data::data : void*` from kernels. If the runtime
1482// were to copy such a tensor through `TfLiteTensorCopy`, the destination data
1483// member will contain the result of invoking the erased type's copy
1484// constructor. Similar for the runtime releasing tensors from memory, the
1485// erased type's destructor will be invoked. There are a few caveats to consider
1486// to use these safely, which we discuss below.
1487//
1488// EXAMPLE: READING VARIANT TENSORS
1489// ```
1490// // retrieve input with `type == kTfLiteVariant`
1491// TfLiteTensor* input = ...
1492// // must first static cast to `VariantData`, more on this below.
1493// VariantData* vd_input = static_cast<VariantData*>(t->data.data);
1494// CustomType* typed_input =
1495// static_cast<CustomType*>(vd_input);
1496// // do custom work on `typed_input`...
1497// ```
1498//
1499// EXAMPLE: WRITING VARIANT TENSORS
1500// ```
1501// TfLiteTensor* output = ...
1502// // construct a new variant object behind the target tensor
1503// TfLiteVariantRealloc<DerivedType, DerivedArgs...>(output, args...);
1504// // again must static cast to `VariantData*` before writing to `void*`.
1505// output->data.data = static_cast<VariantData*>(typed_output);
1506// ```
1507//
1508// WHY STATIC CAST TO `VariantData*`
1509// The Standard defines a `reinterpret_cast` from a derived type to its
1510// parents as undefined behavior when the parent is a non-standard layout.
1511// https://en.cppreference.com/w/cpp/language/reinterpret_cast (see bullet 5).
1512// Due to the `VariantData` having virtual members it is indeed non-standard
1513// layout, and any type derived from `VariantData` fails to be
1514// "transparently-replaceable". I.e. implicit cast from derived to base in this
1515// case may adjust the pointer and by definition `reinterpret_cast` will not
1516// the adjust the pointer.
1517// Thus, dereferencing a pointer of type `VariantData` which addresses
1518// the first byte of an object of said derived type is UB unless it was first
1519// implicitly or statically casted to a `VariantData`. Writing the object of
1520// derived type directly to `void*` which is dereferenced as a `VariantData` is
1521// then UB, and so the intermediate cast through `VariantData` must be enforced.
1522// A good example of this issue is ellucidate in the bottom code snippet
1523// here: https://en.cppreference.com/w/cpp/utility/launder.
1524class VariantData {
1525 public:
1526 // All variant objects must be able to be destroyed and copied.
1527 virtual ~VariantData() = default;
1528 // A "virtual copy-constructor". Often the destination tensor of a variant
1529 // copy may have been previously allocated in a prior call to inference. We
1530 // allow the copy to target the destinations buffer (`maybe_alloc`),
1531 // for potential reuse and optimizations. `maybe_alloc` must be of the same
1532 // underlying derived type. References to whatever object is at
1533 // `maybe_alloc` may be invalidated.
1534 virtual VariantData* CloneTo(VariantData* maybe_alloc) const = 0;
1535};
1536
1537// Concrete implementations extend `AbstractVariantData` with CRPT.
1538template <typename ErasedDerived>
1539class AbstractVariantData : public VariantData {
1540 public:
1541 VariantData* CloneTo(VariantData* maybe_alloc) const override {
1542 if (maybe_alloc != nullptr) {
1543 // If the output is still allocated, then its object may still be
1544 // in its life time and the destructor must be called before re-using the
1545 // buffer.
1546 // This may actual have a non-negligible effect on performance if the
1547 // destructor is complex. A future iteration may
1548 // introduce copy or move assignment semantics, allowing for the
1549 // underlying implementation to optimize for this case.
1550 auto* derived = static_cast<ErasedDerived*>(maybe_alloc);
1551 derived->~ErasedDerived();
1552 return new (derived)
1553 ErasedDerived(static_cast<ErasedDerived const&>(*this));
1554 }
1555 return new ErasedDerived(static_cast<ErasedDerived const&>(*this));
1556 }
1557
1558 protected:
1559 AbstractVariantData() = default;
1560 AbstractVariantData(const AbstractVariantData&) = default;
1561 AbstractVariantData(AbstractVariantData&&) = delete;
1562};

1564// Analogous to `TfLiteTensorRealloc` for allocation of tensors whose
1565// data member points to an arbitrary C++ object. `VariantType` refers
1566// to the erased type of said object and `VariantArgs` refers to
1567// a list of argument types with which to construct a new `VariantType`.
1568// `VariantArgs` must match a constructor of `VariantType`.
1569template <class VariantType, class... VariantArgs>
1570TfLiteStatus TfLiteTensorVariantRealloc(TfLiteTensor* t,
1571 VariantArgs&&... args) {
1572 if (t->type != kTfLiteVariant) return kTfLiteError;
1573 VariantType* new_vd;
1574 if (t->data.raw != nullptr) {
1575 auto* target_vd = static_cast<VariantData*>(t->data.data);
1576 target_vd->~VariantData();
1577 // As above, we assume if `t` is already allocated then it was allocated
1578 // with the same `VariantType` as templated.
1579 new_vd = new (t->data.raw) VariantType(std::forward<VariantArgs>(args)...);
1580 } else {
1581 new_vd = new VariantType(std::forward<VariantArgs>(args)...);
1582 }
1583 t->data.data = static_cast<VariantData*>(new_vd);
1584 t->allocation_type = kTfLiteVariantObject;
1585 return kTfLiteOk;
1586}
