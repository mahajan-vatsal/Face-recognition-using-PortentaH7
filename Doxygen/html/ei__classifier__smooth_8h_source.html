Machine Vision using Portenta H7 2
This project aims to develop a face recognition-based access control system using the Arduino Portenta H7 and Vision Shield, leveraging Edge Impulse for machine learning. The system captures facial images, processes them locally using an AI model deployed on the Portenta H7 and determines access based on authorised personnel.

66const char* ei_classifier_smooth_update(ei_classifier_smooth_t *smooth, ei_impulse_result_t *result) {
67 // clear out the count array
68 memset(smooth->count, 0, EI_CLASSIFIER_LABEL_COUNT + 2);
69
70 // roll through the last_readings buffer
71 numpy::roll(smooth->last_readings, smooth->last_readings_size, -1);
72
73 int reading = -1; // uncertain
74
75 // print the predictions
76 // printf("[");
77 for (size_t ix = 0; ix < EI_CLASSIFIER_LABEL_COUNT; ix++) {
78 if (result->classification[ix].value >= smooth->classifier_confidence) {
79 reading = (int)ix;
80 }
81 }
82#if EI_CLASSIFIER_HAS_ANOMALY
83 if (result->anomaly >= smooth->anomaly_confidence) {
84 reading = -2; // anomaly
85 }
86#endif
87
88 smooth->last_readings[smooth->last_readings_size - 1] = reading;
89
90 // now count last 10 readings and see what we actually see...
91 for (size_t ix = 0; ix < smooth->last_readings_size; ix++) {
92 if (smooth->last_readings[ix] >= 0) {
93 smooth->count[smooth->last_readings[ix]]++;
94 }
95 else if (smooth->last_readings[ix] == -1) { // uncertain
96 smooth->count[EI_CLASSIFIER_LABEL_COUNT]++;
97 }
98 else if (smooth->last_readings[ix] == -2) { // anomaly
99 smooth->count[EI_CLASSIFIER_LABEL_COUNT + 1]++;
100 }
101 }
102
103 // then loop over the count and see which is highest
104 uint8_t top_result = 0;
105 uint8_t top_count = 0;
106 bool met_confidence_threshold = false;
107 uint8_t confidence_threshold = smooth->min_readings_same; // XX% of windows should be the same
108 for (size_t ix = 0; ix < EI_CLASSIFIER_LABEL_COUNT + 2; ix++) {
109 if (smooth->count[ix] > top_count) {
110 top_result = ix;
111 top_count = smooth->count[ix];
112 }
113 if (smooth->count[ix] >= confidence_threshold) {
114 met_confidence_threshold = true;
115 }
116 }
117
118 if (met_confidence_threshold) {
119 if (top_result == EI_CLASSIFIER_LABEL_COUNT) {
120 return "uncertain";
121 }
122 else if (top_result == EI_CLASSIFIER_LABEL_COUNT + 1) {
123 return "anomaly";
124 }
125 else {
126 return result->classification[top_result].label;
127 }
128 }
129 return "uncertain";
130}
