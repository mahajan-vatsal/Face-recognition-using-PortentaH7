Machine Vision using Portenta H7 2
This project aims to develop a face recognition-based access control system using the Arduino Portenta H7 and Vision Shield, leveraging Edge Impulse for machine learning. The system captures facial images, processes them locally using an AI model deployed on the Portenta H7 and determines access based on authorised personnel.

28// composite default audio device interface
29class AudioDevice {
30 public:
31 // Get the current audio configuration.
32 // The current audio configuration is always valid.
33 //
34 // Returns:
35 // copy of AudioConfiguration
36 virtual AudioConfiguration GetCurrentConfiguration() const {
37 return i2s_.GetCurrentConfiguration();
38 }
39
40 // Set the current audio configuration.
41 // This method cannot be used within the audio callback handler.
42 //
43 // Returns:
44 // true if the audio configuration is valid
45 // false otherwise
46 virtual bool SetCurrentConfiguration(const AudioConfiguration& new_config) {
47 if (is_callback_handler_active_) {
48 return false;
49 }
50
51 i2s_.Stop(kPlay);
52 codec_.Stop(kPlay);
53 i2s_.Stop(kRecord);
54 codec_.Stop(kRecord);
55
56 bool result = false;
57 AudioConfiguration old_config;
58 old_config = i2s_.GetCurrentConfiguration();
59 result = i2s_.SetCurrentConfiguration(new_config);
60 if (!result) {
61 return false;
62 }
63
64 result = codec_.SetCurrentConfiguration(new_config);
65 if (!result) {
66 i2s_.SetCurrentConfiguration(old_config);
67 return false;
68 }
69
70 return result;
71 }
72
73 // Start playback or recording.
74 // This method cannot be used within the audio callback handler.
75 //
76 // Notes:
77 // When starting playback, the internal playback buffers are
78 // zero-filled and playback begins immediately
79 virtual void Start(const AudioFunction which) {
80 if (is_callback_handler_active_) {

156 }
157
158 // Mute the playback or recording

160 codec_.Mute(which, enable);
161 }
162
163 // Set the audio callback handler.
164 // The callback handler will be executed on each internal buffer update,
165 // when playback and/or recording is active.
166 // This method cannot be used within the audio callback handler.
167 //
168 // Notes:
169 // The callback handler may be executed from within an interrupt handler.
170 // Therefore, time spent in the callback handler should be minimized.
171 virtual void SetCallbackHandler(const AudioCallback handler) {
172 if (is_callback_handler_active_) {
173 return;
174 }
175 callback_handler_ = handler;
176 if (handler == nullptr) {
177 i2s_.SetCallbackHandler(nullptr);
178 } else {
179 AudioCallback cb = [this](const AudioFunction which) {
180 this->CallbackHandler(which);
181 };
182 i2s_.SetCallbackHandler(cb);
183 }
184 }
185
186 private:
187 IAudioI2S& i2s_;
188 IAudioCodec& codec_;
189 bool is_callback_handler_active_;
190 AudioCallback callback_handler_;
191
192 void CallbackHandler(const AudioFunction which) {
193 if (callback_handler_ != nullptr) {
194 is_callback_handler_active_ = true;
195 callback_handler_(which);
196 is_callback_handler_active_ = false;
197 }
198 }

207 callback_handler_{nullptr} {}
