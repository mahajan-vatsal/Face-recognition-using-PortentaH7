Machine Vision using Portenta H7 2
This project aims to develop a face recognition-based access control system using the Arduino Portenta H7 and Vision Shield, leveraging Edge Impulse for machine learning. The system captures facial images, processes them locally using an AI model deployed on the Portenta H7 and determines access based on authorised personnel.

31class CrossingCounter {
32public:
33 CrossingCounter(std::vector<std::tuple<int, int, int, int>> segments) : segments(segments) {
34 counts.resize(segments.size(), 0);
35 }
36
37 void update(std::tuple<int, int, int, int> other_segment) {
38 for (size_t segment_idx = 0; segment_idx < segments.size(); segment_idx++) {
39 if (_line_intersects(segments[segment_idx], other_segment)) {
40 counts[segment_idx] += 1;
41 }
42 }
43 }
44
45 std::vector<uint32_t> counts;
46 std::vector<std::tuple<int, int, int, int>> segments;
47private:
48 bool ccw(int A[2], int B[2], int C[2]) {
49 return (C[1] - A[1]) * (B[0] - A[0]) > (B[1] - A[1]) * (C[0] - A[0]);
50 }
51
52 bool _line_intersects(std::tuple<int, int, int, int> L1, std::tuple<int, int, int, int> L2) {
53 int A[2] = { std::get<0>(L1), std::get<1>(L1) };
54 int B[2] = { std::get<2>(L1), std::get<3>(L1) };
55 int C[2] = { std::get<0>(L2), std::get<1>(L2) };
56 int D[2] = { std::get<2>(L2), std::get<3>(L2) };
57// verbose debug
58#if EI_LOG_LEVEL == 5
59 ei_printf("A[0] = %d A[1] = %d\n", A[0], A[1]);
60 ei_printf("B[0] = %d B[1] = %d\n", B[0], B[1]);
61 ei_printf("C[0] = %d C[1] = %d\n", C[0], C[1]);
62 ei_printf("D[0] = %d D[1] = %d\n", D[0], D[1]);
63#endif
64 return ccw(A, C, D) != ccw(B, C, D) && ccw(A, B, C) != ccw(A, B, D);
65 }
66};

97EI_IMPULSE_ERROR process_object_counting(ei_impulse_handle_t *handle,
98 ei_impulse_result_t *result,
99 void *config,
100 void *state)
101{
102 const ei_impulse_t *impulse = handle->impulse;
103 CrossingCounter *object_counter = (CrossingCounter*)state;
104
105 if (impulse->sensor == EI_CLASSIFIER_SENSOR_CAMERA) {
106 if((void *)object_counter != NULL) {
107 for (size_t i = 0; i < result->postprocessed_output.object_tracking_output.open_traces_count; i++) {
108 ei_object_tracking_trace_t trace = result->postprocessed_output.object_tracking_output.open_traces[i];
109 object_counter->update(trace.last_centroid_segment);
110 }
111
112 result->postprocessed_output.object_counting_output.counts = object_counter->counts.data();
113 result->postprocessed_output.object_counting_output.counter_num = object_counter->counts.size();
114 }
115 }
116
117 return EI_IMPULSE_OK;
118}
