52inline void histo(const fvec &x, size_t nbins, fvec &h, bool normalize = false)
53{
54 float min = *std::min_element(x.begin(), x.end());
55 float max = *std::max_element(x.begin(), x.end());
56 float step = (max - min) / nbins;
57 h.resize(nbins);
58 for (size_t i = 0; i < x.size(); i++) {
59 size_t bin = (x[i] - min) / step;
60 if (bin >= nbins)
61 bin = nbins - 1;
62 h[bin]++;
63 }
64 if (normalize) {
65 float s = numpy::sum(h.data(), h.size());
66 for (size_t i = 0; i < nbins; i++) {
67 h[i] /= s;
68 }
69 }

143 static void calculate_entropy(const fvec &y, fvec &features)
144 {
145 fvec h;
146 histo(y, 100, h, true);
147 // entropy = -sum(prob * log(prob)
148 float entropy = 0.0f;
149 for (size_t i = 0; i < h.size(); i++) {
150 if (h[i] > 0.0f) {
151 entropy -= h[i] * log(h[i]);
152 }
153 }
154 features.push_back(entropy);
155 }

189 static void calculate_crossings(const fvec &y, fvec &features, float mean)
190 {
191 size_t zc = 0;
192 for (size_t i = 1; i < y.size(); i++) {
193 if (y[i] * y[i - 1] < 0) {
194 zc++;
195 }
196 }
197 features.push_back(zc / (float)y.size());
198
199 size_t mc = 0;
200 for (size_t i = 1; i < y.size(); i++) {
201 if ((y[i] - mean) * (y[i - 1] - mean) < 0) {
202 mc++;
203 }
204 }
205 features.push_back(mc / (float)y.size());
206 }

208 static void
209 dwt(const float *x, size_t nx, const float *h, const float *g, size_t nh, fvec &a, fvec &d)
210 {
211 assert(nh <= 20 && nh > 0 && nx > 0);
212 size_t nx_padded = nx + nh * 2 - 2;
213 fvec x_padded(nx_padded);
214
215 // symmetric padding (default in PyWavelet)
216 for (size_t i = 0; i < nh - 2; i++)
217 x_padded[i] = x[nh - 3 - i];
218 for (size_t i = 0; i < nx; i++)
219 x_padded[i + nh - 2] = x[i];
220 for (size_t i = 0; i < nh; i++)
221 x_padded[i + nx + nh - 2] = x[nx - 1 - i];
222
223 size_t ny = (nx + nh - 1) / 2;
224 a.resize(ny);
225 d.resize(ny);
226
227 // decimate and filter
228 const float *xx = x_padded.data();
229 for (size_t i = 0; i < ny; i++) {
230 a[i] = dot(xx + 2 * i, h, nh);
231 d[i] = dot(xx + 2 * i, g, nh);
232 }
233
234 numpy::underflow_handling(d.data(), d.size());
235 numpy::underflow_handling(a.data(), a.size());
236 }

251 static void
252 wavedec_features(const float *x, int len, const char *wav, int level, fvec &features)
253 {
254 assert(level > 0 && level < 8);
255
256 fvec h;
257 fvec g;
258 find_filter(wav, h, g);
259
260 features.clear();
261 fvec a;
262 fvec d;
263 dwt(x, len, h.data(), g.data(), h.size(), a, d);
264 extract_features(d, features);
265
266 for (int l = 1; l < level; l++) {
267 dwt(a.data(), a.size(), h.data(), g.data(), h.size(), a, d);
268 extract_features(d, features);
269 }
270
271 extract_features(a, features);
272
273 for (int l = 0; l <= level / 2; l++) { // reverse order to match python results.
274 for (int i = 0; i < (int)NUM_FEATHERS_PER_COMP; i++) {
275 std::swap(
276 features[l * NUM_FEATHERS_PER_COMP + i],
277 features[(level - l) * NUM_FEATHERS_PER_COMP + i]);
278 }
279 }

301 static int extract_wavelet_features(
302 matrix_t *input_matrix,
303 matrix_t *output_matrix,
304 ei_dsp_config_spectral_analysis_t *config,
305 const float sampling_freq)
306 {
307 // transpose the matrix so we have one row per axis
308 numpy::transpose_in_place(input_matrix);
309
310 // func tests for scale of 1 and does a no op in that case
311 EI_TRY(numpy::scale(input_matrix, config->scale_axes));
312
313 // apply filter, if enabled
314 // "zero" order filter allowed. will still remove unwanted fft bins later
315 if (strcmp(config->filter_type, "low") == 0) {
316 if (config->filter_order) {
317 EI_TRY(spectral::processing::butterworth_lowpass_filter(
318 input_matrix,
319 sampling_freq,
320 config->filter_cutoff,
321 config->filter_order));
322 }
323 }
324 else if (strcmp(config->filter_type, "high") == 0) {
325 if (config->filter_order) {
326 EI_TRY(spectral::processing::butterworth_highpass_filter(
327 input_matrix,
328 sampling_freq,
329 config->filter_cutoff,
330 config->filter_order));
331 }
332 }
333
334 EI_TRY(processing::subtract_mean(input_matrix));
335
336 int out_idx = 0;
337 for (size_t row = 0; row < input_matrix->rows; row++) {
338 float *data_window = input_matrix->get_row_ptr(row);
339 size_t data_size = input_matrix->cols;
340
341 if (!check_min_size(data_size, config->wavelet_level))
342 EIDSP_ERR(EIDSP_BUFFER_SIZE_MISMATCH);
343
344 fvec features;
345 size_t num_features = dwt_features(
346 data_window,
347 data_size,
348 config->wavelet,
349 config->wavelet_level,
350 features);
351
352 assert(num_features == output_matrix->cols / input_matrix->rows);
353 for (size_t i = 0; i < num_features; i++) {
354 output_matrix->buffer[out_idx++] = features[i];
355 }
356 }
357 return EIDSP_OK;
358 }
