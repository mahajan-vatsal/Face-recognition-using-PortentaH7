37struct BoxCornerEncoding {
38 float y1;
39 float x1;
40 float y2;
41 float x2;
42};

44static inline float ComputeIntersectionOverUnion(const float* boxes, const int i,
45 const int j) {
46 auto& box_i = reinterpret_cast<const BoxCornerEncoding*>(boxes)[i];
47 auto& box_j = reinterpret_cast<const BoxCornerEncoding*>(boxes)[j];
48 const float box_i_y_min = std::min<float>(box_i.y1, box_i.y2);
49 const float box_i_y_max = std::max<float>(box_i.y1, box_i.y2);
50 const float box_i_x_min = std::min<float>(box_i.x1, box_i.x2);
51 const float box_i_x_max = std::max<float>(box_i.x1, box_i.x2);
52 const float box_j_y_min = std::min<float>(box_j.y1, box_j.y2);
53 const float box_j_y_max = std::max<float>(box_j.y1, box_j.y2);
54 const float box_j_x_min = std::min<float>(box_j.x1, box_j.x2);
55 const float box_j_x_max = std::max<float>(box_j.x1, box_j.x2);
56
57 const float area_i =
58 (box_i_y_max - box_i_y_min) * (box_i_x_max - box_i_x_min);
59 const float area_j =
60 (box_j_y_max - box_j_y_min) * (box_j_x_max - box_j_x_min);
61 if (area_i <= 0 || area_j <= 0) return 0.0;
62 const float intersection_ymax = std::min<float>(box_i_y_max, box_j_y_max);
63 const float intersection_xmax = std::min<float>(box_i_x_max, box_j_x_max);
64 const float intersection_ymin = std::max<float>(box_i_y_min, box_j_y_min);
65 const float intersection_xmin = std::max<float>(box_i_x_min, box_j_x_min);
66 const float intersection_area =
67 std::max<float>(intersection_ymax - intersection_ymin, 0.0) *
68 std::max<float>(intersection_xmax - intersection_xmin, 0.0);
69 return intersection_area / (area_i + area_j - intersection_area);
70}

72// Implements (Single-Class) Soft NMS (with Gaussian weighting).
73// Supports functionality of TensorFlow ops NonMaxSuppressionV4 & V5.
74// Reference: "Soft-NMS - Improving Object Detection With One Line of Code"
75// [Bodla et al, https://arxiv.org/abs/1704.04503]
76// Implementation adapted from the TensorFlow NMS code at
77// tensorflow/core/kernels/non_max_suppression_op.cc.
78//
79// Arguments:
80// boxes: box encodings in format [y1, x1, y2, x2], shape: [num_boxes, 4]
81// num_boxes: number of candidates
82// scores: scores for candidate boxes, in the same order. shape: [num_boxes]
83// max_output_size: the maximum number of selections.
84// iou_threshold: Intersection-over-Union (IoU) threshold for NMS
85// score_threshold: All candidate scores below this value are rejected
86// soft_nms_sigma: Soft NMS parameter, used for decaying scores
87//
88// Outputs:
89// selected_indices: all the selected indices. Underlying array must have
90// length >= max_output_size. Cannot be null.
91// selected_scores: scores of selected indices. Defer from original value for
92// Soft NMS. If not null, array must have length >= max_output_size.
93// num_selected_indices: Number of selections. Only these many elements are
94// set in selected_indices, selected_scores. Cannot be null.
95//
96// Assumes inputs are valid (for eg, iou_threshold must be >= 0).
97static inline void NonMaxSuppression(const float* boxes, const int num_boxes,
98 const float* scores, const int max_output_size,
99 const float iou_threshold,
100 const float score_threshold,
101 const float soft_nms_sigma, int* selected_indices,
102 float* selected_scores,
103 int* num_selected_indices) {
104 struct Candidate {
105 int index;
106 float score;
107 int suppress_begin_index;
108 };
109
110 // Priority queue to hold candidates.
111 auto cmp = [](const Candidate bs_i, const Candidate bs_j) {
112 return bs_i.score < bs_j.score;
113 };
114 std::priority_queue<Candidate, std::deque<Candidate>, decltype(cmp)>
115 candidate_priority_queue(cmp);
116 // Populate queue with candidates above the score threshold.
117 for (int i = 0; i < num_boxes; ++i) {
118 if (scores[i] > score_threshold) {
119 candidate_priority_queue.emplace(Candidate({i, scores[i], 0}));
120 }
121 }
122
123 *num_selected_indices = 0;
124 int num_outputs = std::min(static_cast<int>(candidate_priority_queue.size()),
125 max_output_size);
126 if (num_outputs == 0) return;
127
128 // NMS loop.
129 float scale = 0;
130 if (soft_nms_sigma > 0.0) {
131 scale = -0.5 / soft_nms_sigma;
132 }
133 while (*num_selected_indices < num_outputs &&
134 !candidate_priority_queue.empty()) {
135 Candidate next_candidate = candidate_priority_queue.top();
136 const float original_score = next_candidate.score;
137 candidate_priority_queue.pop();
138
139 // Overlapping boxes are likely to have similar scores, therefore we
140 // iterate through the previously selected boxes backwards in order to
141 // see if `next_candidate` should be suppressed. We also enforce a property
142 // that a candidate can be suppressed by another candidate no more than
143 // once via `suppress_begin_index` which tracks which previously selected
144 // boxes have already been compared against next_candidate prior to a given
145 // iteration. These previous selected boxes are then skipped over in the
146 // following loop.
147 bool should_hard_suppress = false;
148 for (int j = *num_selected_indices - 1;
149 j >= next_candidate.suppress_begin_index; --j) {
150 const float iou = ComputeIntersectionOverUnion(
151 boxes, next_candidate.index, selected_indices[j]);
152
153 // First decide whether to perform hard suppression.
154 if (iou >= iou_threshold) {
155 should_hard_suppress = true;
156 break;
157 }
158
159 // Suppress score if NMS sigma > 0.
160 if (soft_nms_sigma > 0.0) {
161 next_candidate.score =
162 next_candidate.score * std::exp(scale * iou * iou);
163 }
164
165 // If score has fallen below score_threshold, it won't be pushed back into
166 // the queue.
167 if (next_candidate.score <= score_threshold) break;
168 }
169 // If `next_candidate.score` has not dropped below `score_threshold`
170 // by this point, then we know that we went through all of the previous
171 // selections and can safely update `suppress_begin_index` to
172 // `selected.size()`. If on the other hand `next_candidate.score`
173 // *has* dropped below the score threshold, then since `suppress_weight`
174 // always returns values in [0, 1], further suppression by items that were
175 // not covered in the above for loop would not have caused the algorithm
176 // to select this item. We thus do the same update to
177 // `suppress_begin_index`, but really, this element will not be added back
178 // into the priority queue.
179 next_candidate.suppress_begin_index = *num_selected_indices;
180
181 if (!should_hard_suppress) {
182 if (next_candidate.score == original_score) {
183 // Suppression has not occurred, so select next_candidate.
184 selected_indices[*num_selected_indices] = next_candidate.index;
185 if (selected_scores) {
186 selected_scores[*num_selected_indices] = next_candidate.score;
187 }
188 ++*num_selected_indices;
189 }
190 if ((soft_nms_sigma > 0.0) && (next_candidate.score > score_threshold)) {
191 // Soft suppression might have occurred and current score is still
192 // greater than score_threshold; add next_candidate back onto priority
193 // queue.
194 candidate_priority_queue.push(next_candidate);
195 }
196 }
197 }
198}

203EI_IMPULSE_ERROR ei_run_nms(
204 const ei_impulse_t *impulse,
205 std::vector<ei_impulse_result_bounding_box_t> *results,
206 float *boxes,
207 float *scores,
208 int *classes,
209 size_t bb_count,
210 bool clip_boxes,
211 bool debug) {
212
213 if (bb_count < 1) {
214 return EI_IMPULSE_OK;
215 }
216
217 int *selected_indices = (int*)ei_malloc(1 * bb_count * sizeof(int));
218 float *selected_scores = (float*)ei_malloc(1 * bb_count * sizeof(float));
219
220 if (!scores || !boxes || !selected_indices || !selected_scores || !classes) {
221 ei_free(selected_indices);
222 ei_free(selected_scores);
223 return EI_IMPULSE_OUT_OF_MEMORY;
224 }
225
226 // boxes: box encodings in format [y1, x1, y2, x2], shape: [num_boxes, 4]
227 // num_boxes: number of candidates
228 // scores: scores for candidate boxes, in the same order. shape: [num_boxes]
229 // max_output_size: the maximum number of selections.
230 // iou_threshold: Intersection-over-Union (IoU) threshold for NMS
231 // score_threshold: All candidate scores below this value are rejected
232 // soft_nms_sigma: Soft NMS parameter, used for decaying scores
233
234 int num_selected_indices;
235
236 NonMaxSuppression(
237 (const float*)boxes, // boxes
238 bb_count, // num_boxes
239 (const float*)scores, // scores
240 bb_count, // max_output_size
241 impulse->object_detection_nms.iou_threshold, // iou_threshold
242 impulse->object_detection_nms.confidence_threshold, // score_threshold
243 0.0f, // soft_nms_sigma
244 selected_indices,
245 selected_scores,
246 &num_selected_indices);
247
248 std::vector<ei_impulse_result_bounding_box_t> new_results;
249
250 for (size_t ix = 0; ix < (size_t)num_selected_indices; ix++) {
251
252 int out_ix = selected_indices[ix];
253 ei_impulse_result_bounding_box_t bb;
254 bb.label = impulse->categories[classes[out_ix]];
255 bb.value = selected_scores[ix];
256
257 float ymin = boxes[(out_ix * 4) + 0];
258 float xmin = boxes[(out_ix * 4) + 1];
259 float ymax = boxes[(out_ix * 4) + 2];
260 float xmax = boxes[(out_ix * 4) + 3];
261
262 if (clip_boxes) {
263 ymin = std::min(std::max(ymin, 0.0f), (float)impulse->input_height);
264 xmin = std::min(std::max(xmin, 0.0f), (float)impulse->input_width);
265 ymax = std::min(std::max(ymax, 0.0f), (float)impulse->input_height);
266 xmax = std::min(std::max(xmax, 0.0f), (float)impulse->input_width);
267 }
268
269 bb.y = static_cast<uint32_t>(ymin);
270 bb.x = static_cast<uint32_t>(xmin);
271 bb.height = static_cast<uint32_t>(ymax) - bb.y;
272 bb.width = static_cast<uint32_t>(xmax) - bb.x;
273 new_results.push_back(bb);
274
275 if (debug) {
276 ei_printf("Found bb with label %s\n", bb.label);
277 }
278
279 }
280
281 results->clear();
282
283 for (size_t ix = 0; ix < new_results.size(); ix++) {
284 results->push_back(new_results[ix]);
285 }
286
287 ei_free(selected_indices);
288 ei_free(selected_scores);
289
290 return EI_IMPULSE_OK;
291
292}

297EI_IMPULSE_ERROR ei_run_nms(
298 const ei_impulse_t *impulse,
299 std::vector<ei_impulse_result_bounding_box_t> *results,
300 bool clip_boxes,
301 bool debug) {
302
303 size_t bb_count = 0;
304 for (size_t ix = 0; ix < results->size(); ix++) {
305 auto bb = results->at(ix);
306 if (bb.value == 0) {
307 continue;
308 }
309 bb_count++;
310 }
311
312 if (bb_count < 1) {
313 return EI_IMPULSE_OK;
314 }
315
316 float *boxes = (float*)ei_malloc(4 * bb_count * sizeof(float));
317 float *scores = (float*)ei_malloc(1 * bb_count * sizeof(float));
318 int *classes = (int*) ei_malloc(bb_count * sizeof(int));
319
320 if (!scores || !boxes || !classes) {
321 ei_free(boxes);
322 ei_free(scores);
323 ei_free(classes);
324 return EI_IMPULSE_OUT_OF_MEMORY;
325 }
326
327 size_t box_ix = 0;
328 for (size_t ix = 0; ix < results->size(); ix++) {
329 auto bb = results->at(ix);
330 if (bb.value == 0) {
331 continue;
332 }
333 boxes[(box_ix * 4) + 0] = bb.y;
334 boxes[(box_ix * 4) + 1] = bb.x;
335 boxes[(box_ix * 4) + 2] = bb.y + bb.height;
336 boxes[(box_ix * 4) + 3] = bb.x + bb.width;
337 scores[box_ix] = bb.value;
338
339 for (size_t j = 0; j < impulse->label_count; j++) {
340 if (strcmp(impulse->categories[j], bb.label) == 0)
341 classes[box_ix] = j;
342 }
343
344 box_ix++;
345 }
346
347 EI_IMPULSE_ERROR nms_res = ei_run_nms(impulse, results,
348 boxes, scores,
349 classes, bb_count,
350 clip_boxes,
351 debug);
352
353
354 ei_free(boxes);
355 ei_free(scores);
356 ei_free(classes);
357
358 return nms_res;
359
360}
