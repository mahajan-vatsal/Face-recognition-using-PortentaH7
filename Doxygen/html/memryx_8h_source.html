Machine Vision using Portenta H7 2
This project aims to develop a face recognition-based access control system using the Arduino Portenta H7 and Vision Shield, leveraging Edge Impulse for machine learning. The system captures facial images, processes them locally using an AI model deployed on the Portenta H7 and determines access based on authorised personnel.

112bool init_memryx(bool debug, const ei_impulse_t *impulse)
113{
114 /* Unpack DFP model to file system */
115 std::string project_file_path = "/tmp/" + std::string(impulse->project_name) + "-" + std::to_string(impulse->project_id) + "-" + std::to_string(impulse->deploy_version);
116 create_project_if_not_exists(project_file_path, model_h_files, model_h_files_len);
117
118 std::string proj_model_path = project_file_path + "/memryx_trained.dfp";
119 const char * model_file_path = proj_model_path.c_str();
120#if (defined(EI_CLASSIFIER_USE_MEMRYX_HARDWARE) && (EI_CLASSIFIER_USE_MEMRYX_HARDWARE == 1))
121#warning "Building EIM for use with MemryX Hardware"
122 memx_status status = MEMX_STATUS_OK;
123 // 1. Bind MPU device group 0 as MX3:Cascade to model 0.
124 status = memx_open(model_id, group_id, MEMX_DEVICE_CASCADE);
125 if(memx_status_error(status)) {
126 return false;
127 }
128 ei_printf("Memryx device opened.\n");
129
130 // 2. Download model from a DFP file to MPU device group, input and
131 // output feature map shape is auto, configured after download complete.
132 status = memx_download_model(model_id, model_file_path, 0, // model_idx = 0
133 MEMX_DOWNLOAD_TYPE_WTMEM_AND_MODEL);
134 if(memx_status_error(status)) {
135 return false;
136 }
137 ei_printf("Memryx model downloaded.\n");
138
139 // 3. Enable data transfer of this model to device. Set to no wait here
140 // since driver will go to data transfer state eventually.
141 status = memx_set_stream_enable(model_id, 0);
142 if(memx_status_error(status)) {
143 return false;
144 }
145 ei_printf("Data streaming to and from the MX3 board is enabled\n");
146#elif (defined(EI_CLASSIFIER_USE_MEMRYX_SOFTWARE) && (EI_CLASSIFIER_USE_MEMRYX_SOFTWARE == 1))
147#warning "MEMRYX model will be run in SIMULATION mode (not on real hardware)!"
148 py::list path;
149 // import Python's memryx module
150 try {
151 memryx = py::module_::import("memryx");
152 if(debug) printf("Memryx PyModule init\n");
153 }
154 catch (py::error_already_set &e) {
155 ei_printf("ERR: Importing 'memryx' library failed:\n%s\n", e.what());
156 return false;
157 }
158
159 Simulator = memryx.attr("Simulator");
160 if(debug) printf("Simulator API init\n");
161
162 // load model
163 try {
164 model = Simulator("dfp"_a = model_file_path);
165 if(debug) printf("Model API init\n");
166 }
167 catch (py::error_already_set &e) {
168 ei_printf("ERR: Can't load model file from %s\n", model_file_path);
169 return false;
170 }
171#else
172#error "Neither EI_CLASSIFIER_USE_MEMRYX_HARDWARE or EI_CLASSIFIER_USE_MEMRYX_SOFTWARE are defined or set to 1"
173#endif
174
175 // clear info
176 engine_info.str("");
177
178 return true;
179}

192#if (defined(EI_CLASSIFIER_USE_MEMRYX_HARDWARE) && (EI_CLASSIFIER_USE_MEMRYX_HARDWARE == 1))
193EI_IMPULSE_ERROR run_nn_inference(
194 const ei_impulse_t *impulse,
195 ei_feature_t *fmatrix,
196 uint32_t learn_block_index,
197 uint32_t* input_block_ids,
198 uint32_t input_block_ids_size,
199 ei_impulse_result_t *result,
200 void *config_ptr,
201 bool debug = false)
202{
203 ei_learning_block_config_tflite_graph_t *block_config = (ei_learning_block_config_tflite_graph_t*)config_ptr;
204
205 memx_status status = MEMX_STATUS_OK;
206 int32_t ifmap_height, ifmap_width, ifmap_channel_number, ifmap_format;
207 int32_t ofmap_height, ofmap_width, ofmap_channel_number, ofmap_format;
208 int32_t z;
209 uint64_t ctx_start_us = 0;
210 uint64_t ctx_end_us = 0;
211
212 // check if we've initialized the interpreter and device?
213 if (memryx_initialized == false) {
214 if(init_memryx(debug, impulse) == false) {
215 return EI_IMPULSE_MEMRYX_ERROR;
216 }
217 memryx_initialized = true;
218 }
219
220 /* 4. get input shape - Not needed during runtime, available only for debugging */
221 if(verbose_debug) {
222 status = memx_get_ifmap_size(model_id, flow_id, &ifmap_height, &ifmap_width, &z, &ifmap_channel_number, &ifmap_format);
223 ei_printf("status = %d, ifmap shape = (%d, %d, %d), format = %d\n",
224 status, ifmap_height, ifmap_width, ifmap_channel_number, ifmap_format);
225 }
226
227 // 5. get output shape
228 status = memx_get_ofmap_size(model_id, flow_id, &ofmap_height, &ofmap_width, &z, &ofmap_channel_number, &ofmap_format);
229 if(debug) {
230 ei_printf("status = %d, ofmap shape = (%d, %d, %d), format = %d\n",
231 status, ofmap_height, ofmap_width, ofmap_channel_number, ofmap_format);
232 }
233 if(memx_status_error(status)) {
234 return EI_IMPULSE_MEMRYX_ERROR;
235 }
236
237 // 6. Prepare input and output buffers
238 float* ofmap = new float [ofmap_width * ofmap_height * ofmap_channel_number];
239
240#if EI_CLASSIFIER_SINGLE_FEATURE_INPUT == 0
241 size_t mtx_size = impulse->dsp_blocks_size + impulse->learning_blocks_size;
242 ei::matrix_t* matrix = NULL;
243
244 ei::matrix_t combined_matrix(1, impulse->nn_input_frame_size);
245 uint32_t buf_pos = 0;
246
247 for (size_t i = 0; i < input_block_ids_size; i++) {
248 size_t cur_mtx = input_block_ids[i];
249
250 if (!find_mtx_by_idx(fmatrix, &matrix, cur_mtx, mtx_size)) {
251 ei_printf("ERR: Cannot find matrix with id %zu\n", cur_mtx);
252 return EI_IMPULSE_INVALID_SIZE;
253 }
254
255 for (size_t ix = 0; ix < matrix->rows * matrix->cols; ix++) {
256 combined_matrix.buffer[buf_pos++] = matrix->buffer[ix];
257 }
258 }
259 matrix = &combined_matrix;
260#else
261 ei::matrix_t* matrix = fmatrix[0].matrix;
262#endif
263
264 float* ifmap = (float*)matrix->buffer;
265
266 if(verbose_debug) {
267 for(int fidx = 0; fidx < (ofmap_width*ofmap_height); fidx++) {
268 ei_printf("%f\t", matrix->buffer[fidx]);
269 if(!(fidx % ofmap_width)) ei_printf("\n");
270 }
271 }
272
273 // TODO stream_ifmap only copies buffer to MX3 board,
274 // we need a different approach to measure latency
275 ctx_start_us = ei_read_timer_us();
276 // 7. Stream inputs to device and start inference.
277 status = memx_stream_ifmap(model_id, 0, ifmap, timeout);
278 ctx_end_us = ei_read_timer_us();
279 if(memx_status_error(status)) {
280 return EI_IMPULSE_MEMRYX_ERROR;
281 }
282
283 result->timing.classification_us = ctx_end_us - ctx_start_us;
284 result->timing.classification = (int)(result->timing.classification_us / 1000);
285
286 engine_info.str("");
287 engine_info << "Inferences per second: " << (1000000 / result->timing.classification_us);
288
289 // 6. Stream output results from device after inference
290 status = memx_stream_ofmap(model_id, 0, ofmap, timeout);
291 if(debug) {
292 ei_printf(" memx_stream_ofmap (status=%d)\n", status);
293 }
294 if(memx_status_error(status)) {
295 return EI_IMPULSE_MEMRYX_ERROR;
296 }
297
298 // init softmax shape
299 std::vector<size_t> output_shape = {static_cast<size_t>(ofmap_height),static_cast<size_t>(ofmap_width),
300 static_cast<size_t>(ofmap_channel_number)};
301 softmax_shape.BuildFrom(output_shape);
302 // dumy beta parameter for softmax purposes
303 dummy_params.beta = 1;
304
305 // apply softmax, becuase MX3 does not support this operation
306 tflite::reference_ops::Softmax(dummy_params, softmax_shape, ofmap, softmax_shape, ofmap);
307
308 // handle inference outputs
309 if (block_config->object_detection) {
310 switch (block_config->object_detection_last_layer) {
311 case EI_CLASSIFIER_LAST_LAYER_FOMO: {
312 ei_printf("FOMO executed on Memryx\n");
313 fill_result_struct_f32_fomo(
314 impulse,
315 block_config,
316 result,
317 ofmap,
318 impulse->fomo_output_size,
319 impulse->fomo_output_size);
320 break;
321 }
322 case EI_CLASSIFIER_LAST_LAYER_SSD: {
323 ei_printf("Mobilenet SSD is not implemented for Edge Impulse MemryX engine, please contact Edge Impulse Support\n");
324 break;
325 }
326 default: {
327 ei_printf("ERR: Unsupported object detection last layer (%d)\n",
328 block_config->object_detection_last_layer);
329 return EI_IMPULSE_UNSUPPORTED_INFERENCING_ENGINE;
330 }
331 }
332 }
333 else {
334 fill_result_struct_f32(impulse, result, ofmap, debug);
335 }
336
337 delete ofmap;
338 // Device is closed only at EIM exit, therefore we do not use memx_close()
339 return EI_IMPULSE_OK;
340}

342#elif (defined(EI_CLASSIFIER_USE_MEMRYX_SOFTWARE) && (EI_CLASSIFIER_USE_MEMRYX_SOFTWARE == 1))
343EI_IMPULSE_ERROR run_nn_inference(
344 const ei_impulse_t *impulse,
345 ei_feature_t *fmatrix,
346 uint32_t learn_block_index,
347 uint32_t* inputBlockIds,
348 ei_impulse_result_t *result,
349 void *config_ptr,
350 bool debug = false)
351{
352 ei_learning_block_config_tflite_graph_t *block_config = (ei_learning_block_config_tflite_graph_t*)config_ptr;
353
354 // init Python embedded interpreter (should be called once!)
355 static py::scoped_interpreter guard{};
356
357 // check if we've initialized the interpreter and device?
358 if (memryx_initialized == false) {
359 if(init_memryx(debug, impulse) == false) {
360 return EI_IMPULSE_MEMRYX_ERROR;
361 }
362 memryx_initialized = true;
363 }
364
365 std::vector<size_t> input_shape = {1, impulse->input_width, impulse->input_height, 3};
366 py::array_t<float> input_data(input_shape); // = zeroes(input_shape, 0);
367
368 printf("impulse->w=%d h=%d\n", impulse->input_width, impulse->input_height);
369
370 /*
371 * convert features data to the expected shape (4dim)
372 * For images RGB shape is (width, height, colors)
373 * For images BW shape is (width, height, 1)
374 * For Audio shape is (width, height, 1) - spectrogram
375 */
376 auto r = input_data.mutable_unchecked<4>();
377
378 for (size_t i = 0; i < input_block_ids_size; i++) {
379 uint16_t cur_mtx = input_block_ids[i];
380#if EI_CLASSIFIER_SINGLE_FEATURE_INPUT == 0
381 ei::matrix_t* matrix = NULL;
382
383 if (!find_mtx_by_idx(fmatrix, &matrix, cur_mtx, mtx_size)) {
384 ei_printf("ERR: Cannot find matrix with id %zu\n", cur_mtx);
385 return EI_IMPULSE_INVALID_SIZE;
386 }
387#else
388 ei::matrix_t* matrix = fmatrix[0].matrix;
389#endif
390 for (py::ssize_t x = 0; x < r.shape(1); x++) {
391 for (py::ssize_t y = 0; y < r.shape(2); y++) {
392 for(py::ssize_t z = 0; z < r.shape(3); z++) {
393 r(0, x, y, z) = (float)(fmatrix.buffer[x * r.shape(2) * r.shape(3) + y * r.shape(3) + z]);
394 }
395 }
396 }
397 }
398
399 py::object runmodel = model.attr("run");
400 // result from mx_sim is {np array, float, float}
401 py::tuple args = py::make_tuple(py::none(), 0.00, 0.00);
402 // run inference in sumualtor
403 printf("start inference\n");
404 uint64_t ctx_start_us = ei_read_timer_us();
405 args = runmodel("inputs"_a=input_data,"frames"_a=1);
406 uint64_t ctx_end_us = ei_read_timer_us();
407 printf("end of inference\n");
408
409 result->timing.classification_us = ctx_end_us - ctx_start_us;
410 result->timing.classification = (int)(result->timing.classification_us / 1000);
411
412 engine_info.str("");
413 engine_info << "Inferences per second: " << (1000000 / result->timing.classification_us);
414
415 py::array outputs = py::list(args[0]);
416 py::array_t<float> potentials;
417 std::vector<float> potentials_v;
418
419 potentials = outputs.squeeze().cast<py::array_t<float>>();
420
421 if (block_config->object_detection == false) {
422 potentials_v = outputs.squeeze().cast<std::vector<float>>();
423 }
424 else {
425 auto q = potentials.unchecked<>();
426 for (py::ssize_t x = 0; x < q.shape(0); x++) {
427 for (py::ssize_t y = 0; y < q.shape(1); y++) {
428 for(py::ssize_t z = 0; z < q.shape(2); z++) {
429 potentials_v.push_back(q(x, y, z));
430 }
431 }
432 }
433 }
434
435 if(debug) {
436 std::string ret_str = py::str(potentials).cast<std::string>();
437 ei_printf("Memryx raw output:\n%s\n", ret_str.c_str());
438 }
439
440 if (block_config->object_detection) {
441 switch (block_config->object_detection_last_layer) {
442 case EI_CLASSIFIER_LAST_LAYER_FOMO: {
443 ei_printf("FOMO executed on Memryx\n");
444 fill_result_struct_f32_fomo(
445 impulse,
446 block_config,
447 result,
448 potentials_v.data(),
449 impulse->fomo_output_size,
450 impulse->fomo_output_size);
451 break;
452 }
453 case EI_CLASSIFIER_LAST_LAYER_SSD: {
454 ei_printf("Mobilenet SSD executed on Memryx\n");
455 break;
456 }
457 default: {
458 ei_printf("ERR: Unsupported object detection last layer (%d)\n",
459 impulse->object_detection_last_layer);
460 return EI_IMPULSE_UNSUPPORTED_INFERENCING_ENGINE;
461 }
462 }
463 }
464 else {
465 fill_result_struct_f32(impulse, result, potentials_v.data(), debug);
466 }
467
468 return EI_IMPULSE_OK;
469}
