212template <typename T>
213TfLiteTensor CreateTensor(const T* data, TfLiteIntArray* dims,
214 const bool is_variable = false,
215 TfLiteType type = kTfLiteNoType) {
216 TfLiteTensor result;
217 result.dims = dims;
218 result.params = {};
219 result.quantization = {kTfLiteNoQuantization, nullptr};
220 result.is_variable = is_variable;
221 result.allocation_type = kTfLiteMemNone;
222 result.data.data = const_cast<T*>(data);
223 result.bytes = ElementCount(*dims) * sizeof(T);
224 result.data.data = const_cast<T*>(data);
225
226 if (type == kTfLiteInt4) {
227 result.type = kTfLiteInt4;
228 PackInt4ValuesDenselyInPlace(tflite::GetTensorData<uint8_t>(&result),
229 ElementCount(*dims));
230 result.bytes = ((ElementCount(*dims) + 1) / 2);
231 } else {
232 // Const cast is used to allow passing in const and non-const arrays within
233 // a single CreateTensor method. A Const array should be used for immutable
234 // input tensors and non-const array should be used for mutable and output
235 // tensors.
236 result.type = typeToTfLiteType<T>();
237 }
238 return result;
239}

304// Derives the asymmetric quantization scaling factor from a min and max range.
305template <typename T>
306inline float ScaleFromMinMax(const float min, const float max) {
307 return (max - min) /
308 static_cast<float>((std::numeric_limits<T>::max() * 1.0) -
309 std::numeric_limits<T>::min());
310}
311
312// Derives the symmetric quantization scaling factor from a min and max range.
313template <typename T>
314inline float SymmetricScaleFromMinMax(const float min, const float max) {
315 const int32_t kScale =
316 std::numeric_limits<typename std::make_signed<T>::type>::max();
317 const float range = std::max(std::abs(min), std::abs(max));
318 if (range == 0) {
319 return 1.0f;
320 } else {
321 return range / kScale;
322 }
323}
324
325// Derives the quantization zero point from a min and max range.
326template <typename T>
327inline int ZeroPointFromMinMax(const float min, const float max) {
328 return static_cast<int>(std::numeric_limits<T>::min()) +
329 static_cast<int>(roundf(-min / ScaleFromMinMax<T>(min, max)));
330}
