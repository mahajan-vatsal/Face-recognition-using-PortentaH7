Machine Vision using Portenta H7 2
This project aims to develop a face recognition-based access control system using the Arduino Portenta H7 and Vision Shield, leveraging Edge Impulse for machine learning. The system captures facial images, processes them locally using an AI model deployed on the Portenta H7 and determines access based on authorised personnel.

53EI_IMPULSE_ERROR run_nn_inference(
54 const ei_impulse_t *impulse,
55 ei_feature_t *fmatrix,
56 uint32_t learn_block_index,
57 uint32_t* input_block_ids,
58 uint32_t input_block_ids_size,
59 ei_impulse_result_t *result,
60 void *config_ptr,
61 bool debug)
62{
63 ei_learning_block_config_tflite_graph_t *block_config = (ei_learning_block_config_tflite_graph_t*)config_ptr;
64
65 static std::unique_ptr<tflite::FlatBufferModel> model = nullptr;
66 static std::unique_ptr<tflite::Interpreter> interpreter = nullptr;
67 static std::vector<int> inputs;
68 static std::vector<int> outputs;
69
70 if (!model) {
71
72 std::string proj_artifacts_path = "/tmp/" + std::string(impulse->project_name) + "-" + std::to_string(impulse->project_id) + "-" + std::to_string(impulse->deploy_version);
73
74 create_project_if_not_exists(proj_artifacts_path, model_h_files, model_h_files_len);
75
76 std::string proj_model_path = proj_artifacts_path + "/trained.tflite";
77
78 model = tflite::FlatBufferModel::BuildFromFile(proj_model_path.c_str());
79 if (!model) {
80 ei_printf("Failed to build TFLite model from buffer\n");
81 return EI_IMPULSE_TFLITE_ERROR;
82 }
83
84 tflite::ops::builtin::BuiltinOpResolver resolver;
85 tflite::InterpreterBuilder builder(*model, resolver);
86 builder(&interpreter);
87
88 if (!interpreter) {
89 ei_printf("Failed to construct interpreter\n");
90 return EI_IMPULSE_TFLITE_ERROR;
91 }
92
93 /* This part creates the dlg_ptr */
94 ei_printf("TIDL delegate mode\n");
95 typedef TfLiteDelegate *(*tflite_plugin_create_delegate)(char **, char **, size_t, void (*report_error)(const char *));
96 tflite_plugin_create_delegate tflite_plugin_dlg_create;
97 char *keys[] = {(char *)"artifacts_folder", (char *)"num_tidl_subgraphs", (char *)"debug_level"};
98 char *values[] = {(char *)proj_artifacts_path.c_str(), (char *)"16", (char *)"0"};
99 void *lib = dlopen("libtidl_tfl_delegate.so", RTLD_NOW);
100 assert(lib);
101 tflite_plugin_dlg_create = (tflite_plugin_create_delegate)dlsym(lib, "tflite_plugin_create_delegate");
102 TfLiteDelegate *dlg_ptr = tflite_plugin_dlg_create(keys, values, 3, NULL);
103 interpreter->ModifyGraphWithDelegate(dlg_ptr);
104 ei_printf("ModifyGraphWithDelegate - Done \n");
105
106
107 if (interpreter->AllocateTensors() != kTfLiteOk) {
108 ei_printf("AllocateTensors failed\n");
109 return EI_IMPULSE_TFLITE_ERROR;
110 }
111
112 int hw_thread_count = (int)std::thread::hardware_concurrency();
113 hw_thread_count -= 1; // leave one thread free for the other application
114 if (hw_thread_count < 1) {
115 hw_thread_count = 1;
116 }
117
118 if (interpreter->SetNumThreads(hw_thread_count) != kTfLiteOk) {
119 ei_printf("SetNumThreads failed\n");
120 return EI_IMPULSE_TFLITE_ERROR;
121 }
122 }
123
124 inputs = interpreter->inputs();
125 outputs = interpreter->outputs();
126
127 ei_printf("device mem enabled\n");
128 for (uint32_t i = 0; i < inputs.size(); i++)
129 {
130 const TfLiteTensor *tensor = interpreter->input_tensor(i);
131 in_ptrs[i] = TIDLRT_allocSharedMem(tflite::kDefaultTensorAlignment, tensor->bytes);
132 if (in_ptrs[i] == NULL)
133 {
134 ei_printf("Could not allocate Memory for input: %s\n", tensor->name);
135 }
136 interpreter->SetCustomAllocationForTensor(inputs[i], {in_ptrs[i], tensor->bytes});
137 }
138 for (uint32_t i = 0; i < outputs.size(); i++)
139 {
140 const TfLiteTensor *tensor = interpreter->output_tensor(i);
141 out_ptrs[i] = TIDLRT_allocSharedMem(tflite::kDefaultTensorAlignment, tensor->bytes);
142 if (out_ptrs[i] == NULL)
143 {
144 ei_printf("Could not allocate Memory for ouput: %s\n", tensor->name);
145 }
146 interpreter->SetCustomAllocationForTensor(outputs[i], {out_ptrs[i], tensor->bytes});
147 }
148
149 // Obtain pointers to the model's input and output tensors.
150#if EI_CLASSIFIER_QUANTIZATION_ENABLED == 1
151 int8_t* input = interpreter->typed_input_tensor<int8_t>(0);
152#else
153 float* input = interpreter->typed_input_tensor<float>(0);
154#endif
155
156 if (!input) {
157 return EI_IMPULSE_INPUT_TENSOR_WAS_NULL;
158 }
159
160 size_t mtx_size = impulse->dsp_blocks_size + impulse->learning_blocks_size;
161
162 for (size_t i = 0; i < input_block_ids_size; i++) {
163#if EI_CLASSIFIER_SINGLE_FEATURE_INPUT == 0
164 uint16_t cur_mtx = input_block_ids[i];
165 ei::matrix_t* matrix = NULL;
166
167 if (!find_mtx_by_idx(fmatrix, &matrix, cur_mtx, mtx_size)) {
168 ei_printf("ERR: Cannot find matrix with id %zu\n", cur_mtx);
169 return EI_IMPULSE_INVALID_SIZE;
170 }
171#else
172 ei::matrix_t* matrix = fmatrix[0].matrix;
173#endif
174
175 for (uint32_t ix = 0; ix < matrix->rows * matrix->cols; ix++) {
176 if (block_config->object_detection) {
177#if EI_CLASSIFIER_QUANTIZATION_ENABLED == 1
178 float pixel = (float)matrix->buffer[ix];
179 input[ix] = static_cast<uint8_t>((pixel / input->tflite_input_scale) + input->tflite_input_zeropoint);
180#else
181 input[ix] = matrix->buffer[ix];
182#endif
183 }
184 else {
185#if EI_CLASSIFIER_QUANTIZATION_ENABLED == 1
186 input[ix] = static_cast<int8_t>(round(matrix->buffer[ix] / input->tflite_input_scale) + input->tflite_input_zeropoint);
187#else
188 input[ix] = matrix->buffer[ix];
189#endif
190 }
191 }
192 }
193
194 uint64_t ctx_start_us = ei_read_timer_us();
195
196 interpreter->Invoke();
197
198 uint64_t ctx_end_us = ei_read_timer_us();
199
200 result->timing.classification_us = ctx_end_us - ctx_start_us;
201 result->timing.classification = (int)(result->timing.classification_us / 1000);
202
203#if EI_CLASSIFIER_QUANTIZATION_ENABLED == 1
204 int8_t* out_data = interpreter->typed_output_tensor<int8_t>(block_config->output_data_tensor);
205#else
206 float* out_data = interpreter->typed_output_tensor<float>(block_config->output_data_tensor);
207#endif
208
209 if (debug) {
210 ei_printf("LOG_INFO tensors size: %ld \n", interpreter->tensors_size());
211 ei_printf("LOG_INFO nodes size: %ld\n", interpreter->nodes_size());
212 ei_printf("LOG_INFO number of inputs: %ld\n", inputs.size());
213 ei_printf("LOG_INFO number of outputs: %ld\n", outputs.size());
214 ei_printf("LOG_INFO input(0) name: %s\n", interpreter->GetInputName(0));
215
216 int t_size = interpreter->tensors_size();
217 for (int i = 0; i < t_size; i++)
218 {
219 if (interpreter->tensor(i)->name) {
220 ei_printf("LOG_INFO %d: %s,%ld,%d,%f,%d,size(", i, interpreter->tensor(i)->name,
221 interpreter->tensor(i)->bytes,
222 interpreter->tensor(i)->type,
223 interpreter->tensor(i)->params.scale,
224 interpreter->tensor(i)->params.zero_point);
225
226 for (int k=0; k < interpreter->tensor(i)->dims->size; k++) {
227 if (k == interpreter->tensor(i)->dims->size - 1) {
228 ei_printf("%d", interpreter->tensor(i)->dims->data[k]);
229 } else {
230 ei_printf("%d,", interpreter->tensor(i)->dims->data[k]);
231 }
232 }
233 ei_printf(")\n");
234 }
235 }
236 }
237
238 if (!out_data) {
239 return EI_IMPULSE_OUTPUT_TENSOR_WAS_NULL;
240 }
241
242 if (debug) {
243 ei_printf("Predictions (time: %d ms.):\n", result->timing.classification);
244 }
245
246 EI_IMPULSE_ERROR fill_res = EI_IMPULSE_OK;
247
248 if (block_config->object_detection) {
249 switch (block_config->object_detection_last_layer) {
250 case EI_CLASSIFIER_LAST_LAYER_FOMO: {
251 if (block_config->quantized == 1) {
252#if EI_CLASSIFIER_QUANTIZATION_ENABLED == 1
253 fill_res = fill_result_struct_i8_fomo(
254 impulse,
255 block_config,
256 result,
257 out_data,
258 out_data->tflite_output_zeropoint,
259 out_data->tflite_output_scale,
260 impulse->fomo_output_size,
261 impulse->fomo_output_size);
262#endif
263 }
264 else {
265 fill_res = fill_result_struct_f32_fomo(
266 impulse,
267 block_config,
268 result,
269 out_data,
270 impulse->fomo_output_size,
271 impulse->fomo_output_size);
272 }
273 break;
274 }
275 case EI_CLASSIFIER_LAST_LAYER_SSD: {
276 float *scores_tensor = interpreter->typed_output_tensor<float>(block_config->output_score_tensor);
277 float *label_tensor = interpreter->typed_output_tensor<float>(block_config->output_labels_tensor);
278 if (!scores_tensor) {
279 return EI_IMPULSE_SCORE_TENSOR_WAS_NULL;
280 }
281 if (!label_tensor) {
282 return EI_IMPULSE_LABEL_TENSOR_WAS_NULL;
283 }
284 if (block_config->quantized == 1) {
285 ei_printf("ERR: MobileNet SSD does not support quantized inference\n");
286 return EI_IMPULSE_UNSUPPORTED_INFERENCING_ENGINE;
287 }
288 else {
289 fill_res = fill_result_struct_f32_object_detection(
290 impulse,
291 block_config,
292 result,
293 out_data,
294 scores_tensor,
295 label_tensor,
296 debug);
297 }
298 break;
299 }
300 case EI_CLASSIFIER_LAST_LAYER_YOLOV5:
301 case EI_CLASSIFIER_LAST_LAYER_YOLOV5_V5_DRPAI: {
302 if (block_config->quantized == 1) {
303 ei_printf("ERR: YOLOv5 does not support quantized inference\n");
304 return EI_IMPULSE_UNSUPPORTED_INFERENCING_ENGINE;
305 }
306 else {
307 int version = block_config->object_detection_last_layer == EI_CLASSIFIER_LAST_LAYER_YOLOV5_V5_DRPAI ?
308 5 : 6;
309 fill_res = fill_result_struct_f32_yolov5(
310 impulse,
311 block_config,
312 result,
313 version,
314 out_data,
315 impulse->tflite_output_features_count,
316 debug);
317 }
318 break;
319 }
320 case EI_CLASSIFIER_LAST_LAYER_YOLOX: {
321 if (block_config->quantized == 1) {
322 ei_printf("ERR: YOLOX does not support quantized inference\n");
323 return EI_IMPULSE_UNSUPPORTED_INFERENCING_ENGINE;
324 }
325 else {
326 fill_res = fill_result_struct_f32_yolox(
327 impulse,
328 block_config,
329 result,
330 out_data,
331 impulse->tflite_output_features_count,
332 debug);
333 }
334 break;
335 }
336 case EI_CLASSIFIER_LAST_LAYER_YOLOV7: {
337 if (block_config->quantized == 1) {
338 ei_printf("ERR: YOLOV7 does not support quantized inference\n");
339 return EI_IMPULSE_UNSUPPORTED_INFERENCING_ENGINE;
340 }
341 else {
342 TfLiteTensor *output = interpreter->output_tensor(0);
343 size_t output_feature_count = 1;
344 for (int ix = 0; ix < output->dims->size; ix++) {
345 output_feature_count *= output->dims->data[ix];
346 }
347 fill_res = fill_result_struct_f32_yolov7(
348 impulse,
349 block_config,
350 result,
351 output->data.f,
352 output_feature_count);
353 }
354 break;
355 }
356 default: {
357 ei_printf("ERR: Unsupported object detection last layer (%d)\n",
358 block_config->object_detection_last_layer);
359 break;
360 }
361 }
362 }
363 else {
364#if EI_CLASSIFIER_QUANTIZATION_ENABLED == 1
365 fill_res = fill_result_struct_i8(impulse, result, out_data, out_data->tflite_output_zeropoint, out_data->tflite_output_scale, debug);
366#else
367 fill_res = fill_result_struct_f32(impulse, result, out_data, debug);
368#endif
369 }
370
371 for (uint32_t i = 0; i < inputs.size(); i++)
372 {
373 if (in_ptrs[i])
374 {
375 TIDLRT_freeSharedMem(in_ptrs[i]);
376 }
377 }
378 for (uint32_t i = 0; i < outputs.size(); i++)
379 {
380 if (out_ptrs[i])
381 {
382 TIDLRT_freeSharedMem(out_ptrs[i]);
383 }
384 }
385
386 if (fill_res != EI_IMPULSE_OK) {
387 return fill_res;
388 }
389
390 // on Linux we're not worried about free'ing (for now)
391
392 return EI_IMPULSE_OK;
393}
