31 if (hi < v) return hi;
32 if (v < lo) return lo;

36inline void StridedSlicePadIndices(tflite::StridedSliceParams* p,
37 int dim_count) {
38 // Add indices and mask bits to fully include extra dimensions
39 TFLITE_CHECK_LE(dim_count, 5);
40 TFLITE_CHECK_GE(dim_count, p->start_indices_count);
41 TFLITE_CHECK_EQ(p->start_indices_count, p->stop_indices_count);
42 TFLITE_CHECK_EQ(p->stop_indices_count, p->strides_count);
43
44 const int pad_count = dim_count - p->start_indices_count;
45
46 // Pad indices at start, so move arrays by pad_count.
47 for (int i = p->start_indices_count - 1; i >= 0; --i) {
48 p->strides[i + pad_count] = p->strides[i];
49 p->start_indices[i + pad_count] = p->start_indices[i];
50 p->stop_indices[i + pad_count] = p->stop_indices[i];
51 }
52 for (int i = 0; i < pad_count; ++i) {
53 p->start_indices[i] = 0;
54 p->stop_indices[i] = 1;
55 p->strides[i] = 1;
56 }
57
58 // Pad masks with 0s or 1s as required.
59 p->shrink_axis_mask <<= pad_count;
60 p->ellipsis_mask <<= pad_count;
61 p->new_axis_mask <<= pad_count;
62 p->begin_mask <<= pad_count;
63 p->end_mask <<= pad_count;
64 p->begin_mask |= (1 << pad_count) - 1;
65 p->end_mask |= (1 << pad_count) - 1;
66
67 p->start_indices_count = dim_count;
68 p->stop_indices_count = dim_count;
69 p->strides_count = dim_count;
70}

72// Return the index for the first element along that axis. This index will be a
73// positive integer between [0, axis_size] (or [-1, axis_size -1] if stride < 0)
74// that can be used to index directly into the data.
75inline int StridedSliceStartForAxis(const tflite::StridedSliceParams& params,
76 const RuntimeShape& input_shape,
77 int32_t axis) {
78 const int32_t axis_size = input_shape.Dims(axis);
79 int32_t start = params.start_indices[axis];
80 const int32_t stride = params.strides[axis];
81 const int32_t begin_mask = (params.begin_mask & 1 << axis);
82 if (start < 0) {
83 start += axis_size;
84 }
85 if (stride > 0) {
86 start = Clamp(start, 0, axis_size);
87 } else {
88 start = Clamp(start, -1, axis_size - 1);
89 }
90 if (begin_mask) {
91 if (stride > 0) {
92 start = 0;
93 } else {
94 start = axis_size - 1;
95 }
96 }
97 return start;
98}

100inline int StridedSliceEndForAxis(const tflite::StridedSliceParams& params,
101 const RuntimeShape& input_shape, int axis,
102 int start) {
103 const auto shrink_axis_mask = params.shrink_axis_mask;
104 const bool shrink_axis = shrink_axis_mask & (1 << axis);
105 const int axis_size = input_shape.Dims(axis);
106 const bool offset = params.offset;
107 if (shrink_axis) {
108 if (start >= axis_size) {
109 return start;
110 } else {
111 return start + 1;
112 }
113 }
114 const auto* indices = params.stop_indices;
115 int end = indices[axis];
116 if (offset) {
117 end += start;
118 }
119 const int32_t stride = params.strides[axis];
120 const int32_t end_mask = (params.end_mask & 1 << axis);
121 if (end < 0) {
122 end += axis_size;
123 }
124 if (stride > 0) {
125 end = Clamp(end, 0, axis_size);
126 } else {
127 end = Clamp(end, -1, axis_size - 1);
128 }
129 if (end_mask) {
130 if (stride > 0) {
131 end = axis_size;
132 } else {
133 end = -1;
134 }
135 }
136 return end;
137}

139// Return the index for the first element along that axis. This index will be a
140// positive integer between [0, axis_size] (or [-1, axis_size -1] if stride < 0)
141// that can be used to index directly into the data.
142inline int StartForAxis(const tflite::StridedSliceParams& params,
143 const RuntimeShape& input_shape, int axis) {
144 const auto begin_mask = params.begin_mask;
145 const auto* start_indices = params.start_indices;
146 const auto* strides = params.strides;
147 const int axis_size = input_shape.Dims(axis);
148 if (axis_size == 0) {
149 return 0;
150 }
151 // Begin with the specified index.
152 int start = start_indices[axis];
153
154 // begin_mask override
155 if (begin_mask & 1 << axis) {
156 if (strides[axis] > 0) {
157 // Forward iteration - use the first element. These values will get
158 // clamped below (Note: We could have set them to 0 and axis_size-1, but
159 // use lowest() and max() to maintain symmetry with StopForAxis())
160 start = std::numeric_limits<int>::lowest();
161 } else {
162 // Backward iteration - use the last element.
163 start = std::numeric_limits<int>::max();
164 }
165 }
166
167 // Handle negative indices
168 if (start < 0) {
169 start += axis_size;
170 }
171
172 // Clamping
173 if (strides[axis] > 0) {
174 // Forward iteration
175 start = Clamp(start, 0, axis_size);
176 } else {
177 // Backward iteration
178 start = Clamp(start, -1, axis_size - 1);
179 }
180
181 return start;
182}

184// Return the "real" index for the end of iteration along that axis. This is an
185// "end" in the traditional C sense, in that it points to one past the last
186// element. ie. So if you were iterating through all elements of a 1D array of
187// size 4, this function would return 4 as the stop, because it is one past the
188// "real" indices of 0, 1, 2 & 3.
189inline int StopForAxis(const tflite::StridedSliceParams& params,
190 const RuntimeShape& input_shape, int axis,
191 int start_for_axis) {
192 const auto end_mask = params.end_mask;
193 const auto shrink_axis_mask = params.shrink_axis_mask;
194 const auto* stop_indices = params.stop_indices;
195 const auto* strides = params.strides;
196 const int axis_size = input_shape.Dims(axis);
197 if (axis_size == 0) {
198 return 0;
199 }
200
201 // Begin with the specified index
202 const bool shrink_axis = shrink_axis_mask & (1 << axis);
203 int stop = stop_indices[axis];
204
205 // When shrinking an axis, the end position does not matter (and can be
206 // incorrect when negative indexing is used, see Issue #19260). Always use
207 // start_for_axis + 1 to generate a length 1 slice, since start_for_axis has
208 // already been adjusted for negative indices.
209 if (shrink_axis) {
210 return start_for_axis + 1;
211 }
212
213 // end_mask override
214 if (end_mask & (1 << axis)) {
215 if (strides[axis] > 0) {
216 // Forward iteration - use the last element. These values will get
217 // clamped below
218 stop = std::numeric_limits<int>::max();
219 } else {
220 // Backward iteration - use the first element.
221 stop = std::numeric_limits<int>::lowest();
222 }
223 }
224
225 // Handle negative indices
226 if (stop < 0) {
227 stop += axis_size;
228 }
229
230 // Clamping
231 // Because the end index points one past the last element, we need slightly
232 // different clamping ranges depending on the direction.
233 if (strides[axis] > 0) {
234 // Forward iteration
235 stop = Clamp(stop, 0, axis_size);
236 } else {
237 // Backward iteration
238 stop = Clamp(stop, -1, axis_size - 1);
239 }
240
241 return stop;
242}

244inline bool LoopCondition(int index, int stop, int stride) {
245 // True when we have reached the end of an axis and should loop.
246 return stride > 0 ? index >= stop : index <= stop;
247}

249inline tflite::StridedSliceParams BuildStridedSliceParams(
250 int begin_mask, int end_mask, int shrink_axis_mask,
251 const std::vector<int>& start_indices, const std::vector<int>& stop_indices,
252 const std::vector<int>& strides) {
253 tflite::StridedSliceParams op_params{};
254 const int dims_count = start_indices.size();
255
256 op_params.start_indices_count = dims_count;
257 op_params.stop_indices_count = dims_count;
258 op_params.strides_count = dims_count;
259 for (int i = 0; i < dims_count; ++i) {
260 op_params.start_indices[i] = start_indices[i];
261 op_params.stop_indices[i] = stop_indices[i];
262 op_params.strides[i] = strides[i];
263 }
264
265 op_params.begin_mask = begin_mask;
266 op_params.ellipsis_mask = 0;
267 op_params.end_mask = end_mask;
268 op_params.new_axis_mask = 0;
269 op_params.shrink_axis_mask = shrink_axis_mask;
270
271 return op_params;
272}
