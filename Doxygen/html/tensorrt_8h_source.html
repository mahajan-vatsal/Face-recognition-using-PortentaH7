Machine Vision using Portenta H7 2
This project aims to develop a face recognition-based access control system using the Arduino Portenta H7 and Vision Shield, leveraging Edge Impulse for machine learning. The system captures facial images, processes them locally using an AI model deployed on the Portenta H7 and determines access based on authorised personnel.

64EI_IMPULSE_ERROR run_nn_inference(
65 const ei_impulse_t *impulse,
66 ei_feature_t *fmatrix,
67 uint32_t learn_block_index,
68 uint32_t* input_block_ids,
69 uint32_t input_block_ids_size,
70 ei_impulse_result_t *result,
71 void *config_ptr,
72 bool debug = false)
73{
74 ei_learning_block_config_tflite_graph_t *block_config = (ei_learning_block_config_tflite_graph_t*)config_ptr;
75 ei_config_tflite_graph_t *graph_config = (ei_config_tflite_graph_t*)block_config->graph_config;
76
77 #if EI_CLASSIFIER_QUANTIZATION_ENABLED == 1
78 #error "TensorRT requires an unquantized network"
79 #endif
80
81 static char current_exe_path[PATH_MAX] = { 0 };
82
83#if __APPLE__
84 uint32_t len = PATH_MAX;
85 if (_NSGetExecutablePath(current_exe_path, &len) != 0) {
86 current_exe_path[0] = '\0'; // buffer too small
87 }
88 else {
89 // resolve symlinks, ., .. if possible
90 char *canonical_path = realpath(current_exe_path, NULL);
91 if (canonical_path != NULL)
92 {
93 strncpy(current_exe_path, canonical_path, len);
94 free(canonical_path);
95 }
96 }
97#else
98 int readlink_res = readlink("/proc/self/exe", current_exe_path, PATH_MAX);
99 if (readlink_res < 0) {
100 printf("readlink_res = %d\n", readlink_res);
101 current_exe_path[0] = '\0'; // failed to find location
102 }
103#endif
104
105 static char model_file_name[PATH_MAX];
106
107 if (strlen(current_exe_path) == 0) {
108 // could not determine current exe path, use /tmp for the engine file
109 snprintf(
110 model_file_name,
111 PATH_MAX,
112 "/tmp/ei-%d-%d.engine",
113 impulse->project_id,
114 impulse->deploy_version);
115 }
116 else {
117 std::filesystem::path p(current_exe_path);
118 snprintf(
119 model_file_name,
120 PATH_MAX,
121 "%s/%s-project%d-v%d.engine",
122 p.parent_path().c_str(),
123 p.stem().c_str(),
124 impulse->project_id,
125 impulse->deploy_version);
126 }
127
128 static bool first_run = true;
129
130 if (first_run) {
131
132 bool fexists = file_exists(model_file_name);
133 if (!fexists) {
134 ei_printf("INFO: Model file '%s' does not exist, creating...\n", model_file_name);
135
136 FILE *file = fopen(model_file_name, "w");
137 if (!file) {
138 ei_printf("ERR: TensorRT init failed to open '%s'\n", model_file_name);
139 return EI_IMPULSE_TENSORRT_INIT_FAILED;
140 }
141
142 if (fwrite(graph_config->model, graph_config->model_size, 1, file) != 1) {
143 ei_printf("ERR: TensorRT init fwrite failed.\n");
144 return EI_IMPULSE_TENSORRT_INIT_FAILED;
145 }
146
147 if (fclose(file) != 0) {
148 ei_printf("ERR: TensorRT init fclose failed.\n");
149 return EI_IMPULSE_TENSORRT_INIT_FAILED;
150 }
151 }
152
153 first_run = false;
154 }
155
156 uint32_t out_data_size = 0;
157
158 if (block_config->object_detection) {
159 switch (block_config->object_detection_last_layer) {
160 case EI_CLASSIFIER_LAST_LAYER_TAO_SSD:
161 case EI_CLASSIFIER_LAST_LAYER_TAO_RETINANET:
162 case EI_CLASSIFIER_LAST_LAYER_TAO_YOLOV3:
163 case EI_CLASSIFIER_LAST_LAYER_TAO_YOLOV4:
164 case EI_CLASSIFIER_LAST_LAYER_FOMO:
165 case EI_CLASSIFIER_LAST_LAYER_YOLOV5:
166 case EI_CLASSIFIER_LAST_LAYER_YOLOV5_V5_DRPAI: {
167 out_data_size = impulse->tflite_output_features_count;
168 break;
169 }
170 default: {
171 ei_printf(
172 "ERR: Unsupported object detection last layer (%d)\n",
173 block_config->object_detection_last_layer);
174 return EI_IMPULSE_UNSUPPORTED_INFERENCING_ENGINE;
175 }
176 }
177 }
178 else {
179 out_data_size = impulse->label_count;
180 }
181
182 float *out_data = (float*)ei_malloc(out_data_size * sizeof(float));
183 if (out_data == nullptr) {
184 ei_printf("ERR: Cannot allocate memory for output data \n");
185 }
186
187 // lazy initialize tensorRT context
188 if (ei_trt_handle == nullptr) {
189 ei_trt_handle = libeitrt::create_EiTrt(model_file_name, debug);
190 }
191
192#if EI_CLASSIFIER_SINGLE_FEATURE_INPUT == 0
193 size_t mtx_size = impulse->dsp_blocks_size + impulse->learning_blocks_size;
194 ei::matrix_t* matrix = NULL;
195
196 ei::matrix_t combined_matrix(1, impulse->nn_input_frame_size);
197 uint32_t buf_pos = 0;
198
199 for (size_t i = 0; i < input_block_ids_size; i++) {
200 size_t cur_mtx = input_block_ids[i];
201
202 if (!find_mtx_by_idx(fmatrix, &matrix, cur_mtx, mtx_size)) {
203 ei_printf("ERR: Cannot find matrix with id %zu\n", cur_mtx);
204 return EI_IMPULSE_INVALID_SIZE;
205 }
206
207 for (size_t ix = 0; ix < matrix->rows * matrix->cols; ix++) {
208 combined_matrix.buffer[buf_pos++] = matrix->buffer[ix];
209 }
210 }
211 matrix = &combined_matrix;
212#else
213 ei::matrix_t* matrix = fmatrix[0].matrix;
214#endif
215
216 uint64_t ctx_start_us = ei_read_timer_us();
217
218 libeitrt::infer(ei_trt_handle, matrix->buffer, out_data, out_data_size);
219
220 uint64_t ctx_end_us = ei_read_timer_us();
221
222 result->timing.classification_us = ctx_end_us - ctx_start_us;
223 result->timing.classification = (int)(result->timing.classification_us / 1000);
224
225 EI_IMPULSE_ERROR fill_res = EI_IMPULSE_OK;
226
227 if (block_config->object_detection) {
228 switch (block_config->object_detection_last_layer) {
229 case EI_CLASSIFIER_LAST_LAYER_FOMO: {
230 fill_res = fill_result_struct_f32_fomo(
231 impulse,
232 block_config,
233 result,
234 out_data,
235 impulse->fomo_output_size,
236 impulse->fomo_output_size);
237 break;
238 }
239 case EI_CLASSIFIER_LAST_LAYER_YOLOV5:
240 case EI_CLASSIFIER_LAST_LAYER_YOLOV5_V5_DRPAI: {
241 int version = block_config->object_detection_last_layer == EI_CLASSIFIER_LAST_LAYER_YOLOV5_V5_DRPAI ?
242 5 : 6;
243 fill_res = fill_result_struct_f32_yolov5(
244 impulse,
245 block_config,
246 result,
247 version,
248 out_data,
249 impulse->tflite_output_features_count);
250 break;
251 }
252 case EI_CLASSIFIER_LAST_LAYER_TAO_SSD:
253 case EI_CLASSIFIER_LAST_LAYER_TAO_RETINANET: {
254 fill_res = fill_result_struct_f32_tao_decode_detections(
255 impulse,
256 block_config,
257 result,
258 out_data,
259 impulse->tflite_output_features_count,
260 debug);
261 break;
262 }
263 case EI_CLASSIFIER_LAST_LAYER_TAO_YOLOV3:
264 fill_res = fill_result_struct_f32_tao_yolov3(
265 impulse,
266 block_config,
267 result,
268 out_data,
269 impulse->tflite_output_features_count,
270 debug);
271 break;
272 case EI_CLASSIFIER_LAST_LAYER_TAO_YOLOV4: {
273 fill_res = fill_result_struct_f32_tao_yolov4(
274 impulse,
275 block_config,
276 result,
277 out_data,
278 impulse->tflite_output_features_count,
279 debug);
280 break;
281 }
282 default: {
283 ei_printf(
284 "ERR: Unsupported object detection last layer (%d)\n",
285 block_config->object_detection_last_layer);
286 return EI_IMPULSE_UNSUPPORTED_INFERENCING_ENGINE;
287 }
288 }
289 }
290 else {
291 fill_res = fill_result_struct_f32(impulse, result, out_data, debug);
292 }
293
294 ei_free(out_data);
295
296 if (fill_res != EI_IMPULSE_OK) {
297 return fill_res;
298 }
299
300 return EI_IMPULSE_OK;
301}
