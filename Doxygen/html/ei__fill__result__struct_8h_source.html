80__attribute__((unused)) static bool ei_cube_check_overlap(ei_classifier_cube_t *c, int x, int y, int width, int height, float confidence) {
81 bool is_overlapping = !(c->x + c->width < x || c->y + c->height < y || c->x > x + width || c->y > y + height);
82 if (!is_overlapping) return false;
83
84 // if we overlap, but the x of the new box is lower than the x of the current box
85 if (x < c->x) {
86 // update x to match new box and make width larger (by the diff between the boxes)
87 c->x = x;
88 c->width += c->x - x;
89 }
90 // if we overlap, but the y of the new box is lower than the y of the current box
91 if (y < c->y) {
92 // update y to match new box and make height larger (by the diff between the boxes)
93 c->y = y;
94 c->height += c->y - y;
95 }
96 // if we overlap, and x+width of the new box is higher than the x+width of the current box
97 if (x + width > c->x + c->width) {
98 // just make the box wider
99 c->width += (x + width) - (c->x + c->width);
100 }
101 // if we overlap, and y+height of the new box is higher than the y+height of the current box
102 if (y + height > c->y + c->height) {
103 // just make the box higher
104 c->height += (y + height) - (c->y + c->height);
105 }
106 // if the new box has higher confidence, then override confidence of the whole box
107 if (confidence > c->confidence) {
108 c->confidence = confidence;
109 }
110 return true;
111}

113__attribute__((unused)) static void ei_handle_cube(std::vector<ei_classifier_cube_t*> *cubes, int x, int y, float vf, const char *label, float detection_threshold) {
114 if (vf < detection_threshold) return;
115
116 bool has_overlapping = false;
117 int width = 1;
118 int height = 1;
119
120 for (auto c : *cubes) {
121 // not cube for same class? continue
122 if (strcmp(c->label, label) != 0) continue;
123
124 if (ei_cube_check_overlap(c, x, y, width, height, vf)) {
125 has_overlapping = true;
126 break;
127 }
128 }
129
130 if (!has_overlapping) {
131 ei_classifier_cube_t *cube = new ei_classifier_cube_t();
132 cube->x = x;
133 cube->y = y;
134 cube->width = 1;
135 cube->height = 1;
136 cube->confidence = vf;
137 cube->label = label;
138 cubes->push_back(cube);
139 }
140}

142__attribute__((unused)) static void fill_result_struct_from_cubes(ei_impulse_result_t *result, std::vector<ei_classifier_cube_t*> *cubes, int out_width_factor, uint32_t object_detection_count) {
143 std::vector<ei_classifier_cube_t*> bbs;
144 static std::vector<ei_impulse_result_bounding_box_t> results;
145 int added_boxes_count = 0;
146 results.clear();
147
148 for (auto sc : *cubes) {
149 bool has_overlapping = false;
150
151 int x = sc->x;
152 int y = sc->y;
153 int width = sc->width;
154 int height = sc->height;
155 const char *label = sc->label;
156 float vf = sc->confidence;
157
158 for (auto c : bbs) {
159 // not cube for same class? continue
160 if (strcmp(c->label, label) != 0) continue;
161
162 if (ei_cube_check_overlap(c, x, y, width, height, vf)) {
163 has_overlapping = true;
164 break;
165 }
166 }
167
168 if (has_overlapping) {
169 continue;
170 }
171
172 bbs.push_back(sc);
173
174 ei_impulse_result_bounding_box_t tmp = {
175 .label = sc->label,
176 .x = (uint32_t)(sc->x * out_width_factor),
177 .y = (uint32_t)(sc->y * out_width_factor),
178 .width = (uint32_t)(sc->width * out_width_factor),
179 .height = (uint32_t)(sc->height * out_width_factor),
180 .value = sc->confidence
181 };
182
183 results.push_back(tmp);
184 added_boxes_count++;
185 }
186
187 // if we didn't detect min required objects, fill the rest with fixed value
188 if (added_boxes_count < object_detection_count) {
189 results.resize(object_detection_count);
190 for (size_t ix = added_boxes_count; ix < object_detection_count; ix++) {
191 results[ix].value = 0.0f;
192 }
193 }
194
195 for (auto c : *cubes) {
196 delete c;
197 }
198
199 result->bounding_boxes = results.data();
200 result->bounding_boxes_count = added_boxes_count;
201}

307 if (xend < xstart) xend = xstart;
308 if (yend < ystart) yend = ystart;

466__attribute__((unused)) static EI_IMPULSE_ERROR fill_result_struct_f32_yolov5(const ei_impulse_t *impulse,
467 const ei_learning_block_config_tflite_graph_t *block_config,
468 ei_impulse_result_t *result,
469 int version,
470 float *data,
471 size_t output_features_count,

475 results.clear();
476
477 size_t col_size = 5 + impulse->label_count;
478 size_t row_count = output_features_count / col_size;
479
480 for (size_t ix = 0; ix < row_count; ix++) {
481 size_t base_ix = ix * col_size;
482 float xc = data[base_ix + 0];
483 float yc = data[base_ix + 1];
484 float w = data[base_ix + 2];
485 float h = data[base_ix + 3];
486 float x = xc - (w / 2.0f);
487 float y = yc - (h / 2.0f);
488 if (x < 0) {
489 x = 0;

491 if (y < 0) {
492 y = 0;
493 }
494 if (x + w > impulse->input_width) {
495 w = impulse->input_width - x;

503 }

506
507 uint32_t label = 0;
508 for (size_t lx = 0; lx < impulse->label_count; lx++) {

510 if (l > 0.5f) {

513 }
514 }
515

537 if (nms_res != EI_IMPULSE_OK) {
538 return nms_res;
539 }
540
541 // if we didn't detect min required objects, fill the rest with fixed value
542 size_t added_boxes_count = results.size();
543 size_t min_object_detection_count = impulse->object_detection_count;
544 if (added_boxes_count < min_object_detection_count) {
545 results.resize(min_object_detection_count);
546 for (size_t ix = added_boxes_count; ix < min_object_detection_count; ix++) {
547 results[ix].value = 0.0f;
548 }
549 }
550
551 result->bounding_boxes = results.data();
552 result->bounding_boxes_count = added_boxes_count;
553
554 return EI_IMPULSE_OK;
555#else
556 return EI_IMPULSE_LAST_LAYER_NOT_AVAILABLE;
557#endif
558}

664__attribute__((unused)) static EI_IMPULSE_ERROR fill_result_struct_f32_yolox(const ei_impulse_t *impulse,
665 const ei_learning_block_config_tflite_graph_t *block_config,
666 ei_impulse_result_t *result,
667 float *data,
668 size_t output_features_count,
669 bool debug = false) {
670#ifdef EI_HAS_YOLOX
671 static std::vector<ei_impulse_result_bounding_box_t> results;
672 results.clear();
673
674 // START: def yolox_postprocess()
675
676 // if not p6:
677 // strides = [8, 16, 32]
678 // else:
679 // strides = [8, 16, 32, 64]
680 const std::vector<int> strides { 8, 16, 32 };
681
682 // hsizes = [img_size[0] // stride for stride in strides]
683 // wsizes = [img_size[1] // stride for stride in strides]
684 std::vector<int> hsizes(strides.size());
685 std::vector<int> wsizes(strides.size());
686 for (int ix = 0; ix < (int)strides.size(); ix++) {
687 hsizes[ix] = (int)floor((float)impulse->input_width / (float)strides[ix]);
688 wsizes[ix] = (int)floor((float)impulse->input_height / (float)strides[ix]);
689 }
690
691 // for hsize, wsize, stride in zip(hsizes, wsizes, strides):
692 // grid = np.stack((xv, yv), 2).reshape(1, -1, 2)
693 // grids.append(grid)
694 // shape = grid.shape[:2]
695 // expanded_strides.append(np.full((*shape, 1), stride))
696 std::vector<matrix_i32_t*> grids;
697 std::vector<matrix_i32_t*> expanded_strides;
698
699 for (int ix = 0; ix < (int)strides.size(); ix++) {
700 int hsize = hsizes.at(ix);
701 int wsize = wsizes.at(ix);
702 int stride = strides.at(ix);
703
704 // xv, yv = np.meshgrid(np.arange(wsize), np.arange(hsize))
705 // grid = np.stack((xv, yv), 2).reshape(1, -1, 2)
706 matrix_i32_t *grid = new matrix_i32_t(hsize * wsize, 2);
707 int grid_ix = 0;
708 for (int h = 0; h < hsize; h++) {
709 for (int w = 0; w < wsize; w++) {
710 grid->buffer[grid_ix + 0] = w;
711 grid->buffer[grid_ix + 1] = h;
712 grid_ix += 2;
713 }
714 }
715 grids.push_back(grid);
716
717 // shape = grid.shape[:2]
718 // expanded_strides.append(np.full((*shape, 1), stride))
719 matrix_i32_t *expanded_stride = new matrix_i32_t(hsize * wsize, 1);
720 for (int ix = 0; ix < hsize * wsize; ix++) {
721 expanded_stride->buffer[ix] = stride;
722 }
723 expanded_strides.push_back(expanded_stride);
724 }
725
726 // grids = np.concatenate(grids, 1)
727 int total_grid_rows = 0;
728 for (auto g : grids) {
729 total_grid_rows += g->rows;
730 }
731 matrix_i32_t c_grid(total_grid_rows, 2);
732 int c_grid_ix = 0;
733 for (auto g : grids) {
734 for (int row = 0; row < (int)g->rows; row++) {
735 c_grid.buffer[c_grid_ix + 0] = g->buffer[(row * 2) + 0];
736 c_grid.buffer[c_grid_ix + 1] = g->buffer[(row * 2) + 1];
737 c_grid_ix += 2;
738 }
739 delete g;
740 }
741
742 // expanded_strides = np.concatenate(expanded_strides, 1)
743 int total_stride_rows = 0;
744 for (auto g : expanded_strides) {
745 total_stride_rows += g->rows;
746 }
747 matrix_i32_t c_expanded_strides(total_stride_rows, 1);
748 int c_expanded_strides_ix = 0;
749 for (auto g : expanded_strides) {
750 for (int row = 0; row < (int)g->rows; row++) {
751 c_expanded_strides.buffer[c_expanded_strides_ix + 0] = g->buffer[(row * 1) + 0];
752 c_expanded_strides_ix += 1;
753 }
754 delete g;
755 }
756
757 const int output_rows = output_features_count / (5 + impulse->label_count);
758 matrix_t outputs(output_rows, 5 + impulse->label_count, data);
759 for (int row = 0; row < (int)outputs.rows; row++) {
760 float v0 = outputs.buffer[(row * outputs.cols) + 0];
761 float v1 = outputs.buffer[(row * outputs.cols) + 1];
762 float v2 = outputs.buffer[(row * outputs.cols) + 2];
763 float v3 = outputs.buffer[(row * outputs.cols) + 3];
764
765 float cgrid0 = (float)c_grid.buffer[(row * c_grid.cols) + 0];
766 float cgrid1 = (float)c_grid.buffer[(row * c_grid.cols) + 1];
767
768 float stride = (float)c_expanded_strides.buffer[row];
769
770 // outputs[..., :2] = (outputs[..., :2] + grids) * expanded_strides
771 outputs.buffer[(row * outputs.cols) + 0] = (v0 + cgrid0) * stride;
772 outputs.buffer[(row * outputs.cols) + 1] = (v1 + cgrid1) * stride;
773
774 // outputs[..., 2:4] = np.exp(outputs[..., 2:4]) * expanded_strides
775 outputs.buffer[(row * outputs.cols) + 2] = exp(v2) * stride;
776 outputs.buffer[(row * outputs.cols) + 3] = exp(v3) * stride;
777 }
778
779 // END: def yolox_postprocess()
780
781 // boxes = predictions[:, :4]
782 matrix_t boxes(outputs.rows, 4);
783 for (int row = 0; row < (int)outputs.rows; row++) {
784 boxes.buffer[(row * boxes.cols) + 0] = outputs.buffer[(row * outputs.cols) + 0];
785 boxes.buffer[(row * boxes.cols) + 1] = outputs.buffer[(row * outputs.cols) + 1];
786 boxes.buffer[(row * boxes.cols) + 2] = outputs.buffer[(row * outputs.cols) + 2];
787 boxes.buffer[(row * boxes.cols) + 3] = outputs.buffer[(row * outputs.cols) + 3];
788 }
789
790 // scores = predictions[:, 4:5] * predictions[:, 5:]
791 matrix_t scores(outputs.rows, impulse->label_count);
792 for (int row = 0; row < (int)outputs.rows; row++) {
793 float confidence = outputs.buffer[(row * outputs.cols) + 4];
794 for (int cc = 0; cc < impulse->label_count; cc++) {
795 scores.buffer[(row * scores.cols) + cc] = confidence * outputs.buffer[(row * outputs.cols) + (5 + cc)];
796 }
797 }
798
799 // iterate through scores to see if we have anything with confidence
800 for (int row = 0; row < (int)scores.rows; row++) {
801 for (int col = 0; col < (int)scores.cols; col++) {
802 float confidence = scores.buffer[(row * scores.cols) + col];
803
804 if (confidence >= block_config->threshold && confidence <= 1.0f) {
805 ei_impulse_result_bounding_box_t r;
806 r.label = impulse->categories[col];
807 r.value = confidence;
808
809 // now find the box...
810 float xcenter = boxes.buffer[(row * boxes.cols) + 0];
811 float ycenter = boxes.buffer[(row * boxes.cols) + 1];
812 float width = boxes.buffer[(row * boxes.cols) + 2];
813 float height = boxes.buffer[(row * boxes.cols) + 3];
814
815 int x = (int)(xcenter - (width / 2.0f));
816 int y = (int)(ycenter - (height / 2.0f));
817
818 if (x < 0) {
819 x = 0;
820 }
821 if (x > (int)impulse->input_width) {
822 x = impulse->input_width;
823 }
824 if (y < 0) {
825 y = 0;
826 }
827 if (y > (int)impulse->input_height) {
828 y = impulse->input_height;
829 }
830
831 r.x = x;
832 r.y = y;
833 r.width = (int)round(width);
834 r.height = (int)round(height);
835
836 results.push_back(r);
837 }
838 }
839 }
840
841 EI_IMPULSE_ERROR nms_res = ei_run_nms(impulse, &results, debug);
842 if (nms_res != EI_IMPULSE_OK) {
843 return nms_res;
844 }
845
846 // if we didn't detect min required objects, fill the rest with fixed value
847 size_t added_boxes_count = results.size();
848 size_t min_object_detection_count = impulse->object_detection_count;
849 if (added_boxes_count < min_object_detection_count) {
850 results.resize(min_object_detection_count);
851 for (size_t ix = added_boxes_count; ix < min_object_detection_count; ix++) {
852 results[ix].value = 0.0f;
853 }
854 }
855
856 result->bounding_boxes = results.data();
857 result->bounding_boxes_count = added_boxes_count;
858
859 return EI_IMPULSE_OK;
860#else
861 return EI_IMPULSE_LAST_LAYER_NOT_AVAILABLE;
862#endif // EI_HAS_YOLOX

1036 ei_impulse_result_t *result,
1037 T *data,
1038 float zero_point,
1039 float scale,
1040 size_t output_features_count,
1041 float threshold,
1042 bool debug = false) {
1043
1044 size_t col_size = 12 + impulse->label_count + 1;
1045 size_t row_count = output_features_count / col_size;
1046
1047 static std::vector<ei_impulse_result_bounding_box_t> results;
1048 static std::vector<ei_impulse_result_bounding_box_t> class_results;
1049 results.clear();
1050
1051 for (size_t cls_idx = 1; cls_idx < (size_t)(impulse->label_count + 1); cls_idx++) {
1052
1053 std::vector<float> boxes;
1054 std::vector<float> scores;
1055 std::vector<int> classes;
1056 class_results.clear();
1057
1058 for (size_t ix = 0; ix < row_count; ix++) {
1059
1060 float score = (static_cast<float>(data[ix * col_size + cls_idx]) - zero_point) * scale;
1061
1062 if ((score < threshold) || (score > 1.0f)) {
1063 continue;
1064 }
1065
1066 // # 1. calculate boxes location
1067 size_t base_ix = ix * col_size + col_size; // references the end of the row
1068
1069 float r_12 = (static_cast<float>(data[base_ix - 12]) - zero_point) * scale;
1070 float r_11 = (static_cast<float>(data[base_ix - 11]) - zero_point) * scale;
1071 float r_10 = (static_cast<float>(data[base_ix - 10]) - zero_point) * scale;
1072 float r_9 = (static_cast<float>(data[base_ix - 9]) - zero_point) * scale;
1073 float r_8 = (static_cast<float>(data[base_ix - 8]) - zero_point) * scale;
1074 float r_7 = (static_cast<float>(data[base_ix - 7]) - zero_point) * scale;
1075 float r_6 = (static_cast<float>(data[base_ix - 6]) - zero_point) * scale;
1076 float r_5 = (static_cast<float>(data[base_ix - 5]) - zero_point) * scale;
1077 float r_4 = (static_cast<float>(data[base_ix - 4]) - zero_point) * scale;
1078 float r_3 = (static_cast<float>(data[base_ix - 3]) - zero_point) * scale;
1079 float r_2 = (static_cast<float>(data[base_ix - 2]) - zero_point) * scale;
1080 float r_1 = (static_cast<float>(data[base_ix - 1]) - zero_point) * scale;
1081
1082 // cx_pred = y_pred[..., -12]
1083 // cy_pred = y_pred[..., -11]
1084 // w_pred = y_pred[..., -10]

1087 float cy_pred = r_11;

1089 float h_pred = r_9;

1094 float h_anchor = r_5 - r_7;

1097 // cy_anchor = tf.truediv(y_pred[..., -5] + y_pred[..., -7], 2.0)

1099 float cy_anchor = (r_5 + r_7) / 2.0f;
1100
1101 // cx_variance = y_pred[..., -4]
1102 // cy_variance = y_pred[..., -3]
1103 float cx_variance = r_4;
1104 float cy_variance = r_3;
1105
1106 // variance_w = y_pred[..., -2]
1107 // variance_h = y_pred[..., -1]
1108 float variance_w = r_2;
1109 float variance_h = r_1;
1110
1111 // # Convert anchor box offsets to image offsets.
1112 // cx = cx_pred * cx_variance * w_anchor + cx_anchor
1113 // cy = cy_pred * cy_variance * h_anchor + cy_anchor
1114 // w = tf.exp(w_pred * variance_w) * w_anchor
1115 // h = tf.exp(h_pred * variance_h) * h_anchor
1116 float cx = cx_pred * cx_variance * w_anchor + cx_anchor;
1117 float cy = cy_pred * cy_variance * h_anchor + cy_anchor;
1118 float w = exp(w_pred * variance_w) * w_anchor;
1119 float h = exp(h_pred * variance_h) * h_anchor;
1120
1121 // # Convert 'centroids' to 'corners'.
1122 float xmin = cx - (w / 2.0f);
1123 float ymin = cy - (h / 2.0f);
1124 float xmax = cx + (w / 2.0f);
1125 float ymax = cy + (h / 2.0f);
1126
1127 xmin *= impulse->input_width;
1128 ymin *= impulse->input_height;
1129 xmax *= impulse->input_width;
1130 ymax *= impulse->input_height;
1131
1132 boxes.push_back(ymin);
1133 boxes.push_back(xmin);
1134 boxes.push_back(ymax);
1135 boxes.push_back(xmax);
1136 scores.push_back(score);
1137 classes.push_back((int)(cls_idx-1));
1138 }
1139
1140 size_t nr_boxes = scores.size();
1141 EI_IMPULSE_ERROR nms_res = ei_run_nms(impulse, &class_results,
1142 boxes.data(), scores.data(), classes.data(),
1143 nr_boxes,
1144 true /*clip_boxes*/,
1145 debug);
1146
1147 if (nms_res != EI_IMPULSE_OK) {
1148 return nms_res;
1149 }
1150
1151 for (auto bb: class_results) {
1152 results.push_back(bb);
1153 }
1154 }
1155
1156 prepare_tao_results_common(impulse, result, &results);
1157
1158 return EI_IMPULSE_OK;
1159}

1471__attribute__((unused)) static void softmax(std::vector<float>& input, const size_t nb_classes)
1472{
1473 const float max = *std::max_element(input.begin(), input.end());
1474 const float min = *std::min_element(input.begin(), input.end());
1475 const float t = -100.0f;
1476
1477 // x = x - np.max(x)
1478 std::transform(input.begin(), input.end(), input.begin(),
1479 [max](float x) { return x - max; });
1480
1481 // if np.min(x) < t: x = x / np.min(x) * t
1482 std::transform(input.begin(), input.end(), input.begin(),
1483 [min, t](float x) { return x < t ? (x / min * t): x; });
1484
1485 // e_x = np.exp(x)
1486 // do it in place as we don't need raw the input anymore
1487 std::transform(input.begin(), input.end(), input.begin(),
1488 [](float x) { return std::exp(x); });
1489
1490 // e_x / e_x.sum(axis, keepdims=True)
1491 // calculated for each 'row', across nb_classes
1492 for(auto it = input.begin(); it != input.end(); it += nb_classes) {
1493 float sum = 0.0f;
1494 // e_x.sum(axis, keepdims=True)
1495 for(auto it2 = it; it2 != it + nb_classes; it2++) {
1496 sum += *it2;
1497 }
1498 // e_x / e_x.sum(axis, keepdims=True)
1499 std::transform(it, it + nb_classes, it,
1500 [sum](float ex) { return ex / sum; });
1501 }

1522 if(x3 < x1) {
1523 if(x4 < x1) {
1524 return 0;
1525 }
1526 return std::min(x2, x4) - x1;

1529 return 0;
1530 }

1532 }

1536 // Implementation of the Intersection over Union calculation
1537 float intersect_w = this->_interval_overlap(this->x1, this->x2, other.x1, other.x2);
1538 float intersect_h = this->_interval_overlap(this->y1, this->y2, other.y1, other.y2);
1539
1540 float intersect = intersect_w * intersect_h;
1541
1542 float w1 = this->x2 - this->x1;
1543 float h1 = this->y2 - this->y1;
1544 float w2 = other.x2 - other.x1;
1545 float h2 = other.y2 - other.y1;
1546
1547 float un = w1 * h1 + w2 * h2 - intersect;
1548
1549 return float(intersect) / un;
1550 }
