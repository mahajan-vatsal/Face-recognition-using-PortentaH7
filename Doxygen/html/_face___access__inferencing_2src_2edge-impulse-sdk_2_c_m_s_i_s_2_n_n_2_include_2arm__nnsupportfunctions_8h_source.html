931__STATIC_FORCEINLINE q31_t arm_nn_doubling_high_mult(const q31_t m1, const q31_t m2)
932{
933 q31_t result = 0;
934 // Rounding offset to add for a right shift of 31
935 q63_t mult = 1 << 30;
936
937 if ((m1 < 0) ^ (m2 < 0))
938 {
939 mult = 1 - mult;
940 }
941 // Gets resolved as a SMLAL instruction
942 mult = mult + (q63_t)m1 * m2;
943
944 // Utilize all of the upper 32 bits. This is the doubling step
945 // as well.
946 result = (int32_t)(mult / (1ll << 31));
947
948 if ((m1 == m2) && (m1 == (int32_t)NN_Q31_MIN))
949 {
950 result = NN_Q31_MAX;
951 }
952 return result;
953}

969__STATIC_FORCEINLINE q31_t arm_nn_doubling_high_mult_no_sat(const q31_t m1, const q31_t m2)
970{
971 q31_t result = 0;
972 union arm_nn_long_long mult;
973
974 // Rounding offset to add for a right shift of 31
975 mult.word.low = 1 << 30;
976 mult.word.high = 0;
977
978 // Gets resolved as a SMLAL instruction
979 mult.long_long = mult.long_long + (q63_t)m1 * m2;
980
981 // Utilize all of the upper 32 bits. This is the doubling step
982 // as well.
983 result = (int32_t)(mult.long_long >> 31);
984
985 return result;
986}

996__STATIC_FORCEINLINE q31_t arm_nn_divide_by_power_of_two(const q31_t dividend, const q31_t exponent)
997{
998 q31_t result = 0;
999 const q31_t remainder_mask = (1 << exponent) - 1;
1000 int32_t remainder = remainder_mask & dividend;
1001
1002 // Basic division
1003 result = dividend >> exponent;
1004
1005 // Adjust 'result' for rounding (mid point away from zero)
1006 q31_t threshold = remainder_mask >> 1;
1007 if (result < 0)
1008 {
1009 threshold++;
1010 }
1011 if (remainder > threshold)
1012 {
1013 result++;
1014 }
1015
1016 return result;
1017}

1028__STATIC_FORCEINLINE q31_t arm_nn_requantize(const q31_t val, const q31_t multiplier, const q31_t shift)
1029{
1030#ifdef CMSIS_NN_USE_SINGLE_ROUNDING
1031 const int64_t total_shift = 31 - shift;
1032 const int64_t new_val = val * (int64_t)multiplier;
1033
1034 int32_t result = new_val >> (total_shift - 1);
1035 result = (result + 1) >> 1;
1036
1037 return result;
1038#else
1039 return arm_nn_divide_by_power_of_two(arm_nn_doubling_high_mult_no_sat(val * (1 << LEFT_SHIFT(shift)), multiplier),
1040 RIGHT_SHIFT(shift));
1041#endif
1042}
