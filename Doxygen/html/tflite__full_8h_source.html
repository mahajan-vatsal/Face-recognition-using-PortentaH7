Machine Vision using Portenta H7 2
This project aims to develop a face recognition-based access control system using the Arduino Portenta H7 and Vision Shield, leveraging Edge Impulse for machine learning. The system captures facial images, processes them locally using an AI model deployed on the Portenta H7 and determines access based on authorised personnel.

47static EI_IMPULSE_ERROR get_interpreter(ei_learning_block_config_tflite_graph_t *block_config, tflite::Interpreter **interpreter) {
48 // not in the map yet...
49 if (!ei_tflite_instances.count(block_config->block_id)) {
50 ei_config_tflite_graph_t *graph_config = (ei_config_tflite_graph_t*)block_config->graph_config;
51 ei_tflite_state_t *new_state = new ei_tflite_state_t();
52
53 auto new_model = tflite::FlatBufferModel::BuildFromBuffer((const char*)graph_config->model, graph_config->model_size);
54 new_state->model = std::move(new_model);
55 if (!new_state->model) {
56 ei_printf("Failed to build TFLite model from buffer\n");
57 return EI_IMPULSE_TFLITE_ERROR;
58 }
59
60 tflite::ops::builtin::BuiltinOpResolver resolver;
61#if EI_CLASSIFIER_HAS_TREE_ENSEMBLE_CLASSIFIER
62 resolver.AddCustom("TreeEnsembleClassifier",
63 tflite::ops::custom::Register_TREE_ENSEMBLE_CLASSIFIER());
64#endif
65 tflite::InterpreterBuilder builder(*new_state->model, resolver);
66 builder(&new_state->interpreter);
67
68 if (!new_state->interpreter) {
69 ei_printf("Failed to construct interpreter\n");
70 return EI_IMPULSE_TFLITE_ERROR;
71 }
72
73 if (new_state->interpreter->AllocateTensors() != kTfLiteOk) {
74 ei_printf("AllocateTensors failed\n");
75 return EI_IMPULSE_TFLITE_ERROR;
76 }
77
78 int hw_thread_count = (int)std::thread::hardware_concurrency();
79 hw_thread_count -= 1; // leave one thread free for the other application
80 if (hw_thread_count < 1) {
81 hw_thread_count = 1;
82 }
83
84 if (new_state->interpreter->SetNumThreads(hw_thread_count) != kTfLiteOk) {
85 ei_printf("SetNumThreads failed\n");
86 return EI_IMPULSE_TFLITE_ERROR;
87 }
88
89 ei_tflite_instances.insert(std::make_pair(block_config->block_id, new_state));
90 }
91
92 auto tflite_state = ei_tflite_instances[block_config->block_id];
93 *interpreter = tflite_state->interpreter.get();
94 return EI_IMPULSE_OK;
95}

97extern "C" EI_IMPULSE_ERROR run_nn_inference_from_dsp(
98 ei_learning_block_config_tflite_graph_t *block_config,
99 signal_t *signal,
100 matrix_t *output_matrix)
101{
102 tflite::Interpreter *interpreter;
103 auto interpreter_ret = get_interpreter(block_config, &interpreter);
104 if (interpreter_ret != EI_IMPULSE_OK) {
105 return interpreter_ret;
106 }
107
108 TfLiteTensor *input = interpreter->input_tensor(0);
109 TfLiteTensor *output = interpreter->output_tensor(0);
110
111 if (!input) {
112 return EI_IMPULSE_INPUT_TENSOR_WAS_NULL;
113 }
114 if (!output) {
115 return EI_IMPULSE_OUTPUT_TENSOR_WAS_NULL;
116 }
117
118 auto input_res = fill_input_tensor_from_signal(signal, input);
119 if (input_res != EI_IMPULSE_OK) {
120 return input_res;
121 }
122
123 TfLiteStatus status = interpreter->Invoke();
124 if (status != kTfLiteOk) {
125 ei_printf("ERR: interpreter->Invoke() failed with %d\n", status);
126 return EI_IMPULSE_TFLITE_ERROR;
127 }
128
129 auto output_res = fill_output_matrix_from_tensor(output, output_matrix);
130 if (output_res != EI_IMPULSE_OK) {
131 return output_res;
132 }
133
134 // on Linux we're not worried about free'ing (for now)
135
136 return EI_IMPULSE_OK;
137}

139EI_IMPULSE_ERROR run_nn_inference(
140 const ei_impulse_t *impulse,
141 ei_feature_t *fmatrix,
142 uint32_t learn_block_index,
143 uint32_t* input_block_ids,
144 uint32_t input_block_ids_size,
145 ei_impulse_result_t *result,
146 void *config_ptr,
147 bool debug = false)
148{
149 ei_learning_block_config_tflite_graph_t *block_config = (ei_learning_block_config_tflite_graph_t*)config_ptr;
150
151 tflite::Interpreter *interpreter;
152 auto interpreter_ret = get_interpreter(block_config, &interpreter);
153 if (interpreter_ret != EI_IMPULSE_OK) {
154 return interpreter_ret;
155 }
156
157 TfLiteTensor *input = interpreter->input_tensor(0);
158 TfLiteTensor *output = interpreter->output_tensor(block_config->output_data_tensor);
159
160 if (!input) {
161 return EI_IMPULSE_INPUT_TENSOR_WAS_NULL;
162 }
163 if (!output) {
164 return EI_IMPULSE_OUTPUT_TENSOR_WAS_NULL;
165 }
166
167 size_t mtx_size = impulse->dsp_blocks_size + impulse->learning_blocks_size;
168 auto input_res = fill_input_tensor_from_matrix(fmatrix, input, input_block_ids, input_block_ids_size, mtx_size);
169 if (input_res != EI_IMPULSE_OK) {
170 return input_res;
171 }
172
173 uint64_t ctx_start_us = ei_read_timer_us();
174
175 TfLiteStatus status = interpreter->Invoke();
176 if (status != kTfLiteOk) {
177 ei_printf("ERR: interpreter->Invoke() failed with %d\n", status);
178 return EI_IMPULSE_TFLITE_ERROR;
179 }
180
181 uint64_t ctx_end_us = ei_read_timer_us();
182
183 result->timing.classification_us = ctx_end_us - ctx_start_us;
184 result->timing.classification = (int)(result->timing.classification_us / 1000);
185
186 if (result->copy_output) {
187 auto output_res = fill_output_matrix_from_tensor(output, fmatrix[impulse->dsp_blocks_size + learn_block_index].matrix);
188 if (output_res != EI_IMPULSE_OK) {
189 return output_res;
190 }
191 }
192
193 if (debug) {
194 ei_printf("Predictions (time: %d ms.):\n", result->timing.classification);
195 }
196
197 TfLiteTensor *scores_tensor = interpreter->output_tensor(block_config->output_score_tensor);
198 TfLiteTensor *labels_tensor = interpreter->output_tensor(block_config->output_labels_tensor);
199
200 EI_IMPULSE_ERROR fill_res = fill_result_struct_from_output_tensor_tflite(
201 impulse, block_config, output, labels_tensor, scores_tensor, result, debug);
202
203 if (fill_res != EI_IMPULSE_OK) {
204 return fill_res;
205 }
206
207 // on Linux we're not worried about free'ing (for now)
208
209 return EI_IMPULSE_OK;
210}
