Machine Vision using Portenta H7 2
This project aims to develop a face recognition-based access control system using the Arduino Portenta H7 and Vision Shield, leveraging Edge Impulse for machine learning. The system captures facial images, processes them locally using an AI model deployed on the Portenta H7 and determines access based on authorised personnel.

42__NN_FORCE_INLINE__ int32_t esp_nn_clz32(uint32_t in)
43{
44#if CONFIG_IDF_TARGET_ARCH_XTENSA
45 __asm__ volatile("nsau %0, %0" : "+r" (in));
46 return in;
47#elif defined(__GNUC__)
48 return __builtin_clz(in);
49#else
50 int32_t count = 32;
51 uint32_t x = in, y = in >> 16;
52 if (y != 0) {
53 count -= 16;
54 x = y;
55 }
56 y = x >> 8;
57 if (y != 0) {
58 count -= 8;
59 x = y;
60 }
61 y = x >> 4;
62 if (y != 0) {
63 count -= 4;
64 x = y;
65 }
66 y = x >> 2;
67 if (y != 0) {
68 count -= 2;
69 x = y;
70 }
71 y = x >> 1;
72 if (y != 0) {
73 return count - 2;
74 }
75 return count - x;
76#endif
77}

92__NN_FORCE_INLINE__ int32_t esp_nn_pick_sat_high32_of64(int64_t val64)
93{
94 int32_t sign = (int32_t) (val64 >> 63);
95 int32_t to_add = sign & ((1ul << 31) - 1);
96 return (int32_t) ((int64_t) (val64 + to_add) >> 31);
97}

99__NN_FORCE_INLINE__ int32_t esp_nn_sat_round_doubling_high_mul(int32_t in0, int32_t in1)
100{
101 int32_t result;
102 int64_t in0_64 = (int64_t) in0;
103 bool overflow = (in0 == in1) && (in0 == (int32_t) INT32_MIN);
104
105 /* Nudge value */
106 int64_t nudge_val = 1 << 30;
107 if ((in0 < 0) ^ (in1 < 0)) {
108 nudge_val = 1 - nudge_val;
109 }
110
111 /* Multiply and add nudge */
112 int64_t mult = in0_64 * in1 + nudge_val;
113
114 /* Round and pickup 32 bits */
115 result = esp_nn_pick_sat_high32_of64(mult);
116
117 return overflow ? INT32_MAX : result;
118}

126__NN_FORCE_INLINE__ int32_t esp_nn_div_by_power_of_two_fast(int32_t val, int32_t exponent)
127{
128 int32_t to_add = (1 << (exponent - 1)) - (val < 0);
129 return (int32_t) ((val + to_add) >> exponent);
130}

132__NN_FORCE_INLINE__ int32_t esp_nn_div_by_power_of_two(int32_t val, int32_t exponent)
133{
134 int32_t result;
135
136 const int32_t mask = (1 << exponent) - 1;
137 const int32_t remainder = val & mask;
138
139 result = val >> exponent;
140 int32_t threshold = (mask >> 1) + (result < 0);
141
142 if (remainder > threshold) {
143 result += 1;
144 }
145 return result;
146}

148__NN_FORCE_INLINE__ int32_t esp_nn_multiply_by_quantized_mult(int32_t x, int32_t mult, int32_t shift)
149{
150 int32_t left_shift = shift > 0 ? shift : 0;
151 int32_t right_shift = shift > 0 ? 0 : -shift;
152 int32_t result = esp_nn_sat_round_doubling_high_mul(x * (1 << left_shift), mult);
153 return esp_nn_div_by_power_of_two(result, right_shift);
154}

156__NN_FORCE_INLINE__ int32_t esp_nn_multiply_by_quantized_mult_fast(int32_t x, int32_t mult, int32_t shift)
157{
158 int32_t left_shift = max(shift, 0);
159 int32_t right_shift = left_shift - shift;
160
161 int64_t nudge_val = 1 << 30;
162 int64_t in0_64 = (int64_t) (x << left_shift);
163
164 /* Multiply and add nudge */
165 int64_t mult_64 = in0_64 * mult + nudge_val;
166 int32_t result = (int32_t) (mult_64 >> 31);
167 if (right_shift) {
168 result = esp_nn_div_by_power_of_two_fast(result, right_shift);
169 }
170 return result;
171}

173static void esp_nn_aligned_s8_pad_with_value(const int8_t *src, int8_t *dst,
174 const uint16_t input_wd,
175 const uint16_t input_ht,
176 const uint16_t channels,
177 const int32_t pad_val,
178 const uint16_t pad_wd,
179 const uint16_t pad_ht)
180{
181 /* memset with pad_val */
182 memset(dst, pad_val, ((input_wd + 2 * pad_wd) * (input_ht + 2 * pad_ht)) * channels);
183 dst += (pad_wd + input_wd + pad_wd) * pad_ht * channels;
184
185 for (int i = 0; i < input_ht; i++) {
186 dst += pad_wd * channels;
187 for (int j = 0; j < input_wd * channels; j++) {
188 *dst++ = *src++;
189 }
190 dst += pad_wd * channels;
191 }
192}

194static void esp_nn_aligned_s8_pad_end_with_value(const int8_t *src, int8_t *dst,
195 const uint16_t input_wd,
196 const uint16_t input_ht,
197 const uint16_t channels,
198 const int32_t pad_val,
199 const uint16_t pad_wd,
200 const uint16_t pad_ht)
201{
202 for (int i = 0; i < input_ht; i++) {
203 for (int j = 0; j < input_wd * channels; j++) {
204 *dst++ = *src++;
205 }
206 if (pad_wd) {
207 memset(dst, pad_val, pad_wd * channels);
208 dst += pad_wd * channels;
209 }
210 }
211 /* pad end `pad_ht` lines at end */
212 if (pad_ht) {
213 memset(dst, pad_val, (input_wd + pad_wd) * pad_ht * channels);
