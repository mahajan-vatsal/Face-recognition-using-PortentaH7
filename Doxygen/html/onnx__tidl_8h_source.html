The system captures facial images, processes them locally using an AI model deployed on the Portenta H7 and determines access based on authorised personnel.

73int printTensorInfo(Ort::Session *session, std::vector<const char *> *input_node_names, std::vector<const char *> *output_node_names)
74{
75 size_t num_input_nodes = (*session).GetInputCount();
76 size_t num_output_nodes = (*session).GetOutputCount();
77 Ort::TypeInfo type_info = (*session).GetInputTypeInfo(0);
78 auto tensor_info = type_info.GetTensorTypeAndShapeInfo();
79 std::vector<int64_t> input_node_dims = tensor_info.GetShape();
80 ei_printf("LOG_INFO: number of inputs: %d\n", (int)num_input_nodes);
81 ei_printf("LOG_INFO: number of outputs: %d\n", (int)num_output_nodes);
82 ei_printf("LOG_INFO: input(0) name: %s\n", (*input_node_names)[0]);
83
84 Ort::TypeInfo type_info_out = (*session).GetOutputTypeInfo(0);
85 auto tensor_info_out = type_info_out.GetTensorTypeAndShapeInfo();
86 std::vector<int64_t> output_node_dims = tensor_info_out.GetShape();
87 /* iterate over all input nodes */
88 for (int i = 0; i < (int)num_input_nodes; i++)
89 {
90 /* print input node names */
91 ei_printf("LOG_INFO: Input %d : name=%s\n", i, (*input_node_names)[i]);
92
93 /* print input node types */
94 Ort::TypeInfo type_info = (*session).GetInputTypeInfo(i);
95 auto tensor_info = type_info.GetTensorTypeAndShapeInfo();
96
97 ONNXTensorElementDataType type = tensor_info.GetElementType();
98 ei_printf("LOG_INFO: Input %d : type=%d\n", i, type);
99 /* print input shapes/dims */
100 input_node_dims = tensor_info.GetShape();
101 ei_printf("LOG_INFO: Input %d : num_dims=%zu\n", i, input_node_dims.size());
102 for (int j = 0; j < input_node_dims.size(); j++)
103 {
104 ei_printf("LOG_INFO: Input %d : dim %d=%jd\n", i, j, input_node_dims[j]);
105 }
106 }
107 if (num_input_nodes != 1)
108 {
109 ei_printf("LOG_INFO: supports only single input model \n");
110 return EI_IMPULSE_ONNX_ERROR;
111 }
112
113 for (int i = 0; i < (int)num_output_nodes; i++)
114 {
115 /* print output node names */
116 ei_printf("LOG_INFO: Output %d : name=%s\n", i, (*output_node_names)[i]);
117
118 /* print output node types */
119 Ort::TypeInfo type_info = (*session).GetOutputTypeInfo(i);
120 auto tensor_info = type_info.GetTensorTypeAndShapeInfo();
121
122 ONNXTensorElementDataType type = tensor_info.GetElementType();
123 ei_printf("LOG_INFO: Output %d : type=%d\n", i, type);
124 /* print output shapes/dims */
125 output_node_dims = tensor_info.GetShape();
126 ei_printf("LOG_INFO: Output %d : num_dims=%zu\n", i, output_node_dims.size());
127 for (int j = 0; j < (int)output_node_dims.size(); j++)
128 {
129 ei_printf("LOG_INFO: Output %d : dim %d=%jd\n", i, j, output_node_dims[j]);
130 }
131 }
132 return EI_IMPULSE_OK;
133}

135void * allocTensorMem(int size, int accel)
136{
137 void * ptr = NULL;
138 if (accel)
139 {
140 #ifdef DEVICE_AM62
141 ei_printf("LOG_INFO: TIDL Delgate mode is not allowed on AM62 devices...\n");
142 ei_printf("LOG_ERROR: Could not allocate memory for a Tensor of size %d \n ", size);
143 exit(0);
144 #else
145 ptr = TIDLRT_allocSharedMem(64, size);
146 #endif
147 }
148 else
149 {
150 ptr = malloc(size);
151 }
152 if (ptr == NULL)
153 {
154 ei_printf("LOG_ERROR: Could not allocate memory for a Tensor of size %d \n ", size);
155 exit(0);
156 }
157 return ptr;
158}

161{
162 if (accel)
163 {
164 #ifndef DEVICE_AM62
165 TIDLRT_freeSharedMem(ptr);
166 #endif
167 }
168 else
169 {
170 free(ptr);
171 }
172}

185static EI_IMPULSE_ERROR inference_onnx_setup(
186 const ei_impulse_t *impulse,
187 uint64_t *ctx_start_us,
188 std::vector<Ort::Value>* input_tensors,
189 std::vector<Ort::Value>* output_tensors,
190 Ort::Session** session_ptr,
191 Ort::RunOptions** run_options_ptr,
192 Ort::IoBinding** binding_ptr) {
193
194 static bool onnx_first_run = true;
195 // Nothing to do after first run
196 if (!onnx_first_run) {
197 return EI_IMPULSE_OK;
198 }
199
200 std::string proj_artifacts_path = "/tmp/" + std::string(impulse->project_name) + "-" + std::to_string(impulse->project_id) + "-" + std::to_string(impulse->deploy_version);
201
202 create_project_if_not_exists(proj_artifacts_path, model_h_files, model_h_files_len);
203
204 std::string proj_model_path = proj_artifacts_path + "/model.onnx";
205
206 ei_printf("test onnx tidl: %s\n", __FUNCTION__);
207 #pragma message ( "test onnx tidl: run_nn_inference")
208
209 /* Initialize enviroment, maintains thread pools and state info */
210 Ort::Env env(ORT_LOGGING_LEVEL_WARNING, "test");
211 /* Initialize session options */
212 Ort::SessionOptions session_options;
213 //TODO: from where do we load number of threads?
214 session_options.SetIntraOpNumThreads(1);
215
216 ei_printf("LOG_INFO: model accelerated \n");
217 c_api_tidl_options *options = (c_api_tidl_options *)malloc(sizeof(c_api_tidl_options));
218 OrtStatus *def_status = OrtSessionsOptionsSetDefault_Tidl(options);
219 ei_printf("LOG_INFO: artifacts: %s \n", proj_artifacts_path.c_str());
220 strcpy(options->artifacts_folder, proj_artifacts_path.c_str());
221 if(NULL == options){
222 ei_printf("LOG_ERROR: faild to allocate c_api_tidl_options \n");
223 return EI_IMPULSE_ONNX_ERROR;
224 }
225 OrtStatus *status = OrtSessionOptionsAppendExecutionProvider_Tidl(session_options, options);
226
227 session_options.SetGraphOptimizationLevel(GraphOptimizationLevel::ORT_ENABLE_EXTENDED);
228 Ort::AllocatorWithDefaultOptions allocator;
229
230 /* ORT Session */
231 Ort::Session* session = new Ort::Session(env, proj_model_path.c_str(), session_options);
232 *session_ptr = session;
233 ei_printf("LOG_INFO: Loaded model %s\n", proj_model_path.c_str());
234
235 /* Input information */
236 size_t num_input_nodes = session->GetInputCount();
237 std::vector<const char *> input_node_names(num_input_nodes);
238 Ort::TypeInfo type_info = session->GetInputTypeInfo(0);
239 auto tensor_info = type_info.GetTensorTypeAndShapeInfo();
240 std::vector<int64_t> input_node_dims = tensor_info.GetShape();
241 ONNXTensorElementDataType input_tensor_type = tensor_info.GetElementType();
242
243 /* output information */
244 size_t num_output_nodes = session->GetOutputCount();
245 std::vector<const char *> output_node_names(num_output_nodes);
246 for (int i = 0; i < (int)num_output_nodes; i++)
247 {
248 output_node_names[i] = session->GetOutputName(i, allocator);
249 }
250 for (int i = 0; i < (int)num_input_nodes; i++)
251 {
252 input_node_names[i] = session->GetInputName(i, allocator);
253 }
254
255 type_info = session->GetOutputTypeInfo(0);
256 auto output_tensor_info = type_info.GetTensorTypeAndShapeInfo();
257 std::vector<int64_t> output_node_dims = output_tensor_info.GetShape();
258 size_t output_tensor_size = output_node_dims[1];
259
260 if (EI_IMPULSE_ONNX_ERROR == printTensorInfo(session, &input_node_names, &output_node_names)) {
261 ei_printf("LOG_ERROR: print tensor information failed!\n");
262 return EI_IMPULSE_ONNX_ERROR;
263 }
264
265 ssize_t input_tensor_size_bytes;
266 /* simplify ... using known dim values to calculate size */
267 size_t input_tensor_size = impulse->nn_input_frame_size;
268 void *inData;
269 if (input_tensor_type == ONNX_TENSOR_ELEMENT_DATA_TYPE_FLOAT)
270 {
271 input_tensor_size_bytes = input_tensor_size * sizeof(float);
272 inData = allocTensorMem(input_tensor_size_bytes, true);
273 }
274 else if (input_tensor_type == ONNX_TENSOR_ELEMENT_DATA_TYPE_UINT8)
275 {
276 input_tensor_size_bytes = input_tensor_size * sizeof(uint8_t);
277 inData = allocTensorMem(input_tensor_size_bytes, true);
278 }
279 else
280 {
281 ei_printf("LOG_ERROR: indata type not supported yet \n ");
282 return EI_IMPULSE_ONNX_ERROR;
283 }
284 auto memory_info = Ort::MemoryInfo::CreateCpu(OrtArenaAllocator, OrtMemTypeDefault);
285
286 Ort::Value input_tensor = Ort::Value::CreateTensor(memory_info, inData, input_tensor_size_bytes, input_node_dims.data(), 4, input_tensor_type);
287 input_tensors->push_back(std::move(input_tensor));
288
289 Ort::RunOptions* run_options = new Ort::RunOptions();
290 *run_options_ptr = run_options;
291 run_options->SetRunLogVerbosityLevel(2);
292 auto output_tensors_warm_up = session->Run(*run_options, input_node_names.data(), input_tensors->data(), 1, output_node_names.data(), num_output_nodes);
293
294 //void *outData = allocTensorMem(output_tensor_size * sizeof(float), true);
295 Ort::IoBinding* binding = new Ort::IoBinding(*session);
296 *binding_ptr = binding;
297 binding->BindInput(input_node_names[0], (*input_tensors)[0]);
298
299 for(int idx=0; idx < (int)num_output_nodes; idx++)
300 {
301 auto node_dims = output_tensors_warm_up[idx].GetTypeInfo().GetTensorTypeAndShapeInfo().GetShape();
302 size_t tensor_size = 1;
303 for(int j = node_dims.size()-1; j >= 0; j--)
304 {
305 tensor_size *= node_dims[j];
306 }
307 ONNXTensorElementDataType tensor_type = output_tensors_warm_up[idx].GetTypeInfo().GetTensorTypeAndShapeInfo().GetElementType();
308 if(tensor_type == ONNX_TENSOR_ELEMENT_DATA_TYPE_FLOAT)
309 {
310 tensor_size *= sizeof(float);
311 }
312 else if(tensor_type == ONNX_TENSOR_ELEMENT_DATA_TYPE_UINT8)
313 {
314 tensor_size *= sizeof(uint8_t);
315 }
316 else if(tensor_type == ONNX_TENSOR_ELEMENT_DATA_TYPE_INT64)
317 {
318 tensor_size *= sizeof(int64_t);
319 }
320 else
321 {
322 ei_printf("LOG_ERROR: Un Supported output tensor_type\n");
323 return EI_IMPULSE_ONNX_ERROR;
324 }
325
326 void * outData = allocTensorMem(tensor_size, true);
327 auto output_tensor = Ort::Value::CreateTensor(memory_info, (void *)outData, tensor_size, node_dims.data(), node_dims.size(),tensor_type);
328 output_tensors->push_back(std::move(output_tensor));
329 binding->BindOutput(output_node_names[idx], (*output_tensors)[idx]);
330 }
331
332 onnx_first_run = false;
333
334 return EI_IMPULSE_OK;
335}

349static EI_IMPULSE_ERROR inference_onnx_run(const ei_impulse_t *impulse,
350 void *config_ptr,
351 uint64_t ctx_start_us,
352 std::vector<Ort::Value>* input_tensors,
353 std::vector<Ort::Value>* output_tensors,
354 Ort::Session* session,
355 Ort::RunOptions* run_options,
356 Ort::IoBinding* binding,
357 ei_impulse_result_t *result,
358 bool debug) {
359
360 ei_learning_block_config_tflite_graph_t *block_config = (ei_learning_block_config_tflite_graph_t*)config_ptr;
361
362 session->Run(*run_options, *binding);
363
364 uint64_t ctx_end_us = ei_read_timer_us();
365 result->timing.classification_us = ctx_end_us - ctx_start_us;
366 result->timing.classification = (int)(result->timing.classification_us / 1000);
367
368 ONNXTensorElementDataType output_tensor_type = (*output_tensors).at(0).GetTypeInfo().GetTensorTypeAndShapeInfo().GetElementType();
369 void *out_data = output_tensors->front().GetTensorMutableData<void>();
370
371 // get output features count from model
372 auto node_dims = (*output_tensors).at(0).GetTypeInfo().GetTensorTypeAndShapeInfo().GetShape();
373 size_t output_tensor_features_count = 1;
374 for(int j = node_dims.size()-1; j >= 0; j--)
375 {
376 output_tensor_features_count *= node_dims[j];
377 }
378
379 // Read the predicted y value from the model's output tensor
380 if (debug) {
381 ei_printf("Predictions (time: %d ms.):\n", result->timing.classification);
382 }
383
384 EI_IMPULSE_ERROR fill_res = EI_IMPULSE_OK;
385
386 // NOTE: for now only yolox object detection supported
387 if (block_config->object_detection) {
388 switch (block_config->object_detection_last_layer) {
389 case EI_CLASSIFIER_LAST_LAYER_YOLOX: {
390 if (block_config->quantized == 1) {
391 ei_printf("ERR: YOLOX does not support quantized inference\n");
392 return EI_IMPULSE_UNSUPPORTED_INFERENCING_ENGINE;
393 }
394 else {
395 if (debug) {
396 ei_printf("YOLOX OUTPUT (%d ms.): ", result->timing.classification);
397 for (size_t ix = 0; ix < output_tensor_features_count; ix++) {
398 ei_printf_float(((float*)out_data)[ix]);
399 ei_printf(" ");
400 }
401 ei_printf("\n");
402 }
403 fill_res = fill_result_struct_f32_yolox_detect(
404 impulse,
405 block_config,
406 result,
407 (float*)out_data,
408 output_tensor_features_count);
409 }
410 break;
411 }
412 default: {
413 ei_printf("ERR: Unsupported object detection last layer (%d)\n",
414 block_config->object_detection_last_layer);
415 break;
416 }
417 }
418 }
419 else {
420#if EI_CLASSIFIER_QUANTIZATION_ENABLED == 1
421
422 switch (output_tensor_type) {
423 case ONNX_TENSOR_ELEMENT_DATA_TYPE_UINT8: {
424 fill_res = fill_result_struct_i8(impulse, result, (int8_t*)out_data, impulse->tflite_output_zeropoint, impulse->tflite_output_scale, debug);
425 break;
426 }
427 case ONNX_TENSOR_ELEMENT_DATA_TYPE_INT8: {
428 fill_res = fill_result_struct_i8(impulse, result, (int8_t*)out_data, impulse->tflite_output_zeropoint, impulse->tflite_output_scale, debug);
429 break;
430 }
431 default: {
432 ei_printf("ERR: Cannot handle output type (%d)\n", output_tensor_type);
433 return EI_IMPULSE_OUTPUT_TENSOR_WAS_NULL;
434 }
435 }
436
437#else
438 switch (output_tensor_type) {
439 case ONNX_TENSOR_ELEMENT_DATA_TYPE_FLOAT: {
440 fill_res = fill_result_struct_f32(impulse, result, (float*)out_data, debug);
441 break;
442 }
443 default: {
444 ei_printf("ERR: Cannot handle output type (%d)\n", output_tensor_type);
445 return EI_IMPULSE_OUTPUT_TENSOR_WAS_NULL;
446 }
447 }
448#endif
449 }
450
452 //for (size_t i = 0; i < output_tensors->size(); i++)
453 //{
454 // void *ptr = (*output_tensors)[i].GetTensorMutableData<void>();
455 // freeTensorMem(ptr, true);
456 //}
457 //for (size_t i = 0; i < input_tensors->size(); i++)
458 //{
459 // void *ptr = (*input_tensors)[i].GetTensorMutableData<void>();
460 // freeTensorMem(ptr, true);
461 //}
462
463 if (fill_res != EI_IMPULSE_OK) {
464 return fill_res;
465 }
466
467 return EI_IMPULSE_OK;
468}

479EI_IMPULSE_ERROR run_nn_inference(
480 const ei_impulse_t *impulse,
481 ei_feature_t *afmatrix,
482 uint32_t learn_block_index,
483 uint32_t* input_block_ids,
484 uint32_t input_block_ids_size,
485 ei_impulse_result_t *result,
486 void *config_ptr,
487 bool debug = false)
488{
489 static std::vector<Ort::Value> input_tensors;
490 static std::vector<Ort::Value> output_tensors;
491 static Ort::Session* session;
492 static Ort::RunOptions* run_options;
493 static Ort::IoBinding* binding;
494 uint64_t ctx_start_us;
495
496 ei_printf("test onnx tidl: %s\n", __FUNCTION__);
497 #pragma message ( "test onnx tidl: run_nn_inference")
498
499 EI_IMPULSE_ERROR init_res = inference_onnx_setup(impulse,
500 &ctx_start_us,
501 &input_tensors,
502 &output_tensors,
503 &session,
504 &run_options,
505 &binding);
506
507 if (init_res != EI_IMPULSE_OK || session == NULL || run_options == NULL ||
508 binding == NULL) {
509 ei_printf("LOG_ERROR: ONNX inference setup failed!\n");
510 return EI_IMPULSE_ONNX_ERROR;
511 }
512
513 uint64_t dsp_chw_start_us;
514 dsp_chw_start_us = ei_read_timer_us();
515
516 /*
517 ** Convert to CHW from HWC
518 */
519 // features matrix maps around the input tensor to not allocate any memory
520 float *input_buffer = input_tensors.front().GetTensorMutableData<float>();
521 ei::matrix_t fmatrix(1, impulse->nn_input_frame_size, input_buffer);
522
523 ei_dsp_config_image_t *config = (ei_dsp_config_image_t *)impulse->dsp_blocks[0].config;
524
525 size_t channels = strcmp(config->channels, "Grayscale") == 0 ? 1 : 3;
526 size_t height = impulse->input_height;
527 size_t width = impulse->input_width;
528
529 ei::matrix_t* matrix = afmatrix[0].matrix;
530
531 int dest_ix = 0;
532 for (size_t c=0; c < channels; c++) {
533 for (size_t h=0; h < height; h++) {
534 for (size_t w=0; w < width; w++) {
535 uint32_t src_ix = channels * width * h + w*channels + c;
536 fmatrix.buffer[dest_ix++] = matrix->buffer[src_ix];
537 }
538 }
539 }
540
541 uint64_t dsp_chw_end_us = ei_read_timer_us();
542 result->timing.dsp_us += dsp_chw_end_us - dsp_chw_start_us;
543 result->timing.dsp = (int)(result->timing.dsp_us / 1000);
544
545 if (debug) {
546 ei_printf("After Features (%ld us.): ", result->timing.dsp_us);
547 for (size_t ix = 0; ix < fmatrix.cols; ix++) {
548 ei_printf_float(fmatrix.buffer[ix]);
549 ei_printf(" ");
550 }
551 ei_printf("\n");
552 }
553
554 ctx_start_us = ei_read_timer_us();
555 EI_IMPULSE_ERROR run_res = inference_onnx_run(impulse,
556 config_ptr,
557 ctx_start_us,
558 &input_tensors,
559 &output_tensors,
560 session,
561 run_options,
562 binding,
563 result, debug);
564
565 if (run_res != EI_IMPULSE_OK) {
566 return run_res;
567 }
568
569 return EI_IMPULSE_OK;
570}

591
592 ei_printf("test onnx tidl: %s\n", __FUNCTION__);
593 #pragma message ( "test onnx tidl: run_nn_inference_image_quantized")
594
595 EI_IMPULSE_ERROR init_res = inference_onnx_setup(impulse,
596 &ctx_start_us,
597 &input_tensors, &output_tensors,
598 &session,
599 &run_options,

601
602 if (init_res != EI_IMPULSE_OK || session == NULL || run_options == NULL ||

613
614 uint64_t dsp_start_us = ei_read_timer_us();
615
616 ei::matrix_i8_t a_features_matrix(1, impulse->nn_input_frame_size);
617
618 // run DSP process and quantize automatically
619 int ret = extract_image_features_quantized(impulse, signal, &a_features_matrix, impulse->dsp_blocks[0].config, impulse->frequency,
620 impulse->learning_blocks[0].image_scaling);
621 if (ret != EIDSP_OK) {
622 ei_printf("ERR: Failed to run DSP process (%d)\n", ret);
623 return EI_IMPULSE_DSP_ERROR;
624 }
625
626 if (ei_run_impulse_check_canceled() == EI_IMPULSE_CANCELED) {
627 return EI_IMPULSE_CANCELED;
628 }
629
630 if (debug) {
631 ei_printf("Before Features: ");
632 for (size_t ix = 0; ix < a_features_matrix.cols; ix++) {
633 ei_printf("%d", (uint8_t)a_features_matrix.buffer[ix]);
634 ei_printf(" ");
635 }
636 ei_printf("\n");
637 }
638
639 /*
640 ** Convert to CHW from HWC
