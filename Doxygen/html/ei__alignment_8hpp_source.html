18float intersection_over_union(const ei_impulse_result_bounding_box_t bbox1, const ei_impulse_result_bounding_box_t bbox2) {
19 uint32_t x_left = std::max(bbox1.x, bbox2.x);
20 uint32_t y_top = std::max(bbox1.y, bbox2.y);
21 uint32_t x_right = std::min(bbox1.x + bbox1.width, bbox2.x + bbox2.width);
22 uint32_t y_bottom = std::min(bbox1.y + bbox1.height, bbox2.y + bbox2.height);
23
24 if (x_right < x_left || y_bottom < y_top) {
25 return 0.0;
26 }
27
28 uint32_t intersection_area = (x_right - x_left) * (y_bottom - y_top);
29 uint32_t bbox1_area = bbox1.width * bbox1.height;
30 uint32_t bbox2_area = bbox2.width * bbox2.height;
31
32 return static_cast<float>(intersection_area) / static_cast<float>(bbox1_area + bbox2_area - intersection_area);
33}
