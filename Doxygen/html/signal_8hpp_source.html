198 static void lfilter(const fvec& b, const fvec& a, const fvec& x, fvec& y, fvec& d)
199 {
200 /*
201 a[0]*y[n] = b[0] * x[n] + d[0][n-1]
202 d[0][n] = b[1] * x[n] - a[1] * y[n] + d[1][n-1]
203 d[1][n] = b[2] * x[n] - a[2] * y[n] + d[2][n-1]
204 ...
205 d[N-2][n] = b[N-1]*x[n] - a[N-1]*y[n] + d[N-1][n-1]
206 d[N-1][n] = b[N] * x[n] - a[N] * y[n]
207 */
208
209 assert(b.size() == a.size() && b.size() == d.size() + 1);
210 assert(d.size() > 0);
211 assert(y.size() >= x.size());
212 assert(a[0] != 0.0f);
213
214 const float one_over_a0 = 1.0f / a[0];
215 for (size_t ix = 0; ix < x.size(); ix++) {
216 const float xx = x[ix];
217 y[ix] = b[0] * xx + d[0];
218 y[ix] *= one_over_a0;
219 size_t jx;
220 for (jx = 1; jx < b.size() - 1; jx++) {
221 d[jx - 1] = b[jx] * xx - a[jx] * y[ix] + d[jx];
222 }
223 d[jx - 1] = b[jx] * xx - a[jx] * y[ix];
224 }
225 }

227 static void iir2(const float* x, float* y, size_t n, const float* b, const float* a, float* d)
228 {
229 /*
230 a[0]*y[n] = b[0] * x[n] + d[0][n-1]
231 d[0][n] = b[1] * x[n] - a[1] * y[n] + d[1][n-1]
232 d[1][n] = b[2] * x[n] - a[2] * y[n]
233 */
234 const float one_over_a0 = 1.0f / a[0];
235 for (size_t ix = 0; ix < n; ix++) {
236 const float xx = x[ix];
237 y[ix] = b[0] * xx + d[0];
238 y[ix] *= one_over_a0;
239 d[0] = b[1] * xx - a[1] * y[ix] + d[1];
240 d[1] = b[2] * xx - a[2] * y[ix];
241 }
242 }

318 static void resample_poly(const float* input, size_t input_size, fvec& output, int up, int down, const fvec& window)
319 {
320 assert(up > 0);
321 assert(down > 0);
322 assert(window.size() > 0 && (window.size() % 2) == 1);
323
324 int gcd_up_down = gcd(up, down);
325 up /= gcd_up_down;
326 down /= gcd_up_down;
327
328 if (up == 1 && down == 1) {
329 // output = std::move(fvec(input, input + input_size));
330 output = fvec(input, input + input_size);
331 return;
332 }
333
334 int n_out = (input_size * up);
335 n_out = n_out / down + (n_out % down == 0 ? 0 : 1);
336
337 fvec h = window;
338 scale(h, float(up));
339
340 output.resize(n_out);
341 upfirdn(input, input_size, output, up, down, h);
342 }
