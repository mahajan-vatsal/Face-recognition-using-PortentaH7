Machine Vision using Portenta H7 2
This project aims to develop a face recognition-based access control system using the Arduino Portenta H7 and Vision Shield, leveraging Edge Impulse for machine learning. The system captures facial images, processes them locally using an AI model deployed on the Portenta H7 and determines access based on authorised personnel.

39bool init_ethos(const uint8_t *model_arr, size_t model_arr_size, bool debug)
40{
41 int ret;
42 try {
43 // eioctl thros exception if ioctl returns negative value
44 ret = device.ioctl(ETHOSU_IOCTL_PING);
45 }
46 catch (EthosU::Exception &e) {
47 ei_printf("ERR: EthosU device not found: %s\n", e.what());
48 return false;
49 }
50
51 try {
52 // eioctl thros exception if ioctl returns negative value
53 ret = device.ioctl(ETHOSU_IOCTL_VERSION_REQ);
54 }
55 catch (EthosU::Exception &e) {
56 ei_printf("ERR: EthosU version request failed: %s\n", e.what());
57 return false;
58 }
59
60 if (debug) {
61 try {
62 EthosU::Capabilities capabilities = device.capabilities();
63 }
64 catch (EthosU::Exception &e) {
65 ei_printf("ERR: EthosU capabilities request failed: %s\n", e.what());
66 return false;
67 }
68
69 // std::cout << "Capabilities:" << std::endl
70 // << "\tversion_status:" << unsigned(capabilities.hwId.versionStatus) << std::endl
71 // << "\tversion:" << capabilities.hwId.version << std::endl
72 // << "\tproduct:" << capabilities.hwId.product << std::endl
73 // << "\tarchitecture:" << capabilities.hwId.architecture << std::endl
74 // << "\tdriver:" << capabilities.driver << std::endl
75 // << "\tmacs_per_cc:" << unsigned(capabilities.hwCfg.macsPerClockCycle) << std::endl
76 // << "\tcmd_stream_version:" << unsigned(capabilities.hwCfg.cmdStreamVersion) << std::endl
77 // << "\tcustom_dma:" << std::boolalpha << capabilities.hwCfg.customDma << std::endl;
78 }
79
80 std::shared_ptr<EthosU::Buffer> modelBuffer = std::make_shared<EthosU::Buffer>(device, model_arr_size);
81 // TODO: not sure why we need to do that, but Buffer constructor calls mmap
82 // while resize calls ioct ETHOSU_IOCTL_BUFFER_SET on the ethos device
83 modelBuffer->resize(model_arr_size);
84 std::copy(model_arr, model_arr + model_arr_size, modelBuffer->data());
85
86 // std::shared_ptr<EthosU::Buffer> networkBuffer = allocAndFill(device, "lamp-plant-int8_vela.tflite");
87 network = std::make_shared<EthosU::Network>(device, modelBuffer);
88
89 if(network->getIfmDims().size() > 1) {
90 ei_printf("ERR: Only single input models are supported\n");
91 return false;
92 }
93
94 init_done = true;
95
96 return true;
97}

99EI_IMPULSE_ERROR run_nn_inference(
100 const ei_impulse_t *impulse,
101 ei_feature_t *fmatrix,
102 uint32_t learn_block_index,
103 uint32_t* input_block_ids,
104 uint32_t input_block_ids_size,
105 ei_impulse_result_t *result,
106 void *config_ptr,
107 bool debug)
108{
109 ei_learning_block_config_tflite_graph_t *block_config = (ei_learning_block_config_tflite_graph_t*)config_ptr;
110 ei_config_ethos_graph_t *io_details = ((ei_config_ethos_graph_t*)block_config->graph_config);
111 std::vector<std::shared_ptr<EthosU::Buffer>> ifm;
112 std::vector<std::shared_ptr<EthosU::Buffer>> ofm;
113 ei::matrix_t* matrix;
114
115
116 if (!init_done) {
117 if(!init_ethos(io_details->model, io_details->model_size, debug)) {
118 return EI_IMPULSE_DEVICE_INIT_ERROR;
119 }
120 }
121
122 // so far only single input models are supported
123 // get the size of the first input tensor
124 auto ifmSize = network->getIfmDims()[0];
125 // create a buffer for the input tensor
126 std::shared_ptr<EthosU::Buffer> buffer = std::make_shared<EthosU::Buffer>(device, ifmSize);
127 // resize the buffer to the size of the input tensor
128 // TODO: not sure if we need to call it, but the reference code does
129 buffer->resize(ifmSize);
130 // // inputType is an enum TensorType
131 // auto inputType = network->getIfmTypes()[0];
132 // // inputShape is a vecor of dims
133 // auto inputShape = network->getIfmShapes()[0];
134
135 // get the matrix with input features
136 for (size_t i = 0; i < input_block_ids_size; i++) {
137#if EI_CLASSIFIER_SINGLE_FEATURE_INPUT == 0
138 uint16_t cur_mtx = input_block_ids[i];
139 matrix = NULL;
140
141 if (!find_mtx_by_idx(fmatrix, &matrix, cur_mtx, impulse->dsp_blocks_size + impulse->learning_blocks_size)) {
142 ei_printf("ERR: Cannot find matrix with id %zu\n", cur_mtx);
143 return EI_IMPULSE_INVALID_SIZE;
144 }
145#else
146 matrix = fmatrix[0].matrix;
147#endif
148 }
149
150 if(ifmSize != matrix->rows * matrix->cols) {
151 ei_printf("ERR: Input size mismatch\n");
152 return EI_IMPULSE_INVALID_SIZE;
153 }
154
155 // copy rescale the input features to int8 and copy to input buffer
156 for (size_t i = 0; i < ifmSize; i++) {
157 buffer->data()[i] = (int8_t)((matrix->buffer[i] / io_details->input_scale) + io_details->input_zeropoint);
158 }
159
160 // put the data into the input tensor
161 ifm.push_back(buffer);
162
163 // create a buffer for the output tensor
164 auto ofmSize = network->getOfmDims()[0];
165 ofm.push_back(std::make_shared<EthosU::Buffer>(device, ofmSize));
166
167 // get start time of the inference
168 uint64_t ctx_start_us = ei_read_timer_us();
169
170 // start inference
171 EthosU::Inference inference(network, ifm.begin(), ifm.end(), ofm.begin(), ofm.end());
172
173 // wait in 1 second (wait gets nanoseconds)
174 if (inference.wait(1000 * 1000 * 1000)) {
175 ei_printf("ERR: Inference timeout\n");
176 return EI_IMPULSE_INFERENCE_ERROR;
177 }
178
179 // get end time of the inference
180 uint64_t ctx_end_us = ei_read_timer_us();
181
182 // check if inference ended successfully
183 if (inference.status() != EthosU::InferenceStatus::OK) {
184 ei_printf("ERR: Inference status: %d\n", inference.status());
185 return EI_IMPULSE_INFERENCE_ERROR;
186 }
187
188 // calculate inference time
189 result->timing.classification_us = ctx_end_us - ctx_start_us;
190 result->timing.classification = (int)(result->timing.classification_us / 1000);
191
192 // get output data from the output tensor
193 auto data = inference.getOfmBuffers()[0]->data();
194
195 EI_IMPULSE_ERROR fill_res = EI_IMPULSE_OK;
196
197 if(block_config->object_detection) {
198 switch (block_config->object_detection_last_layer) {
199 case EI_CLASSIFIER_LAST_LAYER_FOMO: {
200 fill_res = fill_result_struct_i8_fomo(
201 impulse,
202 block_config,
203 result,
204 (int8_t*)data,
205 io_details->output_zeropoint,
206 io_details->output_scale,
207 impulse->fomo_output_size,
208 impulse->fomo_output_size);
209 break;
210 }
211 case EI_CLASSIFIER_LAST_LAYER_SSD: {
212 ei_printf("ERR: MobileNet SSD models are not supported by Ethos\n");
213 return EI_IMPULSE_LAST_LAYER_NOT_SUPPORTED;
214 }
215 case EI_CLASSIFIER_LAST_LAYER_YOLOV5:
216 case EI_CLASSIFIER_LAST_LAYER_YOLOV5_V5_DRPAI: {
217 int version = block_config->object_detection_last_layer == EI_CLASSIFIER_LAST_LAYER_YOLOV5_V5_DRPAI ?
218 5 : 6;
219
220 if (network->getOfmTypes()[0] == EthosU::TensorType_INT8) {
221 fill_res = fill_result_struct_quantized_yolov5(
222 impulse,
223 block_config,
224 result,
225 version,
226 (int8_t*)data,
227 io_details->output_zeropoint,
228 io_details->output_scale,
229 impulse->tflite_output_features_count,
230 debug);
231 }
232 else if (network->getOfmTypes()[0] == EthosU::TensorType_UINT8) {
233 fill_res = fill_result_struct_quantized_yolov5(
234 impulse,
235 block_config,
236 result,
237 version,
238 (uint8_t*)data,
239 io_details->output_zeropoint,
240 io_details->output_scale,
241 impulse->tflite_output_features_count,
242 debug);
243 }
244 else {
245 ei_printf("ERR: Invalid output type (%d) for YOLOv5 last layer\n", network->getOfmTypes()[0]);
246 return EI_IMPULSE_LAST_LAYER_NOT_SUPPORTED;
247 }
248 break;
249 }
250 case EI_CLASSIFIER_LAST_LAYER_TAO_SSD:
251 case EI_CLASSIFIER_LAST_LAYER_TAO_RETINANET: {
252 if (network->getOfmTypes()[0] == EthosU::TensorType_INT8) {
253 fill_res = fill_result_struct_quantized_tao_decode_detections(
254 impulse,
255 block_config,
256 result,
257 (int8_t*)data,
258 io_details->output_zeropoint,
259 io_details->output_scale,
260 impulse->tflite_output_features_count,
261 debug);
262 }
263 else if (network->getOfmTypes()[0] == EthosU::TensorType_UINT8) {
264 fill_res = fill_result_struct_quantized_tao_decode_detections(
265 impulse,
266 block_config,
267 result,
268 (uint8_t*)data,
269 io_details->output_zeropoint,
270 io_details->output_scale,
271 impulse->tflite_output_features_count,
272 debug);
273 }
274 else {
275 ei_printf("ERR: Invalid output type (%d) for TAO last layer\n", network->getOfmTypes()[0]);
276 return EI_IMPULSE_UNSUPPORTED_INFERENCING_ENGINE;
277 }
278 break;
279 }
280 case EI_CLASSIFIER_LAST_LAYER_TAO_YOLOV3: {
281 if (network->getOfmTypes()[0] == EthosU::TensorType_INT8) {
282 fill_res = fill_result_struct_quantized_tao_yolov3(
283 impulse,
284 block_config,
285 result,
286 (int8_t*)data,
287 io_details->output_zeropoint,
288 io_details->output_scale,
289 impulse->tflite_output_features_count,
290 debug);
291 }
292 else if (network->getOfmTypes()[0] == EthosU::TensorType_UINT8) {
293 fill_res = fill_result_struct_quantized_tao_yolov3(
294 impulse,
295 block_config,
296 result,
297 (uint8_t*)data,
298 io_details->output_zeropoint,
299 io_details->output_scale,
300 impulse->tflite_output_features_count,
301 debug);
302 }
303 else {
304 ei_printf("ERR: Invalid output type (%d) for TAO YOLOv3 layer\n", network->getOfmTypes()[0]);
305 return EI_IMPULSE_UNSUPPORTED_INFERENCING_ENGINE;
306 }
307 break;
308 }
309 case EI_CLASSIFIER_LAST_LAYER_TAO_YOLOV4: {
310 if (network->getOfmTypes()[0] == EthosU::TensorType_INT8) {
311 fill_res = fill_result_struct_quantized_tao_yolov4(
312 impulse,
313 block_config,
314 result,
315 (int8_t*)data,
316 io_details->output_zeropoint,
317 io_details->output_scale,
318 impulse->tflite_output_features_count,
319 debug);
320 }
321 else if (network->getOfmTypes()[0] == EthosU::TensorType_UINT8) {
322 fill_res = fill_result_struct_quantized_tao_yolov4(
323 impulse,
324 block_config,
325 result,
326 (uint8_t*)data,
327 io_details->output_zeropoint,
328 io_details->output_scale,
329 impulse->tflite_output_features_count,
330 debug);
331 }
332 else {
333 ei_printf("ERR: Invalid output type (%d) for TAO YOLOv4 layer\n", network->getOfmTypes()[0]);
334 return EI_IMPULSE_UNSUPPORTED_INFERENCING_ENGINE;
335 }
336 break;
337 }
338 default: {
339 ei_printf("ERR: Object Detection layer (%d) not supported\n", block_config->object_detection_last_layer);
340 return EI_IMPULSE_LAST_LAYER_NOT_SUPPORTED;
341 }
342 }
343 }
344 else {
345 fill_res = fill_result_struct_i8(impulse, result, (int8_t*)data, io_details->output_zeropoint, io_details->output_scale, debug);
346 }
347
348 return fill_res;
349}
