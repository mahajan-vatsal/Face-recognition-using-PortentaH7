175 static int find_peak_indexes(
176 matrix_t *input_matrix,
177 matrix_t *output_matrix,
178 float threshold,
179 uint16_t *peaks_found)
180 {
181 if (input_matrix->rows != 1) {
182 EIDSP_ERR(EIDSP_MATRIX_SIZE_MISMATCH);
183 }
184
185 if (output_matrix->cols != 1) {
186 EIDSP_ERR(EIDSP_MATRIX_SIZE_MISMATCH);
187 }
188
189 uint16_t out_ix = 0;
190 size_t in_size = input_matrix->cols;
191 float *in = input_matrix->buffer;
192 size_t out_size = output_matrix->rows;
193 float *out = output_matrix->buffer;
194
195 // for normalized threshold calculation
196 float min = FLT_MAX, max = 0.0f;
197 for (size_t ix = 0; ix < in_size - 1; ix++) {
198 if (in[ix] < min) {
199 min = in[ix];
200 }
201 if (in[ix] > max) {
202 max = in[ix];
203 }
204 }
205
206
207 float prev = in[0];
208
209 // so....
210 for (size_t ix = 1; ix < in_size - 1; ix++) {
211 // first make sure it's actually a peak...
212 if (in[ix] > prev && in[ix] > in[ix+1]) {
213 // then make sure the threshold is met (on both?)
214 float height = (in[ix] - prev) + (in[ix] - in[ix + 1]);
215 // printf("%d inx: %f height: %f threshold: %f\r\n", ix, in[ix], height, threshold);
216 if (height > threshold) {
217 out[out_ix] = ix;
218 out_ix++;
219 if (out_ix == out_size) break;
220 }
221 }
222
223 prev = in[ix];
224 }
225
226 *peaks_found = out_ix;
227
228 return EIDSP_OK;
229 }

239 static int find_fft_peaks(
240 matrix_t *fft_matrix,
241 matrix_t *output_matrix,
242 float sampling_freq,
243 float threshold,
244 uint16_t fft_length)
245 {
246 if (fft_matrix->rows != 1) {
247 EIDSP_ERR(EIDSP_MATRIX_SIZE_MISMATCH);
248 }
249
250 if (output_matrix->cols != 2) {
251 EIDSP_ERR(EIDSP_MATRIX_SIZE_MISMATCH);
252 }
253
254 if (output_matrix->rows == 0) {
255 return EIDSP_OK;
256 }
257
258 int ret;
259
260 int N = static_cast<int>(fft_length);
261 float T = 1.0f / sampling_freq;
262
263 EI_DSP_MATRIX(freq_space, 1, fft_matrix->cols);
264 ret = numpy::linspace(0.0f, 1.0f / (2.0f * T), floor(N / 2), freq_space.buffer);
265 if (ret != EIDSP_OK) {
266 EIDSP_ERR(ret);
267 }
268
269 EI_DSP_MATRIX(peaks_matrix, output_matrix->rows * 10, 1);
270
271 uint16_t peak_count;
272 ret = find_peak_indexes(fft_matrix, &peaks_matrix, 0.0f, &peak_count);
273 if (ret != EIDSP_OK) {
274 EIDSP_ERR(ret);
275 }
276
277 // turn this into C++ vector and sort it based on amplitude
278 ei_vector<freq_peak_t> peaks;
279 for (uint8_t ix = 0; ix < peak_count; ix++) {
280 freq_peak_t d;
281
282 d.freq = freq_space.buffer[static_cast<uint32_t>(peaks_matrix.buffer[ix])];
283 d.amplitude = fft_matrix->buffer[static_cast<uint32_t>(peaks_matrix.buffer[ix])];
284 // printf("freq %f : %f amp: %f\r\n", peaks_matrix.buffer[ix], d.freq, d.amplitude);
285 if (d.amplitude < threshold) {
286 d.freq = 0.0f;
287 d.amplitude = 0.0f;
288 }
289 peaks.push_back(d);
290 }
291 sort(peaks.begin(), peaks.end(),
292 [](const freq_peak_t & a, const freq_peak_t & b) -> bool
293 {
294 return a.amplitude > b.amplitude;
295 });
296
297 // fill with zeros at the end (if needed)
298 for (size_t ix = peaks.size(); ix < output_matrix->rows; ix++) {
299 freq_peak_t d;
300 d.freq = 0;
301 d.amplitude = 0;
302 peaks.push_back(d);
303 }
304
305 for (size_t row = 0; row < output_matrix->rows; row++) {
306 // col 0 is freq, col 1 is ampl
307 output_matrix->buffer[row * output_matrix->cols + 0] = peaks[row].freq;
308 output_matrix->buffer[row * output_matrix->cols + 1] = peaks[row].amplitude;
309 }
310
311 return EIDSP_OK;
312 }

325 int spectral_power_edges(
326 matrix_t *fft_matrix,
327 matrix_t *freq_matrix,
328 matrix_t *edges_matrix,
329 matrix_t *output_matrix,
330 float sampling_freq
331 ) {
332 if (fft_matrix->rows != 1 || freq_matrix->rows != 1) {
333 EIDSP_ERR(EIDSP_MATRIX_SIZE_MISMATCH);
334 }
335
336 if (edges_matrix->cols != 1) {
337 EIDSP_ERR(EIDSP_MATRIX_SIZE_MISMATCH);
338 }
339
340 if (output_matrix->rows != edges_matrix->rows - 1 || output_matrix->cols != edges_matrix->cols) {
341 EIDSP_ERR(EIDSP_MATRIX_SIZE_MISMATCH);
342 }
343
344 if (fft_matrix->cols != freq_matrix->cols) {
345 EIDSP_ERR(EIDSP_MATRIX_SIZE_MISMATCH);
346 }
347
348 EI_DSP_MATRIX(buckets, 1, edges_matrix->rows - 1);
349 EI_DSP_MATRIX(bucket_count, 1, edges_matrix->rows - 1);
350
351 for (uint16_t ix = 0; ix < freq_matrix->cols; ix++) {
352 float t = freq_matrix->buffer[ix];
353 float v = fft_matrix->buffer[ix];
354
355 // does this fit between any edges?
356 for (uint16_t ex = 0; ex < edges_matrix->rows - 1; ex++) {
357 if (t >= edges_matrix->buffer[ex] && t < edges_matrix->buffer[ex + 1]) {
358 buckets.buffer[ex] += v;
359 bucket_count.buffer[ex]++;
360 break;
361 }
362 }
363 }
364
365 // average out and push to vector
366 for (uint16_t ex = 0; ex < edges_matrix->rows - 1; ex++) {
367 if (bucket_count.buffer[ex] == 0.0f) {
368 output_matrix->buffer[ex] = 0.0f;
369 }
370 else {
371 output_matrix->buffer[ex] = buckets.buffer[ex] / bucket_count.buffer[ex];
372 }
373 }
374
375 return EIDSP_OK;
376 }

388 int periodogram(matrix_t *input_matrix, matrix_t *out_fft_matrix, matrix_t *out_freq_matrix, float sampling_freq, uint16_t n_fft)
389 {
390 if (input_matrix->rows != 1) {
391 EIDSP_ERR(EIDSP_MATRIX_SIZE_MISMATCH);
392 }
393
394 if (out_fft_matrix->rows != 1 || out_fft_matrix->cols != static_cast<uint32_t>(n_fft / 2 + 1)) {
395 EIDSP_ERR(EIDSP_MATRIX_SIZE_MISMATCH);
396 }
397
398 if (out_freq_matrix->rows != 1 || out_freq_matrix->cols != static_cast<uint32_t>(n_fft / 2 + 1)) {
399 EIDSP_ERR(EIDSP_MATRIX_SIZE_MISMATCH);
400 }
401
402 if (input_matrix->buffer == NULL) {
403 EIDSP_ERR(EIDSP_OUT_OF_MEM);
404 }
405
406 if (out_fft_matrix->buffer == NULL) {
407 EIDSP_ERR(EIDSP_OUT_OF_MEM);
408 }
409
410 if (out_freq_matrix->buffer == NULL) {
411 EIDSP_ERR(EIDSP_OUT_OF_MEM);
412 }
413
414 // map over the input buffer, so we can manipulate the number of columns
415 EI_DSP_MATRIX_B(welch_matrix, input_matrix->rows, input_matrix->cols, input_matrix->buffer);
416
417 uint16_t nperseg = n_fft;
418
419 if (n_fft > input_matrix->cols) {
420 nperseg = input_matrix->cols;
421 }
422 // make the column align to nperseg in this case
423 else if (n_fft < input_matrix->cols) {
424 welch_matrix.cols = n_fft;
425 }
426
427 EI_DSP_MATRIX(triage_segments, 1, nperseg);
428 for (uint16_t ix = 0; ix < nperseg; ix++) {
429 triage_segments.buffer[ix] = 1.0f;
430 }
431
432 float scale = 1.0f / (sampling_freq * nperseg);
433
434 for (uint16_t ix = 0; ix < n_fft / 2 + 1; ix++) {
435 out_freq_matrix->buffer[ix] = static_cast<float>(ix) * (1.0f / (n_fft * (1.0f / sampling_freq)));
436 }
437
438 int ret;
439
440 // now we need to detrend... which is done constant so just subtract the mean
441 EI_DSP_MATRIX(mean_matrix, 1, 1);
442 ret = numpy::mean(&welch_matrix, &mean_matrix);
443 if (ret != EIDSP_OK) {
444 EIDSP_ERR(ret);
445 }
446
447 ret = numpy::subtract(&welch_matrix, &mean_matrix);
448 if (ret != EIDSP_OK) {
449 EIDSP_ERR(ret);
450 }
451
452 fft_complex_t *fft_output = (fft_complex_t*)ei_dsp_calloc((n_fft / 2 + 1) * sizeof(fft_complex_t), 1);
453 ret = numpy::rfft(welch_matrix.buffer, welch_matrix.cols, fft_output, n_fft / 2 + 1, n_fft);
454 if (ret != EIDSP_OK) {
455 ei_dsp_free(fft_output, (n_fft / 2 + 1) * sizeof(fft_complex_t));
456 EIDSP_ERR(ret);
457 }
458
459 // conjugate and then multiply with itself and scale
460 for (uint16_t ix = 0; ix < n_fft / 2 + 1; ix++) {
461 fft_output[ix].r = (fft_output[ix].r * fft_output[ix].r) +
462 (abs(fft_output[ix].i * fft_output[ix].i));
463 fft_output[ix].i = 0.0f;
464
465 fft_output[ix].r *= scale;
466
467 if (ix != n_fft / 2) {
468 fft_output[ix].r *= 2;
469 }
470
471 // then multiply by itself...
472 out_fft_matrix->buffer[ix] = fft_output[ix].r;
473 }
474
475 ei_dsp_free(fft_output, (n_fft / 2 + 1) * sizeof(fft_complex_t));
476
477 return EIDSP_OK;
478 }
