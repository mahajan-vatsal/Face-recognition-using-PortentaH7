46// Holds placeholder information for a scratch buffer request from a kernel.
47// This struct is only used during the model prepare stage. Each request from a
48// kernel is stored in the head section. During the prepare stage, the head
49// section will at least hold kMaxScratchBuffersPerOp number of requests plus
50// any requests from previous kernel requests.
51//
52// When the memory plan is finalized, these structs are no longer used in favor
53// of a sequential, array of ScratchBufferHandle allocations in the tail
54// section. These allocations are indexed by the request API defined in the
55// TfLiteContext struct.
56struct ScratchBufferRequest {
57 // Number of bytes required by the buffer. The actual allocated size might be
58 // greater than `bytes` due to buffer alignment.
59 size_t bytes;
60 // Node where the buffer is allocated for. This provides useful information to
61 // determine the lifetime of the buffer. In AllocationInfo, this buffer will
62 // have `before` = node_idx and `after` = node_idx.
63 int node_idx;
64 int subgraph_idx;
65};

89// Allocator responsible for allocating memory for all intermediate tensors
90// necessary to invoke a model.
91//
92// The lifetime of the model, tensor arena and error reporter must be at
93// least as long as that of the allocator object, since the allocator needs
94// them to be accessible during its entire lifetime.
95//
96// The MicroAllocator simply plans out additional allocations that are required
97// to standup a model for inference in TF Micro. This class currently relies on
98// an additional allocator - SingleArenaBufferAllocator - for all allocations
99// from an arena. These allocations are divided into head (non-persistent) and
100// tail (persistent) regions:
101//
102// Memory layout to help understand how it works
103// This information could change in the future version.
104// ************** .memory_allocator->GetBuffer()
105// Tensors/Scratch buffers (head)
106// ************** .head_watermark
107// unused memory
108// ************** .memory_allocator->GetBuffer() + ->GetMaxBufferSize()
109// - ->GetDataSize()
110// persistent area (tail)
111// ************** .memory_allocator->GetBuffer() + ->GetMaxBufferSize()
112class MicroAllocator {
113 public:
114 // Creates a MicroAllocator instance from a given tensor arena. This arena
115 // will be managed by the created instance. The GreedyMemoryPlanner will
116 // by default be used and created on the arena.

118 // bytes aligned, otherwise some head room will be wasted.
119 // TODO(b/157615197): Cleanup constructor + factory usage.
120 static MicroAllocator* Create(uint8_t* tensor_arena, size_t arena_size);

148 // Allocates internal resources required for model inference for each subgraph
149 // from the arena.
150 //
151 // This method will run through the flatbuffer data supplied in the model to
152 // properly allocate tensor, node, and op registration data. This method is
153 // expected to be followed with a call to FinishModelAllocation() Returns a
154 // pointer to an array of SubgraphAllocations (also stored in the tail of the
155 // arena) where each index corresponds to a different subgraph in the model.
156 // Return value is nullptr if the allocations failed.
157 SubgraphAllocations* StartModelAllocation(const Model* model);

159 // Finish allocating internal resources required for model inference.
160 //
161 // -Plan the memory for activation tensors and scratch buffers.
162 // -Update eval tensors for each subgraph based on planned offsets.
163 // -Allocate scratch buffer handles array and update based on planned offsets.
164 //
165 // This method should be called after assigning model resources
166 // in StartModelAllocation(). The subgraph_allocations pointer should be the
167 // value passed into this class during StartModelAllocation(). Scratch buffer
168 // handles are stored in the out-param `scratch_buffer_handles` array which is
169 // allocated in this method. This value will be used in `GetScratchBuffer`
170 // call to retrieve scratch buffers.
171 TfLiteStatus FinishModelAllocation(
172 const Model* model, SubgraphAllocations* subgraph_allocations,
173 ScratchBufferHandle** scratch_buffer_handles);
