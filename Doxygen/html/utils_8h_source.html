The system captures facial images, processes them locally using an AI model deployed on the Portenta H7 and determines access based on authorised personnel.

54 q31_t in,
55 q31_t * dst,
56 const q31_t * pRecipTable)
57 {
58 q31_t out;
59 uint32_t tempVal;
60 uint32_t index, i;
61 uint32_t signBits;
62
63 if (in > 0)
64 {
65 signBits = ((uint32_t) (__CLZ( (uint32_t)in) - 1));
66 }
67 else
68 {
69 signBits = ((uint32_t) (__CLZ((uint32_t)(-in)) - 1));
70 }
71
72 /* Convert input sample to 1.31 format */
73 in = (in << signBits);
74
75 /* calculation of index for initial approximated Val */
76 index = (uint32_t)(in >> 24);
77 index = (index & INDEX_MASK);
78
79 /* 1.31 with exp 1 */
80 out = pRecipTable[index];
81
82 /* calculation of reciprocal value */
83 /* running approximation for two iterations */

85 {
86 tempVal = (uint32_t) (((q63_t) in * out) >> 31);

91 }
92
93 /* write output */
94 *dst = out;
95
96 /* return num of signbits of out = 1/in value */
97 return (signBits + 1U);

159__STATIC_INLINE void arm_norm_64_to_32u(uint64_t in, int32_t * normalized, int32_t *norm)
160{
161 int32_t n1;
162 int32_t hi = (int32_t) (in >> 32);
163 int32_t lo = (int32_t) ((in << 32) >> 32);
164
165 n1 = __CLZ((uint32_t)hi) - 32;
166 if (!n1)
167 {
168 /*
169 * input fits in 32-bit
170 */
171 n1 = __CLZ((uint32_t)lo);
172 if (!n1)
173 {
174 /*
175 * MSB set, need to scale down by 1
176 */
177 *norm = -1;
178 *normalized = (((uint32_t) lo) >> 1);
179 } else
180 {
181 if (n1 == 32)
182 {
183 /*
184 * input is zero
185 */
186 *norm = 0;
187 *normalized = 0;
188 } else
189 {
190 /*
191 * 32-bit normalization
192 */
193 *norm = n1 - 1;
194 *normalized = lo << *norm;
195 }
196 }
197 } else
198 {
199 /*
200 * input fits in 64-bit
201 */
202 n1 = 1 - n1;
203 *norm = -n1;
204 /*
205 * 64 bit normalization
206 */
207 *normalized = (int32_t)(((uint32_t)lo) >> n1) | (hi << (32 - n1));
208 }
209}

211__STATIC_INLINE int32_t arm_div_int64_to_int32(int64_t num, int32_t den)
212{
213 int32_t result;
214 uint64_t absNum;
215 int32_t normalized;
216 int32_t norm;
217
218 /*
219 * if sum fits in 32bits
220 * avoid costly 64-bit division
221 */
222 if (num == (int64_t)LONG_MIN)
223 {
224 absNum = LONG_MAX;
225 }
226 else
227 {
228 absNum = (uint64_t) (num > 0 ? num : -num);
229 }
230 arm_norm_64_to_32u(absNum, &normalized, &norm);
231 if (norm > 0)
232 /*
233 * 32-bit division
234 */
235 result = (int32_t) num / den;
236 else
237 /*
238 * 64-bit division
239 */
240 result = (int32_t) (num / den);
241
242 return result;
243}
