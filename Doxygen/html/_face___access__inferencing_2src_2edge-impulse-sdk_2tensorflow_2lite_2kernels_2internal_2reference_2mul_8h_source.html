27// Element-wise mul that can often be used for inner loop of broadcast Mul as
28// well as the non-broadcast Mul.
29inline void MulElementwise(int size, const ArithmeticParams& params,
30 const uint8_t* input1_data,
31 const uint8_t* input2_data, uint8_t* output_data) {
32 for (int i = 0; i < size; ++i) {
33 const int32_t input1_val = params.input1_offset + input1_data[i];
34 const int32_t input2_val = params.input2_offset + input2_data[i];
35 const int32_t unclamped_result =
36 params.output_offset +
37 MultiplyByQuantizedMultiplier(input1_val * input2_val,
38 params.output_multiplier,
39 params.output_shift);
40 const int32_t clamped_output =
41 std::min(params.quantized_activation_max,
42 std::max(params.quantized_activation_min, unclamped_result));
43 output_data[i] = static_cast<uint8_t>(clamped_output);
44 }
45}

54 GetActivationParams(params, &output_activation_min, &output_activation_max);
55

57 MatchingExtendedShapeFlatSize(input1_shape, input2_shape, output_shape);
58 for (int i = 0; i < flat_size; ++i) {

60 input1_data[i] * input2_data[i], output_activation_min,
61 output_activation_max);
62 }

91inline void BroadcastMul4DSlow(const ArithmeticParams& params,
92 const RuntimeShape& input1_shape,
93 const uint8_t* input1_data,
94 const RuntimeShape& input2_shape,
95 const uint8_t* input2_data,
96 const RuntimeShape& output_shape,
97 uint8_t* output_data) {
98 NdArrayDesc<4> desc1;
99 NdArrayDesc<4> desc2;
100 NdArrayDescsForElementwiseBroadcast(input1_shape, input2_shape, &desc1,
101 &desc2);
102 const RuntimeShape extended_output_shape =
103 RuntimeShape::ExtendedShape(4, output_shape);
104
105 for (int b = 0; b < extended_output_shape.Dims(0); ++b) {
106 for (int y = 0; y < extended_output_shape.Dims(1); ++y) {
107 for (int x = 0; x < extended_output_shape.Dims(2); ++x) {
108 for (int c = 0; c < extended_output_shape.Dims(3); ++c) {
109 const int32_t input1_val =
110 params.input1_offset +
111 input1_data[SubscriptToIndex(desc1, b, y, x, c)];
112 const int32_t input2_val =
113 params.input2_offset +
114 input2_data[SubscriptToIndex(desc2, b, y, x, c)];
115 const int32_t unclamped_result =
116 params.output_offset +
117 MultiplyByQuantizedMultiplier(input1_val * input2_val,
118 params.output_multiplier,
119 params.output_shift);
120 const int32_t clamped_output = std::min(
121 params.quantized_activation_max,
122 std::max(params.quantized_activation_min, unclamped_result));
123 output_data[Offset(extended_output_shape, b, y, x, c)] =
124 static_cast<uint8_t>(clamped_output);
125 }
126 }
127 }
128 }

131template <typename T>
132void BroadcastMul4DSlow(const ArithmeticParams& params,
133 const RuntimeShape& unextended_input1_shape,
134 const T* input1_data,
135 const RuntimeShape& unextended_input2_shape,
136 const T* input2_data,
137 const RuntimeShape& unextended_output_shape,
138 T* output_data) {
139 T output_activation_min;
140 T output_activation_max;
141 GetActivationParams(params, &output_activation_min, &output_activation_max);
142
143 TFLITE_DCHECK_LE(unextended_input1_shape.DimensionsCount(), 4);
144 TFLITE_DCHECK_LE(unextended_input2_shape.DimensionsCount(), 4);
145 TFLITE_DCHECK_LE(unextended_output_shape.DimensionsCount(), 4);
146 const RuntimeShape output_shape =
147 RuntimeShape::ExtendedShape(4, unextended_output_shape);
148
149 NdArrayDesc<4> desc1;
150 NdArrayDesc<4> desc2;
151 NdArrayDescsForElementwiseBroadcast(unextended_input1_shape,
152 unextended_input2_shape, &desc1, &desc2);
153
154 // In Tensorflow, the dimensions are canonically named (batch_number, row,
155 // col, channel), with extents (batches, height, width, depth), with the
156 // trailing dimension changing most rapidly (channels has the smallest stride,
157 // typically 1 element).
158 //
159 // In generated C code, we store arrays with the dimensions reversed. The
160 // first dimension has smallest stride.
161 //
162 // We name our variables by their Tensorflow convention, but generate C code
163 // nesting loops such that the innermost loop has the smallest stride for the
164 // best cache behavior.
165 for (int b = 0; b < output_shape.Dims(0); ++b) {
166 for (int y = 0; y < output_shape.Dims(1); ++y) {
167 for (int x = 0; x < output_shape.Dims(2); ++x) {
168 for (int c = 0; c < output_shape.Dims(3); ++c) {
169 output_data[Offset(output_shape, b, y, x, c)] =
170 ActivationFunctionWithMinMax(
171 input1_data[SubscriptToIndex(desc1, b, y, x, c)] *
172 input2_data[SubscriptToIndex(desc2, b, y, x, c)],
173 output_activation_min, output_activation_max);
174 }
175 }
176 }
177 }
