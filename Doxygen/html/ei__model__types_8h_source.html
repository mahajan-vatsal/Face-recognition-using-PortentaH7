Machine Vision using Portenta H7 2
This project aims to develop a face recognition-based access control system using the Arduino Portenta H7 and Vision Shield, leveraging Edge Impulse for machine learning. The system captures facial images, processes them locally using an AI model deployed on the Portenta H7 and determines access based on authorised personnel.

303class ei_impulse_state_t {
304typedef DspHandle* _dsp_handle_ptr_t;
305public:
306 const ei_impulse_t *impulse; // keep a pointer to the impulse
307 _dsp_handle_ptr_t *dsp_handles;
308 bool is_temp_handle = false; // to know if we're using the old (stateless) API
309 ei_impulse_state_t(const ei_impulse_t *impulse)
310 : impulse(impulse)
311 {
312 const auto num_dsp_blocks = impulse->dsp_blocks_size;
313 dsp_handles = (_dsp_handle_ptr_t*)ei_malloc(sizeof(_dsp_handle_ptr_t)*num_dsp_blocks);
314 for(size_t ix = 0; ix < num_dsp_blocks; ix++) {
315 dsp_handles[ix] = nullptr;
316 }
317 }
318
319 DspHandle* get_dsp_handle(size_t ix) {
320 if (dsp_handles[ix] == nullptr) {
321 dsp_handles[ix] = impulse->dsp_blocks[ix].factory(impulse->dsp_blocks[ix].config, impulse->frequency);
322 }
323 return dsp_handles[ix];
324 }
325
326 void reset()
327 {
328 for (size_t ix = 0; ix < impulse->dsp_blocks_size; ix++) {
329 if (dsp_handles[ix] != nullptr) {
330 delete dsp_handles[ix];
331 dsp_handles[ix] = nullptr;
332 }
333 }
334 }
335
336 void* operator new(size_t size) {
337 return ei_malloc(size);
338 }
339
340 void operator delete(void* ptr) {
341 ei_free(ptr);
342 }
343
344 void* operator new[](size_t size) {
345 return ei_malloc(size);
346 }
347
348 void operator delete[](void* ptr) {
349 ei_free(ptr);
350 }
351
352 ~ei_impulse_state_t()
353 {
354 reset();
355 ei_free(dsp_handles);
356 }
357};
