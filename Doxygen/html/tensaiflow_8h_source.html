Machine Vision using Portenta H7 2
This project aims to develop a face recognition-based access control system using the Arduino Portenta H7 and Vision Shield, leveraging Edge Impulse for machine learning. The system captures facial images, processes them locally using an AI model deployed on the Portenta H7 and determines access based on authorised personnel.

71EI_IMPULSE_ERROR run_nn_inference(
72 const ei_impulse_t *impulse,
73 ei_feature_t *fmatrix,
74 uint32_t learn_block_index,
75 uint32_t* input_block_ids,
76 uint32_t input_block_ids_size,
77 ei_impulse_result_t *result,
78 void *config_ptr,
79 bool debug = false)
80{
81 ei_learning_block_config_tflite_graph_t *block_config = (ei_learning_block_config_tflite_graph_t*)config_ptr;
82 ei_config_tensaiflow_graph_t *graph_config = (ei_config_tensaiflow_graph_t*)block_config->graph_config;
83
84 if (block_config->object_detection) {
85 ei_printf("ERR: Object detection models are not supported with TensaiFlow\n");
86 return EI_IMPULSE_UNSUPPORTED_INFERENCING_ENGINE;
87 }
88
89 uint64_t ctx_start_us = ei_read_timer_us();
90 uint32_t time, cycles;
91
92 /* Run tensaiflow inference */
93 infer((const void *)impulse, &time, &cycles);
94
95 // Inference results returned by post_process() and copied into infer_results
96
97 result->timing.classification_us = ei_read_timer_us() - ctx_start_us;
98 result->timing.classification = (int)(result->timing.classification_us / 1000);
99
100 for (uint32_t ix = 0; ix < impulse->label_count; ix++) {
101 float value;
102 // Dequantize the output if it is int8
103 value = static_cast<float>(infer_result[ix] - graph_config->output_zeropoint) *
104 graph_config->output_scale;
105
106 if (debug) {
107 ei_printf("%s:\t", impulse->categories[ix]);
108 ei_printf_float(value);
109 ei_printf("\n");
110 }
111 result->classification[ix].label = impulse->categories[ix];
112 result->classification[ix].value = value;
113 }
114
115 return EI_IMPULSE_OK;
116
117}

124EI_IMPULSE_ERROR run_nn_inference_image_quantized(
125 const ei_impulse_t *impulse,
126 signal_t *signal,
127 ei_impulse_result_t *result,
128 void *config_ptr,
129 bool debug = false)
130{
131 ei_learning_block_config_tflite_graph_t *block_config = (ei_learning_block_config_tflite_graph_t*)config_ptr;
132 ei_config_tensaiflow_graph_t *graph_config = (ei_config_tensaiflow_graph_t*)block_config->graph_config;
133
134 uint64_t ctx_start_us;
135 uint64_t dsp_start_us = ei_read_timer_us();
136
137 ei::matrix_i8_t features_matrix(1, impulse->nn_input_frame_size);
138 processed_features = (int8_t *) features_matrix.buffer;
139
140 // run DSP process and quantize automatically
141 int ret = extract_image_features_quantized(
142 signal,
143 &features_matrix,
144 impulse->dsp_blocks[0].config,
145 graph_config->input_scale,
146 graph_config->input_zeropoint,
147 impulse->frequency,
148 impulse->learning_blocks[0].image_scaling);
149
150 if (ret != EIDSP_OK) {
151 ei_printf("ERR: Failed to run DSP process (%d)\n", ret);
152 return EI_IMPULSE_DSP_ERROR;
153 }
154
155 if (ei_run_impulse_check_canceled() == EI_IMPULSE_CANCELED) {
156 return EI_IMPULSE_CANCELED;
157 }
158
159 result->timing.dsp_us = ei_read_timer_us() - dsp_start_us;
160 result->timing.dsp = (int)(result->timing.dsp_us / 1000);
161
162 if (debug) {
163 ei_printf("Features (%d ms.): ", result->timing.dsp);
164 for (size_t ix = 0; ix < features_matrix.cols; ix++) {
165 ei_printf_float((features_matrix.buffer[ix] - graph_config->input_zeropoint) * graph_config->input_scale);
166 ei_printf(" ");
167 }
168 ei_printf("\n");
169 }
170
171 uint32_t time, cycles;
172 ctx_start_us = ei_read_timer_us();
173
174 /* Run tensaiflow inference */
175 infer((const void *)impulse, &time, &cycles);
176
177 // Inference results returned by post_process() and copied into infer_results
178
179 EI_IMPULSE_ERROR fill_res = EI_IMPULSE_OK;
180
181 if (block_config->object_detection) {
182 switch (block_config->object_detection_last_layer) {
183 case EI_CLASSIFIER_LAST_LAYER_FOMO: {
184 if (block_config->quantized == 1) {
185 fill_res = fill_result_struct_i8_fomo(
186 impulse,
187 block_config,
188 result,
189 infer_result,
190 graph_config->output_zeropoint,
191 graph_config->output_scale,
192 impulse->fomo_output_size,
193 impulse->fomo_output_size);
194 }
195 else {
196 ei_printf("ERR: TensaiFlow does not support float32 inference\n");
197 return EI_IMPULSE_UNSUPPORTED_INFERENCING_ENGINE;
198 }
199 break;
200 }
201 default: {
202 ei_printf("ERR: Unsupported object detection last layer (%d)\n",
203 block_config->object_detection_last_layer);
204 return EI_IMPULSE_UNSUPPORTED_INFERENCING_ENGINE;
205 }
206 }
207 }
208 else {
209 if (block_config->quantized == 1) {
210 fill_res = fill_result_struct_i8(
211 impulse,
212 result,
213 infer_result,
214 graph_config->output_zeropoint,
215 graph_config->output_scale,
216 debug);
217 }
218 else {
219 ei_printf("ERR: TensaiFlow does not support float32 inference\n");
220 return EI_IMPULSE_UNSUPPORTED_INFERENCING_ENGINE;
221 }
222 }
223
224 if (fill_res != EI_IMPULSE_OK) {
225 return fill_res;
226 }
227
228 result->timing.classification_us = ei_read_timer_us() - ctx_start_us;
229 result->timing.classification = (int)(result->timing.classification_us / 1000);
230 return EI_IMPULSE_OK;
231
232}
