Machine Vision using Portenta H7 2

The system captures facial images, processes them locally using an AI model deployed on the Portenta H7 and determines access based on authorised personnel.

41static EI_IMPULSE_ERROR inference_tflite_setup(
42 ei_learning_block_config_tflite_graph_t *block_config,
43 uint64_t *ctx_start_us,
44 TfLiteTensor* input,
45 TfLiteTensor* output,
46 TfLiteTensor* output_labels,
47 TfLiteTensor* output_scores,
48 ei_unique_ptr_t& p_tensor_arena) {
49
50 ei_config_tflite_eon_graph_t *graph_config = (ei_config_tflite_eon_graph_t*)block_config->graph_config;
51
52 *ctx_start_us = ei_read_timer_us();
53
54 TfLiteStatus init_status = graph_config->model_init(ei_aligned_calloc);
55 if (init_status != kTfLiteOk) {
56 ei_printf("Failed to initialize the model (error code %d)\n", init_status);
57 return EI_IMPULSE_TFLITE_ARENA_ALLOC_FAILED;
58 }
59
60 TfLiteStatus status;
61
62 status = graph_config->model_input(0, input);
63 if (status != kTfLiteOk) {
64 return EI_IMPULSE_TFLITE_ERROR;
65 }
66 status = graph_config->model_output(block_config->output_data_tensor, output);
67 if (status != kTfLiteOk) {
68 return EI_IMPULSE_TFLITE_ERROR;
69 }
70
71 if (block_config->object_detection_last_layer == EI_CLASSIFIER_LAST_LAYER_SSD) {
72 status = graph_config->model_output(block_config->output_score_tensor, output_scores);
73 if (status != kTfLiteOk) {
74 return EI_IMPULSE_TFLITE_ERROR;
75 }
76 status = graph_config->model_output(block_config->output_labels_tensor, output_labels);
77 if (status != kTfLiteOk) {
78 return EI_IMPULSE_TFLITE_ERROR;
79 }
80 }
81
82 return EI_IMPULSE_OK;
83}

97static EI_IMPULSE_ERROR inference_tflite_run(
98 const ei_impulse_t *impulse,
99 ei_learning_block_config_tflite_graph_t *block_config,
100 uint64_t ctx_start_us,
101 TfLiteTensor* output,
102 TfLiteTensor* labels_tensor,
103 TfLiteTensor* scores_tensor,
104 uint8_t* tensor_arena,
105 ei_impulse_result_t *result,
106 bool debug) {
107
108 ei_config_tflite_eon_graph_t *graph_config = (ei_config_tflite_eon_graph_t*)block_config->graph_config;
109
110 if (graph_config->model_invoke() != kTfLiteOk) {
111 return EI_IMPULSE_TFLITE_ERROR;
112 }
113
114 uint64_t ctx_end_us = ei_read_timer_us();
115
116 result->timing.classification_us = ctx_end_us - ctx_start_us;
117 result->timing.classification = (int)(result->timing.classification_us / 1000);
118
119 // Read the predicted y value from the model's output tensor
120 if (debug) {
121 ei_printf("Predictions (time: %d ms.):\n", result->timing.classification);
122 }
123
124 EI_IMPULSE_ERROR fill_res = fill_result_struct_from_output_tensor_tflite(
125 impulse, block_config, output, labels_tensor, scores_tensor, result, debug);
126
127 if (fill_res != EI_IMPULSE_OK) {
128 return fill_res;
129 }
130
131 if (ei_run_impulse_check_canceled() == EI_IMPULSE_CANCELED) {
132 return EI_IMPULSE_CANCELED;
133 }
134
135 return EI_IMPULSE_OK;
136}

147EI_IMPULSE_ERROR run_nn_inference_from_dsp(
148 ei_learning_block_config_tflite_graph_t *block_config,
149 signal_t *signal,
150 matrix_t *output_matrix)
151{
152 TfLiteTensor input;
153 TfLiteTensor output;
154 TfLiteTensor output_scores;
155 TfLiteTensor output_labels;
156 uint64_t ctx_start_us = ei_read_timer_us();
157 ei_unique_ptr_t p_tensor_arena(nullptr, ei_aligned_free);
158 ei_config_tflite_eon_graph_t *graph_config = (ei_config_tflite_eon_graph_t*)block_config->graph_config;
159
160 EI_IMPULSE_ERROR init_res = inference_tflite_setup(
161 block_config,
162 &ctx_start_us,
163 &input,
164 &output,
165 &output_labels,
166 &output_scores,
167 p_tensor_arena);
168
169 if (init_res != EI_IMPULSE_OK) {
170 return init_res;
171 }
172
173 auto input_res = fill_input_tensor_from_signal(signal, &input);
174 if (input_res != EI_IMPULSE_OK) {
175 return input_res;
176 }
177
178 // invoke the model
179 if (graph_config->model_invoke() != kTfLiteOk) {
180 return EI_IMPULSE_TFLITE_ERROR;
181 }
182
183 auto output_res = fill_output_matrix_from_tensor(&output, output_matrix);
184 if (output_res != EI_IMPULSE_OK) {
185 return output_res;
186 }
187
188 if (graph_config->model_reset(ei_aligned_free) != kTfLiteOk) {
189 return EI_IMPULSE_TFLITE_ERROR;
190 }
191
192 return EI_IMPULSE_OK;
193}

204EI_IMPULSE_ERROR run_nn_inference(
205 const ei_impulse_t *impulse,
206 ei_feature_t *fmatrix,
207 uint32_t learn_block_index,
208 uint32_t* input_block_ids,
209 uint32_t input_block_ids_size,
210 ei_impulse_result_t *result,
211 void *config_ptr,
212 bool debug = false)
213{
214 ei_learning_block_config_tflite_graph_t *block_config = (ei_learning_block_config_tflite_graph_t*)config_ptr;
215 ei_config_tflite_eon_graph_t *graph_config = (ei_config_tflite_eon_graph_t*)block_config->graph_config;
216
217 TfLiteTensor input;
218 TfLiteTensor output;
219 TfLiteTensor output_scores;
220 TfLiteTensor output_labels;
221
222 uint64_t ctx_start_us = ei_read_timer_us();
223 ei_unique_ptr_t p_tensor_arena(nullptr, ei_aligned_free);
224
225 EI_IMPULSE_ERROR init_res = inference_tflite_setup(
226 block_config,
227 &ctx_start_us,
228 &input,
229 &output,
230 &output_labels,
231 &output_scores,
232 p_tensor_arena);
233
234 if (init_res != EI_IMPULSE_OK) {
235 return init_res;
236 }
237
238 uint8_t* tensor_arena = static_cast<uint8_t*>(p_tensor_arena.get());
239
240 size_t mtx_size = impulse->dsp_blocks_size + impulse->learning_blocks_size;
241 auto input_res = fill_input_tensor_from_matrix(fmatrix, &input, input_block_ids, input_block_ids_size, mtx_size);
242 if (input_res != EI_IMPULSE_OK) {
243 return input_res;
244 }
245
246 EI_IMPULSE_ERROR run_res = inference_tflite_run(
247 impulse,
248 block_config,
249 ctx_start_us,
250 &output,
251 &output_labels,
252 &output_scores,
253 tensor_arena, result, debug);
254
255 if (result->copy_output) {
256 auto output_res = fill_output_matrix_from_tensor(&output, fmatrix[impulse->dsp_blocks_size + learn_block_index].matrix);
257 if (output_res != EI_IMPULSE_OK) {
258 return output_res;
259 }
260 }
261
262 graph_config->model_reset(ei_aligned_free);
263
264 if (run_res != EI_IMPULSE_OK) {
265 return run_res;
266 }
267
268 return EI_IMPULSE_OK;
269}
