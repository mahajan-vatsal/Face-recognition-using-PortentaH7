The system captures facial images, processes them locally using an AI model deployed on the Portenta H7 and determines access based on authorised personnel.

80 int ret;

82 // input matrix from the raw signal
83 matrix_t input_matrix(signal->total_length / config.axes, config.axes);
84 if (!input_matrix.buffer) {

96 // transpose the matrix so we have one row per axis
97 numpy::transpose_in_place(&input_matrix);
98
99 size_t out_matrix_ix = 0;
100
101 for (size_t row = 0; row < input_matrix.rows; row++) {
102 matrix_t row_matrix(1, input_matrix.cols, input_matrix.buffer + (row * input_matrix.cols));
103
104 float mean; // to use with moving average
105
106 if (config.average || config.moving_avg_num_windows) {
107 float fbuffer;
108 matrix_t out_matrix(1, 1, &fbuffer);
109 numpy::mean(&row_matrix, &out_matrix);
110 mean = out_matrix.buffer[0];
111 if (config.average) {
112 output_matrix->buffer[out_matrix_ix++] = mean;
113 }
114 }
115
116 if (config.minimum) {
117 float fbuffer;
118 matrix_t out_matrix(1, 1, &fbuffer);
119 numpy::min(&row_matrix, &out_matrix);
120 output_matrix->buffer[out_matrix_ix++] = out_matrix.buffer[0];
121 }
122
123 if (config.maximum) {
124 float fbuffer;
125 matrix_t out_matrix(1, 1, &fbuffer);
126 numpy::max(&row_matrix, &out_matrix);
127 output_matrix->buffer[out_matrix_ix++] = out_matrix.buffer[0];
128 }
129
130 if (config.rms) {
131 float fbuffer;
132 matrix_t out_matrix(1, 1, &fbuffer);
133 numpy::rms(&row_matrix, &out_matrix);
134 output_matrix->buffer[out_matrix_ix++] = out_matrix.buffer[0];
135 }
136
137 if (config.stdev) {
138 float fbuffer;
139 matrix_t out_matrix(1, 1, &fbuffer);
140 numpy::stdev(&row_matrix, &out_matrix);
141 output_matrix->buffer[out_matrix_ix++] = out_matrix.buffer[0];
142 }
143
144 if (config.skewness) {
145 float fbuffer;
146 matrix_t out_matrix(1, 1, &fbuffer);
147 numpy::skew(&row_matrix, &out_matrix);
148 output_matrix->buffer[out_matrix_ix++] = out_matrix.buffer[0];
149 }
150
151 if (config.kurtosis) {
152 float fbuffer;
153 matrix_t out_matrix(1, 1, &fbuffer);
154 numpy::kurtosis(&row_matrix, &out_matrix);
155 output_matrix->buffer[out_matrix_ix++] = out_matrix.buffer[0];
156 }
157
158 if (config.moving_avg_num_windows) {
159 push_mean(row, mean);
160 output_matrix->buffer[out_matrix_ix++] = numpy::mean(means[row].data(), means[row].size());
161 }
162 }
163
164 // flatten again
165 output_matrix->cols = output_matrix->rows * output_matrix->cols;
166 output_matrix->rows = 1;
167
168 return EIDSP_OK;
169 }

188 flatten_class(int moving_avg_num_windows, int axes_count) : means(axes_count), head_indexes(axes_count, 0) {
189 this->moving_avg_num_windows = moving_avg_num_windows;
190 }
191
192 void push_mean(int axis, float mean) {
193 auto& head = head_indexes[axis];
194 if (head_indexes[axis] >= means[axis].size()) {
195 means[axis].push_back(mean);
196 } else {
197 means[axis][head] = mean;
198 }
199 head = head + 1;
200 // This is a lot cheaper than mod (%)
201 if (head >= moving_avg_num_windows) {
202 head = 0;
203 }
204 }
