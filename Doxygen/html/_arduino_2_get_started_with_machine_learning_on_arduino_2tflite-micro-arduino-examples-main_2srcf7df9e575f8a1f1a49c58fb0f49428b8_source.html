Machine Vision using Portenta H7 2
This project aims to develop a face recognition-based access control system using the Arduino Portenta H7 and Vision Shield, leveraging Edge Impulse for machine learning. The system captures facial images, processes them locally using an AI model deployed on the Portenta H7 and determines access based on authorised personnel.

52// Note that new error status values may be added in future in order to
53// indicate more fine-grained internal states, therefore, applications should
54// not rely on status values being members of the enum.
55typedef enum TfLiteStatus {
56 kTfLiteOk = 0,
57
58 // Generally referring to an error in the runtime (i.e. interpreter)
59 kTfLiteError = 1,
60
61 // Generally referring to an error from a TfLiteDelegate itself.
62 kTfLiteDelegateError = 2,
63
64 // Generally referring to an error in applying a delegate due to
65 // incompatibility between runtime and delegate, e.g., this error is returned
66 // when trying to apply a TF Lite delegate onto a model graph that's already
67 // immutable.
68 kTfLiteApplicationError = 3,
69
70 // Generally referring to serialized delegate data not being found.
71 // See tflite::delegates::Serialization.
72 kTfLiteDelegateDataNotFound = 4,
73
74 // Generally referring to data-writing issues in delegate serialization.
75 // See tflite::delegates::Serialization.
76 kTfLiteDelegateDataWriteError = 5,
77
78 // Generally referring to data-reading issues in delegate serialization.
79 // See tflite::delegates::Serialization.
80 kTfLiteDelegateDataReadError = 6,
81
82 // Generally referring to issues when the TF Lite model has ops that cannot be
83 // resolved at runtime. This could happen when the specific op is not
84 // registered or built with the TF Lite framework.
85 kTfLiteUnresolvedOps = 7,
86
87 // Generally referring to invocation cancelled by the user.
88 // See `interpreter::Cancel`.

90 // TODO(b/250636993): Cancellation triggered by `SetCancellationFunction`
91 // should also return this status code.
92 kTfLiteCancelled = 8,
