55 preemphasis(ei_signal_t *signal, int shift, float cof, bool rescale)
56 : _signal(signal), _shift(shift), _cof(cof), _rescale(rescale)
57 {
58 _prev_buffer = (float*)ei_dsp_calloc(shift * sizeof(float), 1);
59 _end_of_signal_buffer = (float*)ei_dsp_calloc(shift * sizeof(float), 1);
60 _next_offset_should_be = 0;
61
62 if (shift < 0) {
63 _shift = signal->total_length + shift;
64 }
65
66 if (!_prev_buffer || !_end_of_signal_buffer) return;
67
68 // we need to get the shift bytes from the end of the buffer...
69 signal->get_data(signal->total_length - shift, shift, _end_of_signal_buffer);
70 }

78 int get_data(size_t offset, size_t length, float *out_buffer) {
79 if (!_prev_buffer || !_end_of_signal_buffer) {
80 EIDSP_ERR(EIDSP_OUT_OF_MEM);
81 }
82 if (offset + length > _signal->total_length) {
83 EIDSP_ERR(EIDSP_OUT_OF_BOUNDS);
84 }
85
86 int ret;
87 if (static_cast<int32_t>(offset) - _shift >= 0) {
88 ret = _signal->get_data(offset - _shift, _shift, _prev_buffer);
89 if (ret != 0) {
90 EIDSP_ERR(ret);
91 }
92 }
93 // else we'll use the end_of_signal_buffer; so no need to check
94
95 ret = _signal->get_data(offset, length, out_buffer);
96 if (ret != 0) {
97 EIDSP_ERR(ret);
98 }
99
100 // now we have the signal and we can preemphasize
101 for (size_t ix = 0; ix < length; ix++) {
102 float now = out_buffer[ix];
103
104 // under shift? read from end
105 if (offset + ix < static_cast<uint32_t>(_shift)) {
106 out_buffer[ix] = now - (_cof * _end_of_signal_buffer[offset + ix]);
107 }
108 // otherwise read from history buffer
109 else {
110 out_buffer[ix] = now - (_cof * _prev_buffer[0]);
111 }
112
113 // roll through and overwrite last element
114 if (_shift != 1) {
115 numpy::roll(_prev_buffer, _shift, -1);
116 }
117 _prev_buffer[_shift - 1] = now;
118 }
119
120 _next_offset_should_be += length;
121
122 // rescale from [-1 .. 1] ?
123 if (_rescale) {
124 matrix_t scale_matrix(length, 1, out_buffer);
125 ret = numpy::scale(&scale_matrix, 1.0f / 32768.0f);
126 if (ret != 0) {
127 EIDSP_ERR(ret);
128 }
129 }
130
131 return EIDSP_OK;
132 }
133
134 ~preemphasis() {
135 if (_prev_buffer) {
136 ei_dsp_free(_prev_buffer, _shift * sizeof(float));
137 }
138 if (_end_of_signal_buffer) {
139 ei_dsp_free(_end_of_signal_buffer, _shift * sizeof(float));
140 }
141 }

143private:
144 ei_signal_t *_signal;
145 int _shift;
146 float _cof;
147 float *_prev_buffer;
148 float *_end_of_signal_buffer;
149 size_t _next_offset_should_be;
150 bool _rescale;
151 };

164 __attribute__((unused)) static int preemphasis(float *signal, size_t signal_size, int shift = 1, float cof = 0.98f)
165 {
166 if (shift < 0) {
167 shift = signal_size + shift;
168 }
169
170 // so we need to keep some history
171 float *prev_buffer = (float*)ei_dsp_calloc(shift * sizeof(float), 1);
172
173 // signal - cof * xt::roll(signal, shift)
174 for (size_t ix = 0; ix < signal_size; ix++) {
175 float now = signal[ix];
176
177 // under shift? read from end
178 if (ix < static_cast<uint32_t>(shift)) {
179 signal[ix] = now - (cof * signal[signal_size - shift + ix]);
180 }
181 // otherwise read from history buffer
182 else {
183 signal[ix] = now - (cof * prev_buffer[0]);
184 }
185
186 // roll through and overwrite last element
187 numpy::roll(prev_buffer, shift, -1);
188 prev_buffer[shift - 1] = now;
189 }
190
191 ei_dsp_free(prev_buffer, shift * sizeof(float));
192
193 return EIDSP_OK;
194 }

220 size_t signal_size,
221 uint32_t sampling_frequency,
222 float frame_length,
223 float frame_stride,
224 bool zero_padding,
225 uint16_t version)
226 {
227 int frame_sample_length;
228 int length;
229 if (version == 1) {

231 frame_stride = round(static_cast<float>(sampling_frequency) * frame_stride);
232 length = frame_sample_length;
233 }
234 else {
235 frame_sample_length = static_cast<int>(ceil_unless_very_close_to_floor(static_cast<float>(sampling_frequency) * frame_length));
236 float frame_stride_arg = frame_stride;
237 frame_stride = ceil_unless_very_close_to_floor(static_cast<float>(sampling_frequency) * frame_stride_arg);
238 length = (frame_sample_length - (int)frame_stride);

248
249 // Zero padding

251 }

253 numframes = static_cast<int>(

255 len_sig = static_cast<int>(

273 static int stack_frames(stack_frames_info_t *info,
274 float sampling_frequency,
275 float frame_length,
276 float frame_stride,
277 bool zero_padding,
278 uint16_t version)
279 {
280 if (!info->signal || !info->signal->get_data || info->signal->total_length == 0) {
281 EIDSP_ERR(EIDSP_SIGNAL_SIZE_MISMATCH);

283
284 size_t length_signal = info->signal->total_length;
285 int frame_sample_length;
286 int length;
287 if (version == 1) {
288 frame_sample_length = static_cast<int>(round(static_cast<float>(sampling_frequency) * frame_length));
289 frame_stride = round(static_cast<float>(sampling_frequency) * frame_stride);
290 length = frame_sample_length;
291 }
292 else {
293 frame_sample_length = static_cast<int>(ceil_unless_very_close_to_floor(static_cast<float>(sampling_frequency) * frame_length));
294 float frame_stride_arg = frame_stride;
295 frame_stride = ceil_unless_very_close_to_floor(static_cast<float>(sampling_frequency) * frame_stride_arg);
296 length = (frame_sample_length - (int)frame_stride);
297 }
298
299 volatile int numframes;
300 volatile int len_sig;
301
302 if (zero_padding) {
303 // Calculation of number of frames
304 numframes = static_cast<int>(
305 ceil(static_cast<float>(length_signal - length) / frame_stride));
306
307 // Zero padding
308 len_sig = static_cast<int>(static_cast<float>(numframes) * frame_stride) + frame_sample_length;
309
310 info->signal->total_length = static_cast<size_t>(len_sig);
311 }
312 else {
313 numframes = static_cast<int>(
314 floor(static_cast<float>(length_signal - length) / frame_stride));
315 len_sig = static_cast<int>(
316 (static_cast<float>(numframes - 1) * frame_stride + frame_sample_length));
317
318 info->signal->total_length = static_cast<size_t>(len_sig);
319 }
320
321 info->frame_ixs.clear();
322 info->frame_ixs.reserve(numframes); //limit the memory allocation
323
324 int frame_count = 0;
325
326 for (size_t ix = 0; ix < static_cast<uint32_t>(len_sig); ix += static_cast<size_t>(frame_stride)) {
327 if (++frame_count > numframes) break;
328
329 info->frame_ixs.push_back(ix);
330 }
331
332 info->frame_length = frame_sample_length;
333
334 return EIDSP_OK;
335 }

400 static int cmvnw(matrix_t *features_matrix, uint16_t win_size = 301, bool variance_normalization = false,
401 bool scale = false)
402 {
403 if (win_size == 0) {
404 return EIDSP_OK;
405 }
406
407 uint16_t pad_size = (win_size - 1) / 2;
408
409 int ret;
410 float *features_buffer_ptr;
411
412 // mean & variance normalization
413 EI_DSP_MATRIX(vec_pad, features_matrix->rows + (pad_size * 2), features_matrix->cols);
414 if (!vec_pad.buffer) {
415 EIDSP_ERR(EIDSP_OUT_OF_MEM);
416 }
417
418 ret = numpy::pad_1d_symmetric(features_matrix, &vec_pad, pad_size, pad_size);
419 if (ret != EIDSP_OK) {
420 EIDSP_ERR(ret);
421 }
422
423 EI_DSP_MATRIX(mean_matrix, vec_pad.cols, 1);
424 if (!mean_matrix.buffer) {
425 EIDSP_ERR(EIDSP_OUT_OF_MEM);
426 }
427
428 EI_DSP_MATRIX(window_variance, vec_pad.cols, 1);
429 if (!window_variance.buffer) {
430 return EIDSP_OUT_OF_MEM;
431 }
432
433 for (size_t ix = 0; ix < features_matrix->rows; ix++) {
434 // create a slice on the vec_pad
435 EI_DSP_MATRIX_B(window, win_size, vec_pad.cols, vec_pad.buffer + (ix * vec_pad.cols));
436 if (!window.buffer) {
437 EIDSP_ERR(EIDSP_OUT_OF_MEM);
438 }
439
440 ret = numpy::mean_axis0(&window, &mean_matrix);
441 if (ret != EIDSP_OK) {
442 EIDSP_ERR(ret);
443 }
444
445 // subtract the mean for the features
446 for (size_t fm_col = 0; fm_col < features_matrix->cols; fm_col++) {
447 features_matrix->buffer[(ix * features_matrix->cols) + fm_col] =
448 features_matrix->buffer[(ix * features_matrix->cols) + fm_col] - mean_matrix.buffer[fm_col];
449 }
450 }
451
452 ret = numpy::pad_1d_symmetric(features_matrix, &vec_pad, pad_size, pad_size);
453 if (ret != EIDSP_OK) {
454 EIDSP_ERR(ret);
455 }
456
457 for (size_t ix = 0; ix < features_matrix->rows; ix++) {
458 // create a slice on the vec_pad
459 EI_DSP_MATRIX_B(window, win_size, vec_pad.cols, vec_pad.buffer + (ix * vec_pad.cols));
460 if (!window.buffer) {
461 EIDSP_ERR(EIDSP_OUT_OF_MEM);
462 }
463
464 if (variance_normalization == true) {
465 ret = numpy::std_axis0(&window, &window_variance);
466 if (ret != EIDSP_OK) {
467 EIDSP_ERR(ret);
468 }
469
470 features_buffer_ptr = &features_matrix->buffer[ix * vec_pad.cols];
471 for (size_t col = 0; col < vec_pad.cols; col++) {
472 *(features_buffer_ptr) = (*(features_buffer_ptr)) /
473 (window_variance.buffer[col] + 1e-10);
474 features_buffer_ptr++;
475 }
476 }
477 }
478
479 if (scale) {
480 ret = numpy::normalize(features_matrix);
481 if (ret != EIDSP_OK) {
482 EIDSP_ERR(ret);
483 }
484 }
485
486 return EIDSP_OK;
487 }

494 static int mfe_normalization(matrix_t *features_matrix, int noise_floor_db) {
495 const float noise = static_cast<float>(noise_floor_db * -1);
496 const float noise_scale = 1.0f / (static_cast<float>(noise_floor_db * -1) + 12.0f);
497
498 for (size_t ix = 0; ix < features_matrix->rows * features_matrix->cols; ix++) {
499 float f = features_matrix->buffer[ix];
500 if (f < 1e-30) {
501 f = 1e-30;
502 }
503 f = numpy::log10(f);
504 f *= 10.0f; // scale by 10
505 f += noise;
506 f *= noise_scale;
507 // clip again
508
509 /* Here is the python code we're duplicating:
510 # Quantize to 8 bits and dequantize back to float32
511 mfe = np.uint8(np.around(mfe * 2**8))
512 # clip to 2**8
513 mfe = np.clip(mfe, 0, 255)
514 mfe = np.float32(mfe / 2**8)
515 */
516
517 f = roundf(f*256)/256;
518
519 if (f < 0.0f) f = 0.0f;
520 else if (f > 1.0f) f = 1.0f;
521 features_matrix->buffer[ix] = f;
522 }
523
524 return EIDSP_OK;
525 }
