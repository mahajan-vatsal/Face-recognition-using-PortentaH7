826__STATIC_FORCEINLINE q31_t arm_nn_doubling_high_mult(const q31_t m1, const q31_t m2)
827{
828 q31_t result = 0;
829 // Rounding offset to add for a right shift of 31
830 q63_t mult = 1 << 30;
831
832 if ((m1 < 0) ^ (m2 < 0))
833 {
834 mult = 1 - mult;
835 }
836 // Gets resolved as a SMLAL instruction
837 mult = mult + (q63_t)m1 * m2;
838
839 // Utilize all of the upper 32 bits. This is the doubling step
840 // as well.
841 result = (int32_t)(mult / (1ll << 31));
842
843 if ((m1 == m2) && (m1 == (int32_t)NN_Q31_MIN))
844 {
845 result = NN_Q31_MAX;
846 }
847 return result;
848}

864__STATIC_FORCEINLINE q31_t arm_nn_doubling_high_mult_no_sat(const q31_t m1, const q31_t m2)
865{
866 q31_t result = 0;
867 union arm_nn_long_long mult;
868
869 // Rounding offset to add for a right shift of 31
870 mult.word.low = 1 << 30;
871 mult.word.high = 0;
872
873 // Gets resolved as a SMLAL instruction
874 mult.long_long = mult.long_long + (q63_t)m1 * m2;
875
876 // Utilize all of the upper 32 bits. This is the doubling step
877 // as well.
878 result = (int32_t)(mult.long_long >> 31);
879
880 return result;
881}

891__STATIC_FORCEINLINE q31_t arm_nn_divide_by_power_of_two(const q31_t dividend, const q31_t exponent)
892{
893 q31_t result = 0;
894 const q31_t remainder_mask = (1 << exponent) - 1;
895 int32_t remainder = remainder_mask & dividend;
896
897 // Basic division
898 result = dividend >> exponent;
899
900 // Adjust 'result' for rounding (mid point away from zero)
901 q31_t threshold = remainder_mask >> 1;
902 if (result < 0)
903 {
904 threshold++;
905 }
906 if (remainder > threshold)
907 {
908 result++;
909 }
910
911 return result;
912}

923__STATIC_FORCEINLINE q31_t arm_nn_requantize(const q31_t val, const q31_t multiplier, const q31_t shift)
924{
925#ifdef CMSIS_NN_USE_SINGLE_ROUNDING
926 const int64_t total_shift = 31 - shift;
927 const int64_t new_val = val * (int64_t)multiplier;
928
929 int32_t result = new_val >> (total_shift - 1);
930 result = (result + 1) >> 1;
931
932 return result;
933#else
934 return arm_nn_divide_by_power_of_two(arm_nn_doubling_high_mult_no_sat(val * (1 << LEFT_SHIFT(shift)), multiplier),
935 RIGHT_SHIFT(shift));
936#endif
937}

1065__STATIC_FORCEINLINE int32x4_t arm_requantize_mve_32x4(const int32x4_t val,
1066 const int32x4_t multiplier,
1067 const int32x4_t shift)
1068{
1069#ifdef CMSIS_NN_USE_SINGLE_ROUNDING
1070 const int32x4_t right_shift = vminq_s32(vdupq_n_s32(-1), shift);
1071 const int32x4_t left_shift = vqsubq_s32(shift, right_shift);
1072
1073 int32x4_t result = vqdmulhq_s32(vshlq_s32(val, left_shift), multiplier);
1074 result = vrshlq_s32(result, right_shift);
1075
1076 return result;
1077#else
1078 const int32x4_t zz = vdupq_n_s32(0);
1079 const mve_pred16_t p = vcmpgtq_n_s32(shift, 0);
1080
1081 const int32x4_t left_shift = vpselq_s32(shift, zz, p);
