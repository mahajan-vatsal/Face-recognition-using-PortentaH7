Machine Vision using Portenta H7 2
This project aims to develop a face recognition-based access control system using the Arduino Portenta H7 and Vision Shield, leveraging Edge Impulse for machine learning. The system captures facial images, processes them locally using an AI model deployed on the Portenta H7 and determines access based on authorised personnel.

47void standard_scaler(float *input, const float *scale, const float *mean, size_t input_size) {
48 for (size_t ix = 0; ix < input_size; ix++) {
49 input[ix] = (input[ix] - mean[ix]) / scale[ix];
50 }
51}

59float calculate_cluster_distance(float *input, size_t input_size, const ei_classifier_anom_cluster_t *cluster) {
60 // todo: check input_size and centroid size?
61
62 float dist = 0.0f;
63 for (size_t ix = 0; ix < input_size; ix++) {
64 dist += pow(input[ix] - cluster->centroid[ix], 2);
65 }
66 return sqrt(dist) - cluster->max_error;
67}

76float get_min_distance_to_cluster(float *input, size_t input_size, const ei_classifier_anom_cluster_t *clusters, size_t cluster_size) {
77 float min = 1000.0f;
78 for (size_t ix = 0; ix < cluster_size; ix++) {
79 float dist = calculate_cluster_distance(input, input_size, &clusters[ix]);
80 if (dist < min) {
81 min = dist;
82 }
83 }
84 return min;
85}

101EI_IMPULSE_ERROR extract_anomaly_input_values(
102 ei_feature_t *fmatrix,
103 uint32_t* input_block_ids,
104 uint32_t input_block_ids_size,
105 uint32_t anom_axes_size,
106 const uint16_t *anom_axis,
107 float *input)
108{
109 if (input_block_ids_size == 1) {
110 for (size_t ix = 0; ix < anom_axes_size; ix++) {
111 input[ix] = fmatrix[0].matrix->buffer[anom_axis[ix]];
112 }
113 }
114 else {
115#if EI_CLASSIFIER_SINGLE_FEATURE_INPUT == 0
116 ei::matrix_t* matrix = NULL;
117#endif
118 // tracks where we are now in the combined feature matrix
119 uint32_t global_buf_pos = 0;
120 // we add the size of passed matrix to it
121 uint32_t buf_offset = 0;
122 // current index of input feature
123 uint32_t input_pos = 0;
124
125 for (size_t i = 0; i < input_block_ids_size; i++) {
126#if EI_CLASSIFIER_SINGLE_FEATURE_INPUT == 0
127 size_t cur_mtx = input_block_ids[i];
128 if (!find_mtx_by_idx(fmatrix, &matrix, cur_mtx, anom_axes_size)) {
129 ei_printf("ERR: Cannot find matrix with id %zu\n", cur_mtx);
130 return EI_IMPULSE_INVALID_SIZE;
131 }
132#else
133 ei::matrix_t* matrix = fmatrix[0].matrix;
134#endif
135 for (size_t ix = 0; ix < anom_axes_size; ix++) {
136 global_buf_pos = anom_axis[input_pos];
137 if (global_buf_pos <= buf_offset + (matrix->rows * matrix->cols)) {
138 input[input_pos] = matrix->buffer[anom_axis[input_pos] - buf_offset];
139 input_pos++;
140 if (input_pos >= anom_axes_size) { goto end; }
141 }
142 else {
143 break;
144 }
145 }
146 buf_offset += matrix->rows * matrix->cols;
147 }
148 end:;
149 }
150 return EI_IMPULSE_OK;
151}

154EI_IMPULSE_ERROR run_kmeans_anomaly(
155 const ei_impulse_t *impulse,
156 ei_feature_t *fmatrix,
157 uint32_t learn_block_index,
158 uint32_t* input_block_ids,
159 uint32_t input_block_ids_size,
160 ei_impulse_result_t *result,
161 void *config_ptr,
162 bool debug = false)
163{
164 ei_learning_block_config_anomaly_kmeans_t *block_config = (ei_learning_block_config_anomaly_kmeans_t*)config_ptr;
165
166 uint64_t anomaly_start_us = ei_read_timer_us();
167
168 float *input = (float*)ei_malloc(block_config->anom_axes_size * sizeof(float));
169 if (!input) {
170 ei_printf("Failed to allocate memory for anomaly input buffer");
171 return EI_IMPULSE_OUT_OF_MEMORY;
172 }
173
174 extract_anomaly_input_values(fmatrix, input_block_ids, input_block_ids_size, block_config->anom_axes_size, block_config->anom_axis, input);
175
176 standard_scaler(input, block_config->anom_scale, block_config->anom_mean, block_config->anom_axes_size);
177 float anomaly = get_min_distance_to_cluster(
178 input, block_config->anom_axes_size, block_config->anom_clusters, block_config->anom_cluster_count);
179
180 uint64_t anomaly_end_us = ei_read_timer_us();
181
182 if (debug) {
183 ei_printf("Anomaly score (time: %d ms.): ", static_cast<int>(anomaly_end_us - anomaly_start_us));
184 ei_printf_float(anomaly);
185 ei_printf("\n");
186 }
187
188 result->timing.anomaly_us = anomaly_end_us - anomaly_start_us;
189 result->timing.anomaly = (int)(result->timing.anomaly_us/1000);
190 result->anomaly = anomaly;
191 ei_free(input);
192
193 return EI_IMPULSE_OK;
194}
