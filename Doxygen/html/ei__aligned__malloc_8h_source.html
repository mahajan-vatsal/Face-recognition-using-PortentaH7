50__attribute__((unused)) void * ei_aligned_calloc(size_t align, size_t size)
51{
52 void * ptr = NULL;
53
54 //We want it to be a power of two since align_up operates on powers of two
55 assert((align & (align - 1)) == 0);
56
57 if(align && size)
58 {
59 /*
60 * We know we have to fit an offset value
61 * We also allocate extra bytes to ensure we can meet the alignment
62 */
63 uint32_t hdr_size = PTR_OFFSET_SZ + (align - 1);
64 void * p = ei_calloc(size + hdr_size, 1);
65
66 if(p)
67 {
68 /*
69 * Add the offset size to malloc's pointer (we will always store that)
70 * Then align the resulting value to the arget alignment
71 */
72 ptr = (void *) align_up(((uintptr_t)p + PTR_OFFSET_SZ), align);
73
74 //Calculate the offset and store it behind our aligned pointer
75 *((offset_t *)ptr - 1) = (offset_t)((uintptr_t)ptr - (uintptr_t)p);
76
77 } // else NULL, could not malloc
78 } //else NULL, invalid arguments
79
80 return ptr;
81}

88__attribute__((unused)) void ei_aligned_free(void * ptr)
89{
90 assert(ptr);
91
92 /*
93 * Walk backwards from the passed-in pointer to get the pointer offset
94 * We convert to an offset_t pointer and rely on pointer math to get the data
95 */
96 offset_t offset = *((offset_t *)ptr - 1);
97
98 /*
99 * Once we have the offset, we can get our original pointer and call free
100 */
101 void * p = (void *)((uint8_t *)ptr - offset);
102 ei_free(p);
103}
