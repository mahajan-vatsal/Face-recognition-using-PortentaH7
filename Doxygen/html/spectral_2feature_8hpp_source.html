68 static int spectral_analysis(
69 matrix_t *out_features,
70 matrix_t *input_matrix,
71 float sampling_freq,
72 filter_t filter_type,
73 float filter_cutoff,
74 uint8_t filter_order,
75 uint16_t fft_length,
76 uint8_t fft_peaks,
77 float fft_peaks_threshold,
78 matrix_t *edges_matrix_in
79 ) {
80 if (out_features->rows != input_matrix->rows) {
81 EIDSP_ERR(EIDSP_MATRIX_SIZE_MISMATCH);
82 }
83
84 if (out_features->cols != calculate_spectral_buffer_size(true, fft_peaks, edges_matrix_in->rows)) {
85 EIDSP_ERR(EIDSP_MATRIX_SIZE_MISMATCH);
86 }
87
88 if (edges_matrix_in->cols != 1) {
89 EIDSP_ERR(EIDSP_MATRIX_SIZE_MISMATCH);
90 }
91
92 int ret;
93
94 size_t axes = input_matrix->rows;
95
96 EI_TRY(processing::subtract_mean(input_matrix) );
97
98 // apply filter
99 if (filter_type == filter_lowpass) {
100 ret = spectral::processing::butterworth_lowpass_filter(
101 input_matrix, sampling_freq, filter_cutoff, filter_order);
102 if (ret != EIDSP_OK) {
103 EIDSP_ERR(EIDSP_MATRIX_SIZE_MISMATCH);
104 }
105 }
106 else if (filter_type == filter_highpass) {
107 ret = spectral::processing::butterworth_highpass_filter(
108 input_matrix, sampling_freq, filter_cutoff, filter_order);
109 if (ret != EIDSP_OK) {
110 EIDSP_ERR(EIDSP_MATRIX_SIZE_MISMATCH);
111 }
112 }
113
114 // calculate RMS
115 EI_DSP_MATRIX(rms_matrix, axes, 1);
116 ret = numpy::rms(input_matrix, &rms_matrix);
117 if (ret != EIDSP_OK) {
118 EIDSP_ERR(EIDSP_MATRIX_SIZE_MISMATCH);
119 }
120
121 // find peaks in FFT
122 EI_DSP_MATRIX(peaks_matrix, axes, fft_peaks * 2);
123
124 for (size_t row = 0; row < input_matrix->rows; row++) {
125 // per axis code
126
127 // get a slice of the current axis
128 EI_DSP_MATRIX_B(axis_matrix, 1, input_matrix->cols, input_matrix->buffer + (row * input_matrix->cols));
129
130 // calculate FFT
131 EI_DSP_MATRIX(fft_matrix, 1, fft_length / 2 + 1);
132 ret = numpy::rfft(axis_matrix.buffer, axis_matrix.cols, fft_matrix.buffer, fft_matrix.cols, fft_length);
133 if (ret != EIDSP_OK) {
134 EIDSP_ERR(EIDSP_MATRIX_SIZE_MISMATCH);
135 }
136
137 // multiply by 2/N
138 numpy::scale(&fft_matrix, (2.0f / static_cast<float>(fft_length)));
139
140 // we're now using the FFT matrix to calculate peaks etc.
141 EI_DSP_MATRIX(peaks_matrix, fft_peaks, 2);
142 ret = spectral::processing::find_fft_peaks(&fft_matrix, &peaks_matrix,
143 sampling_freq, fft_peaks_threshold, fft_length);
144 if (ret != EIDSP_OK) {
145 EIDSP_ERR(EIDSP_MATRIX_SIZE_MISMATCH);
146 }
147
148 // calculate periodogram for spectral power buckets
149 EI_DSP_MATRIX(period_fft_matrix, 1, fft_length / 2 + 1);
150 EI_DSP_MATRIX(period_freq_matrix, 1, fft_length / 2 + 1);
151 ret = spectral::processing::periodogram(&axis_matrix,
152 &period_fft_matrix, &period_freq_matrix, sampling_freq, fft_length);
153 if (ret != EIDSP_OK) {
154 EIDSP_ERR(ret);
155 }
156
157 EI_DSP_MATRIX(edges_matrix_out, edges_matrix_in->rows - 1, 1);
158 ret = spectral::processing::spectral_power_edges(
159 &period_fft_matrix,
160 &period_freq_matrix,
161 edges_matrix_in,
162 &edges_matrix_out,
163 sampling_freq);
164 if (ret != EIDSP_OK) {
165 EIDSP_ERR(ret);
166 }
167
168 float *features_row = out_features->buffer + (row * out_features->cols);
169
170 size_t fx = 0;
171
172 features_row[fx++] = rms_matrix.buffer[row];
173 for (size_t peak_row = 0; peak_row < peaks_matrix.rows; peak_row++) {
174 features_row[fx++] = peaks_matrix.buffer[peak_row * peaks_matrix.cols + 0];
175 features_row[fx++] = peaks_matrix.buffer[peak_row * peaks_matrix.cols + 1];
176 }
177 for (size_t edge_row = 0; edge_row < edges_matrix_out.rows; edge_row++) {
178 features_row[fx++] = edges_matrix_out.buffer[edge_row * edges_matrix_out.cols] / 10.0f;
179 }
180 }
181
182 return EIDSP_OK;
183 }

192 static size_t calculate_spectral_buffer_size(
193 bool rms, size_t peaks_count, size_t spectral_edges_count)
194 {
195 size_t count = 0;
196 if (rms) count++;
197 count += (peaks_count * 2);
198 if (spectral_edges_count > 0) {
199 count += (spectral_edges_count - 1);
200 }
201 return count;
202 }

204 static int extract_spectral_analysis_features_v1(
205 matrix_t *input_matrix,
206 matrix_t *output_matrix,
207 ei_dsp_config_spectral_analysis_t *config_ptr,
208 const float sampling_freq)
209 {
210 // scale the signal
211 int ret = numpy::scale(input_matrix, config_ptr->scale_axes);
212 if (ret != EIDSP_OK) {
213 ei_printf("ERR: Failed to scale signal (%d)\n", ret);
214 EIDSP_ERR(ret);
215 }
216
217 // transpose the matrix so we have one row per axis (nifty!)
218 ret = numpy::transpose(input_matrix);
219 if (ret != EIDSP_OK) {
220 ei_printf("ERR: Failed to transpose matrix (%d)\n", ret);
221 EIDSP_ERR(ret);
222 }
223
224 // the spectral edges that we want to calculate
225 matrix_t edges_matrix_in(64, 1);
226 size_t edge_matrix_ix = 0;
227
228 char spectral_str[128] = { 0 };
229 if (strlen(config_ptr->spectral_power_edges) > sizeof(spectral_str) - 1) {
230 EIDSP_ERR(EIDSP_PARAMETER_INVALID);
231 }
232 memcpy(
233 spectral_str,
234 config_ptr->spectral_power_edges,
235 strlen(config_ptr->spectral_power_edges));
236
237 // convert spectral_power_edges (string) into float array
238 char *spectral_ptr = spectral_str;
239 while (spectral_ptr != NULL) {
240 while ((*spectral_ptr) == ' ') {
241 spectral_ptr++;
242 }
243
244 edges_matrix_in.buffer[edge_matrix_ix++] = atof(spectral_ptr);
245
246 // find next (spectral) delimiter (or '\0' character)
247 while ((*spectral_ptr != ',')) {
248 spectral_ptr++;
249 if (*spectral_ptr == '\0')
250 break;
251 }
252
253 if (*spectral_ptr == '\0') {
254 spectral_ptr = NULL;
255 }
256 else {
257 spectral_ptr++;
258 }
259 }
260 edges_matrix_in.rows = edge_matrix_ix;
261
262 // calculate how much room we need for the output matrix
263 size_t output_matrix_cols = spectral::feature::calculate_spectral_buffer_size(
264 true,
265 config_ptr->spectral_peaks_count,
266 edges_matrix_in.rows);
267 // ei_printf("output_matrix_size %hux%zu\n", input_matrix.rows, output_matrix_cols);
268 if (output_matrix->cols * output_matrix->rows !=
269 static_cast<uint32_t>(output_matrix_cols * config_ptr->axes)) {
270 EIDSP_ERR(EIDSP_MATRIX_SIZE_MISMATCH);
271 }
272
273 output_matrix->cols = output_matrix_cols;
274 output_matrix->rows = config_ptr->axes;
275
276 spectral::filter_t filter_type;
277 if (strcmp(config_ptr->filter_type, "low") == 0) {
278 filter_type = spectral::filter_lowpass;
279 }
280 else if (strcmp(config_ptr->filter_type, "high") == 0) {
281 filter_type = spectral::filter_highpass;
282 }
283 else {
284 filter_type = spectral::filter_none;
285 }
286
287 ret = spectral::feature::spectral_analysis(
288 output_matrix,
289 input_matrix,
290 sampling_freq,
291 filter_type,
292 config_ptr->filter_cutoff,
293 config_ptr->filter_order,
294 config_ptr->fft_length,
295 config_ptr->spectral_peaks_count,
296 config_ptr->spectral_peaks_threshold,
297 &edges_matrix_in);
298 if (ret != EIDSP_OK) {
299 ei_printf("ERR: Failed to calculate spectral features (%d)\n", ret);
300 EIDSP_ERR(ret);
301 }
302
303 // flatten again
304 output_matrix->cols = config_ptr->axes * output_matrix_cols;
305 output_matrix->rows = 1;
306
307 return EIDSP_OK;
308 }

310 static void get_start_stop_bin(
311 float sampling_freq,
312 size_t fft_length,
313 float filter_cutoff,
314 size_t *start_bin,
315 size_t *stop_bin,
316 bool is_high_pass)
317 {
318 // we want to find n such that fcutoff < sample_f / fft * n ( or > for high pass )
319 // also, + - half bin width (sample_f/(fft*2)) for high / low pass
320 if (filter_cutoff > sampling_freq / 2) {
321 filter_cutoff = sampling_freq / 2;
322 }
323 float bin = filter_cutoff * fft_length / sampling_freq;
324 if (is_high_pass) {
325 *start_bin = static_cast<size_t>(bin - 0.5) + 1; // add one b/c we want to always round up
326 // don't use the DC bin b/c it's zero
327 *start_bin = *start_bin == 0 ? 1 : *start_bin;
328 *stop_bin = fft_length / 2 + 1; // go one past
329 }
330 else {
331 *start_bin = 1;
332 *stop_bin = static_cast<size_t>(bin + 0.5) + 1; // go one past
333 }
334 }

341 static size_t extract_spec_features(
342 matrix_t *input_matrix,
343 matrix_t *output_matrix,
344 ei_dsp_config_spectral_analysis_t *config,
345 const float sampling_freq,
346 const bool remove_mean = true,
347 const bool transpose_and_scale_input = true)
348 {
349 if (transpose_and_scale_input) {
350 // transpose the matrix so we have one row per axis
351 numpy::transpose_in_place(input_matrix);
352
353 // func tests for scale of 1 and does a no op in that case
354 EI_TRY(numpy::scale(input_matrix, config->scale_axes));
355 }
356
357 bool do_filter = false;
358 bool is_high_pass;
359
360 // apply filter, if enabled
361 // "zero" order filter allowed. will still remove unwanted fft bins later
362 if (strcmp(config->filter_type, "low") == 0) {
363 if( config->filter_order ) {
364 EI_TRY(spectral::processing::butterworth_lowpass_filter(
365 input_matrix,
366 sampling_freq,
367 config->filter_cutoff,
368 config->filter_order));
369 }
370 do_filter = true;
371 is_high_pass = false;
372 }
373 else if (strcmp(config->filter_type, "high") == 0) {
374 if( config->filter_order ) {
375 EI_TRY(spectral::processing::butterworth_highpass_filter(
376 input_matrix,
377 sampling_freq,
378 config->filter_cutoff,
379 config->filter_order));
380 }
381 do_filter = true;
382 is_high_pass = true;
383 }
384
385 if (remove_mean){
386 EI_TRY(processing::subtract_mean(input_matrix));
387 }
388
389 // Figure bins we remove based on filter cutoff
390 size_t start_bin, stop_bin;
391 if (do_filter) {
392 get_start_stop_bin(
393 sampling_freq,
394 config->fft_length,
395 config->filter_cutoff,
396 &start_bin,
397 &stop_bin,
398 is_high_pass);
399 }
400 else {
401 start_bin = 1;
402 stop_bin = config->fft_length / 2 + 1;
403 }
404 size_t num_bins = stop_bin - start_bin;
405
406 float *feature_out = output_matrix->buffer;
407 const float *feature_out_ori = feature_out;
408 for (size_t row = 0; row < input_matrix->rows; row++) {
409 float *data_window = input_matrix->get_row_ptr(row);
410 size_t data_size = input_matrix->cols;
411
412 matrix_t rms_in_matrix(1, data_size, data_window);
413 matrix_t rms_out_matrix(1, 1, feature_out);
414 EI_TRY(numpy::rms(&rms_in_matrix, &rms_out_matrix));
415
416 feature_out++;
417
418 // Standard Deviation
419 float stddev = *(feature_out-1); //= sqrt(numpy::variance(data_window, data_size));
420 if (stddev == 0.0f) {
421 stddev = 1e-10f;
422 }
423 // Don't add std dev as a feature b/c it's the same as RMS
424 // Skew and Kurtosis w/ shortcut:
425 // See definition at https://en.wikipedia.org/wiki/Skewness
426 // See definition at https://en.wikipedia.org/wiki/Kurtosis
427 // Substitute 0 for mean (b/c it is subtracted out above)
428 // Skew becomes: mean(X^3) / stddev^3
429 // Kurtosis becomes: mean(X^4) / stddev^4
430 // Note, this is the Fisher definition of Kurtosis, so subtract 3
431 // (see https://docs.scipy.org/doc/scipy/reference/generated/scipy.stats.kurtosis.html)
432 float s_sum = 0;
433 float k_sum = 0;
434 float temp;
435 for (size_t i = 0; i < data_size; i++) {
436 temp = data_window[i] * data_window[i] * data_window[i];
437 s_sum += temp;
438 k_sum += temp * data_window[i];
439 }
440 // Skewness out
441 temp = stddev * stddev * stddev;
442 *feature_out++ = (s_sum / data_size) / temp;
443 // Kurtosis out
444 *feature_out++ = ((k_sum / data_size) / (temp * stddev)) - 3;
445
446 if (config->implementation_version == 4) {
447
448 size_t fft_out_size = config->fft_length / 2 + 1;
449 ei_vector<float> fft_out(fft_out_size);
450 EI_TRY(numpy::welch_max_hold(
451 data_window,
452 data_size,
453 fft_out.data(),
454 0,
455 fft_out_size,
456 config->fft_length,
457 config->do_fft_overlap));
458
459 matrix_t x(1, fft_out.size(), const_cast<float *>(fft_out.data()));
460 matrix_t out(1, 1);
461
462 *feature_out++ = (numpy::skew(&x, &out) == EIDSP_OK) ? (out.get_row_ptr(0)[0]) : 0.0f;
463 *feature_out++ = (numpy::kurtosis(&x, &out) == EIDSP_OK) ? (out.get_row_ptr(0)[0]) : 0.0f;
464
465 for (size_t i = start_bin; i < stop_bin; i++) {
466 feature_out[i - start_bin] = fft_out[i];
467 }
468 } else {
469 EI_TRY(numpy::welch_max_hold(
470 data_window,
471 data_size,
472 feature_out,
473 start_bin,
474 stop_bin,
475 config->fft_length,
476 config->do_fft_overlap));
477 }
478 if (config->do_log) {
479 numpy::zero_handling(feature_out, num_bins);
480 ei_matrix temp(num_bins, 1, feature_out);
481 numpy::log10(&temp);
482 }
483 feature_out += num_bins;
484 }
485 size_t num_features = feature_out - feature_out_ori;
486 return num_features;
487 }

591 static int extract_spectral_analysis_features_v4(
592 matrix_t *input_matrix,
593 matrix_t *output_matrix,
594 ei_dsp_config_spectral_analysis_t *config_p,
595 const float sampling_freq)
596 {
597 auto config_copy = *config_p;
598 auto config = &config_copy;
599 if (strcmp(config->analysis_type, "Wavelet") == 0) {
600 return wavelet::extract_wavelet_features(input_matrix, output_matrix, config, sampling_freq);
601 }
602 else if (config->extra_low_freq == false && config->input_decimation_ratio == 1) {
603 size_t n_features =
604 extract_spec_features(input_matrix, output_matrix, config, sampling_freq);
605 return n_features == output_matrix->cols ? EIDSP_OK : EIDSP_MATRIX_SIZE_MISMATCH;
606 }
607 else {
608 numpy::transpose_in_place(input_matrix);
609 EI_TRY(numpy::scale(input_matrix, config->scale_axes));
610
611 if (config->input_decimation_ratio > 1) {
612 ei_vector<int> ratio_combo = get_ratio_combo(config->input_decimation_ratio);
613 size_t out_size = input_matrix->cols;
614 for (int r : ratio_combo) {
615 out_size = _decimate(input_matrix, input_matrix, r);
616 }
617
618 // rearrange input matrix to be in the right shape after decimation
619 float* out = input_matrix->get_row_ptr(0) + out_size;
620 for(uint32_t r = 1; r < input_matrix->rows; r++) {
621 float *row = input_matrix->get_row_ptr(r);
622 for(size_t c = 0; c < out_size; c++) {
623 *out++ = row[c];
624 }
625 }
626 input_matrix->cols = out_size;
627 }
628
629 float new_sampling_freq = sampling_freq / config->input_decimation_ratio;
630
631 // filter here, before decimating, instead of inside extract_spec_features
632 if (strcmp(config->filter_type, "low") == 0) {
633 if( config->filter_order ) {
634 EI_TRY(spectral::processing::butterworth_lowpass_filter(
635 input_matrix,
636 new_sampling_freq,
637 config->filter_cutoff,
638 config->filter_order));
639 }
640 }
641 else if (strcmp(config->filter_type, "high") == 0) {
642 if( config->filter_order ) {
643 EI_TRY(spectral::processing::butterworth_highpass_filter(
644 input_matrix,
