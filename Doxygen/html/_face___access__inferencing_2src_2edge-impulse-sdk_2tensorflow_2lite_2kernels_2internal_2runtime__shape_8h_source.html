26class RuntimeShape {
27 public:
28 RuntimeShape& operator=(RuntimeShape const&) = delete;

30 // RuntimeShape in TFLM supports up to 5 dimensions.
31 // The name kMaxSmallSize comes from the same file of the upstream
32 // tensorflow lite repo and need to be kept the same for max reuse.
33 static constexpr int kMaxSmallSize = 5;
34
35 RuntimeShape() : size_(0) {}
36
37 explicit RuntimeShape(int dimensions_count) : size_(dimensions_count) {}
38
39 RuntimeShape(int shape_size, int32_t value) : size_(shape_size) {
40 for (int i = 0; i < shape_size; ++i) {
41 SetDim(i, value);
42 }
43 }

45 RuntimeShape(int dimensions_count, const int32_t* dims_data)
46 : size_(dimensions_count) {
47 ReplaceWith(dimensions_count, dims_data);
48 }
49
50 bool operator==(const RuntimeShape& comp) const {
51 return this->size_ == comp.size_ &&
52 std::memcmp(DimsData(), comp.DimsData(), size_ * sizeof(int32_t)) ==
53 0;
54 }
55
56 ~RuntimeShape() {}
57
58 int32_t DimensionsCount() const { return size_; }
59 int32_t Dims(int i) const {
60 TFLITE_DCHECK_GE(i, 0);
61 TFLITE_DCHECK_LT(i, size_);
62 return dims_[i];
63 }
64 void SetDim(int i, int32_t val) {
65 TFLITE_DCHECK_GE(i, 0);
66 TFLITE_DCHECK_LT(i, size_);
67 dims_[i] = val;
68 }

70 static RuntimeShape ExtendedShape(int new_shape_size,
71 const RuntimeShape& shape) {
72 return RuntimeShape(new_shape_size, shape, 1);
73 }
74 int32_t* DimsData() { return dims_; }
75 const int32_t* DimsData() const { return dims_; }
76 const int32_t* DimsDataUpTo5D() const { return dims_; }

78 void ReplaceWith(int dimensions_count, const int32_t* dims_data) {
79 size_ = dimensions_count;
80 int32_t* dst_dims = DimsData();
81 std::memcpy(dst_dims, dims_data, dimensions_count * sizeof(int32_t));
82 }

84 // Returns the total count of elements, that is the size when flattened into a
85 // vector.
86 int FlatSize() const {
87 int buffer_size = 1;
88 const int* dims_data = reinterpret_cast<const int*>(DimsData());
89 for (int i = 0; i < size_; i++) {
90 buffer_size *= dims_data[i];
91 }
92 return buffer_size;
93 }
94
95 private:
96 // For use only by ExtendedShape(), written to guarantee (return-value) copy
97 // elision in C++17.
98 // This creates a shape padded to the desired size with the specified value.
99 RuntimeShape(int new_shape_size, const RuntimeShape& shape, int pad_value)
100 : size_(new_shape_size) {
101 // If the following check fails, it is likely because a 4D-only kernel is
102 // being used with an array of larger dimension count.
103 TFLITE_CHECK_GE(new_shape_size, shape.DimensionsCount());
104 const int size_increase = new_shape_size - shape.DimensionsCount();
105 for (int i = 0; i < size_increase; ++i) {
106 SetDim(i, pad_value);
107 }
108 std::memcpy(DimsData() + size_increase, shape.DimsData(),
109 sizeof(int32_t) * shape.DimensionsCount());
110 }
111
112 int32_t size_;
113 union {
114 int32_t dims_[kMaxSmallSize];
115 };
116};

118// Since tensors with '0' in their shape are valid in TF, these offset functions
119// allow that as long as the corresponding index is also 0. It is upto the
120// calling ops to ensure that they perform verification checks on tensor shapes
121// if they don't support a particular behavior.
122
123inline int Offset(const RuntimeShape& shape, int i0, int i1, int i2, int i3) {
124 TFLITE_DCHECK_EQ(shape.DimensionsCount(), 4);
125 const int* dims_data = reinterpret_cast<const int*>(shape.DimsData());
126 TFLITE_DCHECK((dims_data[0] == 0 && i0 == 0) ||
127 (i0 >= 0 && i0 < dims_data[0]));
128 TFLITE_DCHECK((dims_data[1] == 0 && i1 == 0) ||
129 (i1 >= 0 && i1 < dims_data[1]));
130 TFLITE_DCHECK((dims_data[2] == 0 && i2 == 0) ||
131 (i2 >= 0 && i2 < dims_data[2]));
132 TFLITE_DCHECK((dims_data[3] == 0 && i3 == 0) ||
133 (i3 >= 0 && i3 < dims_data[3]));
134 return ((i0 * dims_data[1] + i1) * dims_data[2] + i2) * dims_data[3] + i3;
135}
136
137inline int Offset(const RuntimeShape& shape, int i0, int i1, int i2, int i3,
138 int i4) {
139 TFLITE_DCHECK_EQ(shape.DimensionsCount(), 5);
140 const int* dims_data = reinterpret_cast<const int*>(shape.DimsData());
141 TFLITE_DCHECK((dims_data[0] == 0 && i0 == 0) ||
142 (i0 >= 0 && i0 < dims_data[0]));
143 TFLITE_DCHECK((dims_data[1] == 0 && i1 == 0) ||
144 (i1 >= 0 && i1 < dims_data[1]));
145 TFLITE_DCHECK((dims_data[2] == 0 && i2 == 0) ||
146 (i2 >= 0 && i2 < dims_data[2]));
147 TFLITE_DCHECK((dims_data[3] == 0 && i3 == 0) ||
148 (i3 >= 0 && i3 < dims_data[3]));
149 TFLITE_DCHECK((dims_data[4] == 0 && i4 == 0) ||
150 (i4 >= 0 && i4 < dims_data[4]));
151 return (((i0 * dims_data[1] + i1) * dims_data[2] + i2) * dims_data[3] + i3) *
152 dims_data[4] +
153 i4;
154}
