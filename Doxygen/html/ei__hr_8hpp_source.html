Machine Vision using Portenta H7 2
This project aims to develop a face recognition-based access control system using the Arduino Portenta H7 and Vision Shield, leveraging Edge Impulse for machine learning. The system captures facial images, processes them locally using an AI model deployed on the Portenta H7 and determines access based on authorised personnel.

67 int extract(
68 ei::signal_t *signal,
69 ei::matrix_t *output_matrix,
70 void *config_ptr,
71 const float frequency,
72 ei_impulse_result_t *result) override
73 {
74 using namespace ei;
75
76 // Using reference avoids a copy
77 ei_dsp_config_hr_t& config = *((ei_dsp_config_hr_t*)config_ptr);
78 size_t floats_per_inc = ppg.win_inc_samples * ppg.axes;
79 size_t hrv_inc_samples = config.hrv_update_interval_s * frequency * ppg.axes;
80 // Greater than b/c can have multiple increments (HRs) per window
81 assert(signal->total_length >= floats_per_inc);
82
83 int out_idx = 0;
84 size_t hrv_count = 0;
85 for (size_t i = 0; i <= signal->total_length - floats_per_inc; i += floats_per_inc) {
86 // TODO ask for smaller increments and bp them into place
87 // Copy into the end of the buffer
88 matrix_t temp(ppg.win_inc_samples, ppg.axes);
89 signal->get_data(i, floats_per_inc, temp.buffer);
90 auto hr = ppg.stream(&temp);
91 if (result) {
92 result->hr_calcs.heart_rate = hr;
93 }
94 if(!hrv || (hrv && config.include_hr)) {
95 output_matrix->buffer[out_idx++] = hr;
96 }
97 if(hrv) {
98 auto peaks = ppg.get_peaks();
99 hrv->add_streaming_beats(peaks);
100 hrv_count += floats_per_inc;
101 if(hrv_count >= hrv_inc_samples) {
102 fvec features = hrv->get_hrv_features(0);
103 for(size_t j = 0; j < features.size(); j++) {
104 output_matrix->buffer[out_idx++] = features[j];
105 }
106 hrv_count = 0;
107 }
108 }
109 }
110 return EIDSP_OK;
111 }
