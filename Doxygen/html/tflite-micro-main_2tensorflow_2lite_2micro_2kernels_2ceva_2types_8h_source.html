157 const int* index, const int num_axis,

159 if (num_dims == 0) {
160 return 0;
161 }
162 // TFLITE_DCHECK(dims != nullptr);
163 // TFLITE_DCHECK(index != nullptr);
164 size_t offset = 0;
165 for (int idx = 0; idx < num_dims; ++idx) {
166 // if we need to skip this axis
167 bool is_axis = false;
168 if (axis != nullptr) {
169 for (int axis_idx = 0; axis_idx < num_axis; ++axis_idx) {
170 if (idx == axis[axis_idx]) {
171 is_axis = true;
172 break;
173 }
174 }
175 }
176 if (!is_axis) {
177 offset = offset * static_cast<size_t>(dims[idx]) +
178 static_cast<size_t>(index[idx]);
179 }
180 }
181 return offset;
182}

228// This enumeration allows for non-default formats for the weights array
229// of a fully-connected operator, allowing the use of special optimized
230// runtime paths.
231enum class FullyConnectedWeightsFormat : uint8_t {
232 // Default format (flat 2D layout, the inner contiguous dimension
233 // is input_depth, the outer non-contiguous dimension is output_depth)
234 kDefault,
235 // Summary: optimized layout for fast CPU runtime implementation,
236 // aimed specifically at ARM CPUs at the moment, and specialized for
237 // 8-bit quantized layers.
238 //
239 // The use case we're concerned with here is: 8-bit quantization,
240 // large weights matrix that doesn't fit in cache (e.g. 4096x2048 in
241 // a key application that drove this), very small batch size (e.g. 1 -- 4).
242 //
243 // Even with 8-bit quantization of weights, the performance of memory
244 // accesses to the weights can become the dominant issue when
245 // the batch size is small, so each weight value is used in only a few
246 // arithmetic ops, i.e. the fully-connected node has a low arithmetic
247 // intensity. The specific issues that arise are of three kinds:
248 // (1) One may, ideally, max out DRAM bandwidth, i.e. be truly memory
249 // bound. That's the "good" issue to run into.
250 // (2) One may run into sub-optimal pre-fetching: the data hasn't been
251 // prefetched into the cache by the time we need it.
252 // (3) One may run into cache aliasing: multiple values that are
253 // pre-fetched, alias each other in the L1 cache (which typically
254 // has only 4-way set associativity in ARM CPUs) and thus evict
255 // each other before we get to using them.
256 //
257 // The point of this shuffling is to avoid issues (2) and (3) so that
258 // we get as fast as possible given only the hard constraint (1).
259 // This is achieved by turning the difficulty into a solution: the
260 // difficulty, that each value loaded from memory is used only in
261 // one kernel iteration, making this operation memory-intensive, hints at
262 // the solution, of shuffling the weights so that they are stored in the
263 // exact order as the kernel needs to load them, so that the memory
264 // accesses made by the kernel are trivial. This solves (2) because the
265 // trivial memory access pattern allows the CPU's automatic prefetching
266 // to perform very well (no need even for preload instructions), and this
267 // solves (3) because the values being loaded concurrently are now
268 // contiguous in the address space, thus don't alias each other in the cache.
269 //
270 // On ARM, we typically want our kernel to process a 4x16 block of weights
271 // at a time, because:
272 // - 16 is the number of bytes in a NEON register.
273 // - 4 is how many rows we need to handle concurrently in the kernel in
274 // order to have sufficient mutual independence of instructions to
275 // maximize arithmetic throughput.
276 //
277 // Finally, the 'int8_t' part in the name refers to the fact that this
278 // weights format has each weights value encoded as a signed int8_t value,
279 // even if the data type of the weights buffer is uint8_t. This is intended
280 // to save runtime kernels the effort to have to XOR the top bit of these
281 // bytes before using them in signed arithmetic, see this file for more
282 // explanations on the 'signed int8_t trick' in matrix multiplication kernels:
283 //
284 // tensorflow/lite/toco/graph_transformations/ensure_uint8_weights_safe_for_fast_int8_kernels.cc
285 //
286 kShuffled4x16Int8,
287};

289// Quantization parameters, determining the mapping of quantized values
290// to real values (i.e. determining how quantized values are mathematically
291// interpreted).
292//
293// The correspondence is as follows:
294//
295// real_value = scale * (quantized_value - zero_point);
296//
297// In other words, zero_point designates which quantized value corresponds to
298// the real 0 value, and scale designates the difference between the real values
299// corresponding to consecutive quantized values differing by 1.
300struct QuantizationParams {
301 int32_t zero_point = 0;
302 double scale = 0.0;
303};

306 const QuantizationParams& qp2) {
307 return qp1.zero_point == qp2.zero_point && qp1.scale == qp2.scale;

464 // For use only by ExtendedShape(), written to guarantee (return-value) copy

466 // This creates a shape padded to the desired size with the specified value.

488// Converts inference-style shape to legacy tflite::Dims<4>.
489inline tflite::Dims<4> ToRuntimeDims(const tflite::RuntimeShape& array_shape) {
490 tflite::Dims<4> result;
491 const int dimensions_count = array_shape.DimensionsCount();
492 TFLITE_CHECK_LE(dimensions_count, 4);
493 int cum_prod = 1;
494 for (int i = 0; i < 4; i++) {
495 const int new_dim =
496 (i < dimensions_count) ? array_shape.Dims(dimensions_count - 1 - i) : 1;
497 result.sizes[i] = new_dim;
498 result.strides[i] = cum_prod;
499 cum_prod *= new_dim;
500 }
501 return result;
502}

532// Gets offset of index if reducing on axis. When reducing, the flattened offset
533// will not change, if the input index changes on the given axis. For example,
534// if you have a 3D tensor and you are reducing to 2D by eliminating axis 0,
535// then index (0, 1, 2) and index (1, 1, 2) will map to the same flattened
536// offset.
537// TODO(kanlig): uses Dims to represent dimensions.
538inline size_t ReducedOutputOffset(const int num_dims, const int* dims,
539 const int* index, const int num_axis,
540 const int* axis) {
541 if (num_dims == 0) {
542 return 0;
543 }
544 TFLITE_DCHECK(dims != nullptr);
545 TFLITE_DCHECK(index != nullptr);
546 size_t offset = 0;
547 for (int idx = 0; idx < num_dims; ++idx) {
548 // if we need to skip this axis
549 bool is_axis = false;
550 if (axis != nullptr) {
551 for (int axis_idx = 0; axis_idx < num_axis; ++axis_idx) {
552 if (idx == axis[axis_idx]) {
553 is_axis = true;
554 break;
555 }
556 }
557 }
558 if (!is_axis) {
559 offset = offset * static_cast<size_t>(dims[idx]) +
560 static_cast<size_t>(index[idx]);
561 }
562 }
563 return offset;
564}
