% File: board.tex
% Define the custom TikZ command
\newcommand{\drawrectanglewithcircles}{
	\begin{tikzpicture}
		% Parameters
		\def\rectwidth{10} % width of the rectangle
		\def\rectheight{5} % height of the rectangle
		\def\circleRadius{0.1} % radius of the circles
		\def\numCircles{16} % number of circles in each row
		
		% Draw rectangle
		\draw (0, 0) rectangle (\rectwidth, \rectheight);
		
		% Draw circles in the four corners
		\foreach \x in {0, \rectwidth} {
			\foreach \y in {0, \rectheight} {
				\fill (\x, \y) circle (\circleRadius);
			}
		}
		
		% Draw circles in the upper middle part
		\foreach \i in {1,...,\numCircles} {
			\fill (\i * \rectwidth / (\numCircles + 1), \rectheight) circle (\circleRadius);
		}
		
		% Draw circles in the lower middle part
		\foreach \i in {1,...,\numCircles} {
			\fill (\i * \rectwidth / (\numCircles + 1), 0) circle (\circleRadius);
		}
	\end{tikzpicture}
}
